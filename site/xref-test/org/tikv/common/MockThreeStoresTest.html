18  package org.tikv.common;
19  
20  import com.google.common.collect.ImmutableList;
21  import com.google.protobuf.ByteString;
22  import java.io.IOException;
23  import java.net.ServerSocket;
24  import java.util.ArrayList;
25  import java.util.List;
26  import java.util.stream.Collectors;
27  import org.junit.After;
28  import org.junit.Before;
29  import org.tikv.common.region.TiRegion;
30  import org.tikv.common.region.TiStore;
31  import org.tikv.kvproto.Metapb;
32  import org.tikv.kvproto.Pdpb;
33  import org.tikv.kvproto.Pdpb.GetAllStoresResponse;
34  
35  public class MockThreeStoresTest extends PDMockServerTest {
36  
37    protected TiRegion region;
38    protected List<KVMockServer> servers = new ArrayList<>();
39    protected List<Metapb.Store> stores;
40  
41    @Before
42    @Override
43    public void setup() throws IOException {
44      super.setup();
45  
46      int[] ports = new int[3];
47      for (int i = 0; i < ports.length; i++) {
48        try (ServerSocket s = new ServerSocket(0)) {
49          ports[i] = s.getLocalPort();
50        }
51      }
52  
53      ImmutableList<Metapb.Peer> peers =
54          ImmutableList.of(
55              Metapb.Peer.newBuilder().setId(0x1).setStoreId(0x1).build(),
56              Metapb.Peer.newBuilder().setId(0x2).setStoreId(0x2).build(),
57              Metapb.Peer.newBuilder().setId(0x3).setStoreId(0x3).build());
58  
59      Metapb.Region region =
60          Metapb.Region.newBuilder()
61              .setRegionEpoch(Metapb.RegionEpoch.newBuilder().setConfVer(1).setVersion(2))
62              .setId(0xff)
63              .setStartKey(ByteString.EMPTY)
64              .setEndKey(ByteString.EMPTY)
65              .addAllPeers(peers)
66              .build();
67  
68      stores =
69          ImmutableList.of(
70              Metapb.Store.newBuilder()
71                  .setAddress("127.0.0.1:" + ports[0])
72                  .setVersion(Version.API_V2)
73                  .setId(0x1)
74                  .build(),
75              Metapb.Store.newBuilder()
76                  .setAddress("127.0.0.1:" + ports[1])
77                  .setVersion(Version.API_V2)
78                  .setId(0x2)
79                  .build(),
80              Metapb.Store.newBuilder()
81                  .setAddress("127.0.0.1:" + ports[2])
82                  .setVersion(Version.API_V2)
83                  .setId(0x3)
84                  .build());
85  
86      for (PDMockServer server : pdServers) {
87        server.addGetRegionListener(
88            request ->
89                Pdpb.GetRegionResponse.newBuilder()
90                    .setLeader(peers.get(0))
91                    .setRegion(region)
92                    .build());
93        server.addGetStoreListener(
94            (request) -> {
95              int i = (int) request.getStoreId() - 1;
96              return Pdpb.GetStoreResponse.newBuilder().setStore(stores.get(i)).build();
97            });
98        server.addGetAllStoresListener(
99            request -> {
100             return GetAllStoresResponse.newBuilder().addAllStores(stores).build();
101           });
102     }
103 
104     this.region =
105         new TiRegion(
106             session.getConf(),
107             region,
108             region.getPeers(0),
109             region.getPeersList(),
110             stores.stream().map(TiStore::new).collect(Collectors.toList()));
111     for (int port : ports) {
112       KVMockServer server = new KVMockServer();
113       server.start(this.region, port);
114       servers.add(server);
115     }
116   }
117 
118   public void put(ByteString key, ByteString value) {
119     for (KVMockServer server : servers) {
120       server.put(key, value);
121     }
122   }
123 
124   public void remove(ByteString key, ByteString value) {
125     for (KVMockServer server : servers) {
126       server.remove(key);
127     }
128   }
129 
130   @After
131   public void tearDown() {
132     for (KVMockServer server : servers) {
133       server.stop();
134     }
135   }
136 }
