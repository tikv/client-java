28    @Test
29    public void fromStringTest() {
30      List<MyDecimalTestStruct> tests = new ArrayList<>();
31      tests.add(
32          new MyDecimalTestStruct(
33              "1111111111111111111111111.111111111111111111111111111111",
34              "1111111111111111111111111.111111111111111111111111111111",
35              65,
36              30));
37      tests.add(new MyDecimalTestStruct("12345", "12345", 5, 0));
38      tests.add(new MyDecimalTestStruct("12345.", "12345", 5, 0));
39      tests.add(new MyDecimalTestStruct("123.45", "123.45", 5, 2));
40      tests.add(new MyDecimalTestStruct("-123.45", "-123.45", 5, 2));
41      tests.add(new MyDecimalTestStruct(".00012345000098765", "0.00012345000098765", 17, 17));
42      tests.add(new MyDecimalTestStruct(".12345000098765", "0.12345000098765", 14, 14));
43      tests.add(
44          new MyDecimalTestStruct("-.000000012345000098765", "-0.000000012345000098765", 21, 21));
45      tests.add(
46          new MyDecimalTestStruct("-.000000012345000098765", "-0.000000012345000098765", 2, 21));
47      tests.add(
48          new MyDecimalTestStruct("-.000000012345000098765", "-0.000000012345000098765", 21, 2));
49      tests.add(new MyDecimalTestStruct("0000000.001", "0.001", 3, 3));
50      tests.add(new MyDecimalTestStruct("1234500009876.5", "1234500009876.5", 14, 1));
51      for (MyDecimalTestStruct t : tests) {
52        MyDecimal dec = new MyDecimal();
53        dec.fromString(t.in);
54        assertEquals(t.out, dec.toString());
55      }
56    }

73    @Test
74    public void toBinFromBinTest() {
75      List<MyDecimalTestStruct> tests = new ArrayList<>();
76      String decValStr = "11111111111111111111111111111111111.111111111111111111111111111111";
77      tests.add(new MyDecimalTestStruct(decValStr, decValStr, 65, 30));
78      tests.add(new MyDecimalTestStruct("12345000098765", "12345000098765", 14, 0));
79      tests.add(new MyDecimalTestStruct("-10.55", "-10.55", 4, 2));
80      tests.add(new MyDecimalTestStruct("12345", "12345", 5, 0));
81      tests.add(new MyDecimalTestStruct("-12345", "-12345", 5, 0));
82      tests.add(new MyDecimalTestStruct("0000000.001", "0.001", 3, 3));
83      tests.add(new MyDecimalTestStruct("0.00012345000098765", "0.00012345000098765", 17, 17));
84      tests.add(new MyDecimalTestStruct("-0.00012345000098765", "-0.00012345000098765", 17, 17));
85      tests.add(new MyDecimalTestStruct("0", "0", 1, 0));
86      for (MyDecimalTestStruct a : tests) {
87        MyDecimal dec = new MyDecimal();
88        dec.fromString(a.in);
89        assertEquals(a.out, dec.toString());
90        int[] bin = dec.toBin(a.precision, a.frac);
91        dec.clear();
92        dec.fromBin(a.precision, a.frac, bin);
93        assertEquals(a.out, dec.toString());
94      }
95    }

97    @Test
98    public void toBinTest() {
99      MyDecimal dec = new MyDecimal();
100     dec.fromDecimal(-1234567890.1234);
101     int[] data = dec.toBin(dec.precision(), dec.frac());
102     int[] expected =
103         new int[] {
104           0x7E, 0xF2, 0x04, 0xC7, 0x2D, 0xFB, 0x2D,
105         };
106     assertArrayEquals(expected, data);
107   }

121     }
122   }
