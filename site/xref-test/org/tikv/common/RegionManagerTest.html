51    @Test
52    public void testRegionBorder() {
53      RangeMap<Key, Long> map = TreeRangeMap.create();
54  
55      map.put(KeyRangeUtils.makeRange(ByteString.EMPTY, ByteString.copyFromUtf8("abc")), 1L);
56      map.put(KeyRangeUtils.makeRange(ByteString.copyFromUtf8("abc"), ByteString.EMPTY), 2L);
57      assert map.get(Key.toRawKey(ByteString.EMPTY)) == null;
58      assert map.get(Key.toRawKey(ByteString.EMPTY, true)) == 1L;
59    }

61    @Test
62    public void getRegionByKey() {
63      ByteString startKey = ByteString.copyFrom(new byte[] {1});
64      ByteString endKey = ByteString.copyFrom(new byte[] {10});
65      ByteString searchKey = ByteString.copyFrom(new byte[] {5});
66      int confVer = 1026;
67      int ver = 1027;
68      long regionId = 233;
69      String testAddress = "127.0.0.1";
70      leader.addGetRegionListener(
71          request ->
72              GrpcUtils.makeGetRegionResponse(
73                  leader.getClusterId(),
74                  GrpcUtils.makeRegion(
75                      regionId,
76                      GrpcUtils.encodeKey(startKey.toByteArray()),
77                      GrpcUtils.encodeKey(endKey.toByteArray()),
78                      GrpcUtils.makeRegionEpoch(confVer, ver),
79                      GrpcUtils.makePeer(1, 10),
80                      GrpcUtils.makePeer(2, 20))));
81  
82      AtomicInteger i = new AtomicInteger(0);
83      long[] ids = new long[] {10, 20};
84      leader.addGetStoreListener(
85          (request ->
86              GrpcUtils.makeGetStoreResponse(
87                  leader.getClusterId(),
88                  GrpcUtils.makeStore(
89                      ids[i.getAndIncrement()],
90                      testAddress,
91                      StoreState.Up,
92                      GrpcUtils.makeStoreLabel("k1", "v1"),
93                      GrpcUtils.makeStoreLabel("k2", "v2")))));
94  
95      TiRegion region = mgr.getRegionByKey(startKey);
96      assertEquals(region.getId(), regionId);
97  
98      TiRegion regionToSearch = mgr.getRegionByKey(searchKey);
99      assertEquals(region, regionToSearch);
100   }

144     long storeId = 234;

146     leader.addGetStoreListener(
147         request ->
148             GrpcUtils.makeGetStoreResponse(
149                 leader.getClusterId(),
150                 GrpcUtils.makeStore(
151                     storeId,
152                     testAddress,
153                     Metapb.StoreState.Up,
154                     GrpcUtils.makeStoreLabel("k1", "v1"),
155                     GrpcUtils.makeStoreLabel("k2", "v2"))));
156     TiStore store = mgr.getStoreById(storeId);
157     assertEquals(store.getStore().getId(), storeId);
158 
159     leader.addGetStoreListener(
160         request ->
161             GrpcUtils.makeGetStoreResponse(
162                 leader.getClusterId(),
163                 GrpcUtils.makeStore(
164                     storeId + 1,
165                     testAddress,
166                     StoreState.Tombstone,
167                     GrpcUtils.makeStoreLabel("k1", "v1"),
168                     GrpcUtils.makeStoreLabel("k2", "v2"))));
169 
170     try {
171       mgr.getStoreById(storeId + 1);
172       fail();
173     } catch (Exception ignored) {
174     }
175 
176     mgr.invalidateStore(storeId);
177     try {
178       mgr.getStoreById(storeId);
179       fail();
180     } catch (Exception ignored) {
181     }

184   @Test
185   public void getRegionStorePairByKeyWithTiFlash() {
186 
187     ByteString startKey = ByteString.copyFrom(new byte[] {1});
188     ByteString endKey = ByteString.copyFrom(new byte[] {10});
189     ByteString searchKey = ByteString.copyFrom(new byte[] {5});
190     String testAddress = "testAddress";
191     long firstStoreId = 233;
192     long secondStoreId = 234;
193     int confVer = 1026;
194     int ver = 1027;
195     long regionId = 233;
196     leader.addGetRegionListener(
197         request ->
198             GrpcUtils.makeGetRegionResponse(
199                 leader.getClusterId(),
200                 GrpcUtils.makeRegion(
201                     regionId,
202                     GrpcUtils.encodeKey(startKey.toByteArray()),
203                     GrpcUtils.encodeKey(endKey.toByteArray()),
204                     GrpcUtils.makeRegionEpoch(confVer, ver),
205                     GrpcUtils.makeLearnerPeer(1, firstStoreId),
206                     GrpcUtils.makeLearnerPeer(2, secondStoreId))));
207 
208     AtomicInteger i = new AtomicInteger(0);
209     long[] ids = new long[] {firstStoreId, secondStoreId};
210     leader.addGetStoreListener(
211         (request ->
212             GrpcUtils.makeGetStoreResponse(
213                 leader.getClusterId(),
214                 GrpcUtils.makeStore(
215                     ids[i.getAndIncrement()],
216                     testAddress,
217                     StoreState.Up,
218                     GrpcUtils.makeStoreLabel("engine", "tiflash"),
219                     GrpcUtils.makeStoreLabel("k1", "v1"),
220                     GrpcUtils.makeStoreLabel("k2", "v2")))));
221 
222     Pair<TiRegion, TiStore> pair = mgr.getRegionStorePairByKey(searchKey, TiStoreType.TiFlash);
223     assertEquals(pair.first.getId(), regionId);
224     assertEquals(pair.second.getId(), firstStoreId);
225 
226     Pair<TiRegion, TiStore> secondPair =
227         mgr.getRegionStorePairByKey(searchKey, TiStoreType.TiFlash);
228     assertEquals(secondPair.first.getId(), regionId);
229     assertEquals(secondPair.second.getId(), secondStoreId);
230   }
