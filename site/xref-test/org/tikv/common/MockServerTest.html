18  package org.tikv.common;
19  
20  import com.google.common.collect.ImmutableList;
21  import com.google.protobuf.ByteString;
22  import java.io.IOException;
23  import java.util.List;
24  import java.util.stream.Collectors;
25  import org.junit.Before;
26  import org.tikv.common.region.TiRegion;
27  import org.tikv.common.region.TiStore;
28  import org.tikv.kvproto.Metapb;
29  import org.tikv.kvproto.Pdpb;
30  
31  public class MockServerTest extends PDMockServerTest {
32  
33    public KVMockServer server;
34    public int port;
35    public TiRegion region;
36  
37    @Before
38    @Override
39    public void setup() throws IOException {
40      super.setup();
41  
42      port = GrpcUtils.getFreePort();
43  
44      Metapb.Region r =
45          Metapb.Region.newBuilder()
46              .setRegionEpoch(Metapb.RegionEpoch.newBuilder().setConfVer(1).setVersion(2))
47              .setId(233)
48              .setStartKey(ByteString.EMPTY)
49              .setEndKey(ByteString.EMPTY)
50              .addPeers(Metapb.Peer.newBuilder().setId(11).setStoreId(13))
51              .build();
52  
53      List<Metapb.Store> s =
54          ImmutableList.of(
55              Metapb.Store.newBuilder()
56                  .setAddress(LOCAL_ADDR + ":" + port)
57                  .setVersion("5.0.0")
58                  .setId(13)
59                  .build());
60  
61      region =
62          new TiRegion(
63              session.getConf(),
64              r,
65              r.getPeers(0),
66              r.getPeersList(),
67              s.stream().map(TiStore::new).collect(Collectors.toList()));
68      leader.addGetRegionListener(
69          request -> Pdpb.GetRegionResponse.newBuilder().setRegion(r).build());
70      for (Metapb.Store store : s) {
71        leader.addGetStoreListener(
72            (request) -> Pdpb.GetStoreResponse.newBuilder().setStore(store).build());
73      }
74      server = new KVMockServer();
75      server.start(region, port);
76    }
77  }
