86    @Test
87    public void testGetRegionByKey() throws Exception {
88      byte[] startKey = new byte[] {1, 0, 2, 4};
89      byte[] endKey = new byte[] {1, 0, 2, 5};
90      int confVer = 1026;
91      int ver = 1027;
92      leader.addGetRegionListener(
93          request ->
94              GrpcUtils.makeGetRegionResponse(
95                  leader.getClusterId(),
96                  GrpcUtils.makeRegion(
97                      1,
98                      ByteString.copyFrom(startKey),
99                      ByteString.copyFrom(endKey),
100                     GrpcUtils.makeRegionEpoch(confVer, ver),
101                     GrpcUtils.makePeer(1, 10),
102                     GrpcUtils.makePeer(2, 20))));
103     try (PDClient client = session.getPDClient()) {
104       Pair<Metapb.Region, Metapb.Peer> rl =
105           client.getRegionByKey(defaultBackOff(), ByteString.EMPTY);
106       Metapb.Region r = rl.first;
107       Metapb.Peer l = rl.second;
108       assertEquals(r.getStartKey(), ByteString.copyFrom(startKey));
109       assertEquals(r.getEndKey(), ByteString.copyFrom(endKey));
110       assertEquals(r.getRegionEpoch().getConfVer(), confVer);
111       assertEquals(r.getRegionEpoch().getVersion(), ver);
112       assertEquals(1, l.getId());
113       assertEquals(10, l.getStoreId());
114     }
115   }

148   @Test
149   public void testGetStore() throws Exception {
150     long storeId = 1;
151     String testAddress = "testAddress";
152     leader.addGetStoreListener(
153         request ->
154             GrpcUtils.makeGetStoreResponse(
155                 leader.getClusterId(),
156                 GrpcUtils.makeStore(
157                     storeId,
158                     testAddress,
159                     Metapb.StoreState.Up,
160                     GrpcUtils.makeStoreLabel("k1", "v1"),
161                     GrpcUtils.makeStoreLabel("k2", "v2"))));
162     try (PDClient client = session.getPDClient()) {
163       Store r = client.getStore(defaultBackOff(), storeId);
164       assertEquals(storeId, r.getId());
165       assertEquals(testAddress, r.getAddress());
166       assertEquals(Metapb.StoreState.Up, r.getState());
167       assertEquals("k1", r.getLabels(0).getKey());
168       assertEquals("k2", r.getLabels(1).getKey());
169       assertEquals("v1", r.getLabels(0).getValue());
170       assertEquals("v2", r.getLabels(1).getValue());
171 
172       leader.addGetStoreListener(
173           request ->
174               GrpcUtils.makeGetStoreResponse(
175                   leader.getClusterId(),
176                   GrpcUtils.makeStore(storeId, testAddress, Metapb.StoreState.Tombstone)));
177       assertEquals(StoreState.Tombstone, client.getStore(defaultBackOff(), storeId).getState());
178     }
179   }

185   @Test
186   public void testRetryPolicy() throws Exception {
187     long storeId = 1024;
188     ExecutorService service = Executors.newCachedThreadPool();
189     AtomicInteger i = new AtomicInteger();
190     leader.addGetStoreListener(
191         request -> {
192           if (i.getAndIncrement() < 2) {
193             return null;
194           } else {
195             return GrpcUtils.makeGetStoreResponse(
196                 leader.getClusterId(), GrpcUtils.makeStore(storeId, "", Metapb.StoreState.Up));
197           }
198         });
199     try (PDClient client = session.getPDClient()) {
200       Callable<Store> storeCallable =
201           () -> client.getStore(ConcreteBackOffer.newCustomBackOff(5000), 0);
202       Future<Store> storeFuture = service.submit(storeCallable);
203       try {
204         Store r = storeFuture.get(50, TimeUnit.SECONDS);
205         assertEquals(r.getId(), storeId);
206       } catch (TimeoutException e) {
207         fail();
208       }
209 
210       // Should fail
211       AtomicInteger j = new AtomicInteger();
212       leader.addGetStoreListener(
213           request -> {
214             if (j.getAndIncrement() < 6) {
215               return null;
216             } else {
217               return GrpcUtils.makeGetStoreResponse(
218                   leader.getClusterId(), GrpcUtils.makeStore(storeId, "", Metapb.StoreState.Up));
219             }
220           });
221 
222       try {
223         client.getStore(defaultBackOff(), 0);
224       } catch (GrpcException e) {
225         assertTrue(true);
226         return;
227       }
228       fail();
229     }
