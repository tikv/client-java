42  public class TXNTest extends BaseTxnKVTest {
43    static final int DEFAULT_TTL = 10;
44    private TiSession session;
45    RegionStoreClient.RegionStoreClientBuilder builder;
46  
47    @Before
48    public void setUp() {
49      TiConfiguration conf = createTiConfiguration();
50      try {
51        session = TiSession.create(conf);
52        this.builder = session.getRegionStoreClientBuilder();
53      } catch (Exception e) {
54        fail("TiDB cluster may not be present");
55      }
56    }
57  
58    @After
59    public void tearDown() throws Exception {
60      if (session != null) {
61        session.close();
62      }
63    }
64  
65    void putKV(String key, String value) {
66      long startTS = session.getTimestamp().getVersion();
67      long commitTS = session.getTimestamp().getVersion();
68      putKV(key, value, startTS, commitTS);
69    }
70  
71    void putKV(String key, String value, long startTS, long commitTS) {
72      Kvrpcpb.Mutation m =
73          Kvrpcpb.Mutation.newBuilder()
74              .setKey(ByteString.copyFromUtf8(key))
75              .setOp(Kvrpcpb.Op.Put)
76              .setValue(ByteString.copyFromUtf8(value))
77              .build();
78  
79      boolean res = prewriteString(Collections.singletonList(m), startTS, key, DEFAULT_TTL);
80      assertTrue(res);
81      res = commitString(Collections.singletonList(key), startTS, commitTS);
82      assertTrue(res);
83    }
84  
85    boolean prewriteString(List<Kvrpcpb.Mutation> mutations, long startTS, String primary, long ttl) {
86      return prewrite(mutations, startTS, ByteString.copyFromUtf8(primary), ttl);
87    }
88  
89    boolean prewrite(List<Kvrpcpb.Mutation> mutations, long startTS, ByteString primary, long ttl) {
90      if (mutations.size() == 0) return true;
91      BackOffer backOffer = ConcreteBackOffer.newCustomBackOff(1000);
92  
93      for (Kvrpcpb.Mutation m : mutations) {
94        while (true) {
95          try {
96            TiRegion region = session.getRegionManager().getRegionByKey(m.getKey());
97            RegionStoreClient client = builder.build(region);
98            client.prewrite(backOffer, primary, Collections.singletonList(m), startTS, ttl, false);
99            break;
100         } catch (RegionException e) {
101           backOffer.doBackOff(BackOffFunction.BackOffFuncType.BoRegionMiss, e);
102         }
103       }
104     }
105     return true;
106   }
107 
108   boolean commitString(List<String> keys, long startTS, long commitTS) {
109     return commit(
110         keys.stream().map(ByteString::copyFromUtf8).collect(Collectors.toList()),
111         startTS,
112         commitTS);
113   }
114 
115   boolean commit(List<ByteString> keys, long startTS, long commitTS) {
116     if (keys.size() == 0) return true;
117     BackOffer backOffer = ConcreteBackOffer.newCustomBackOff(1000);
118 
119     for (ByteString byteStringK : keys) {
120       while (true) {
121         try {
122           TiRegion tiRegion = session.getRegionManager().getRegionByKey(byteStringK);
123           RegionStoreClient client = builder.build(tiRegion);
124           client.commit(backOffer, Collections.singletonList(byteStringK), startTS, commitTS);
125           break;
126         } catch (RegionException e) {
127           backOffer.doBackOff(BackOffFunction.BackOffFuncType.BoRegionMiss, e);
128         }
129       }
130     }
131     return true;
132   }
133 
134   String genRandomKey(int strLength) {
135     Random rnd = ThreadLocalRandom.current();
136     String prefix = rnd.nextInt(2) % 2 == 0 ? "a-test-" : "z-test-";
137     StringBuilder ret = new StringBuilder(prefix);
138     for (int i = 0; i < strLength; i++) {
139       boolean isChar = (rnd.nextInt(2) % 2 == 0);
140       if (isChar) {
141         int choice = rnd.nextInt(2) % 2 == 0 ? 65 : 97;
142         ret.append((char) (choice + rnd.nextInt(26)));
143       } else {
144         ret.append(rnd.nextInt(10));
145       }
146     }
147     return ret.toString();
148   }
