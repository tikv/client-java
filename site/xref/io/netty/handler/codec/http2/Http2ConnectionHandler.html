166   /**
167    * Handles the client-side (cleartext) upgrade from HTTP to HTTP/2. Reserves local stream 1 for
168    * the HTTP/2 response.
169    */
170   public void onHttpClientUpgrade() throws Http2Exception {
171     if (connection().isServer()) {
172       throw connectionError(PROTOCOL_ERROR, "Client-side HTTP upgrade requested for a server");
173     }
174     if (!prefaceSent()) {
175       // If the preface was not sent yet it most likely means the handler was not added to the
176       // pipeline before
177       // calling this method.
178       throw connectionError(INTERNAL_ERROR, "HTTP upgrade must occur after preface was sent");
179     }
180     if (decoder.prefaceReceived()) {
181       throw connectionError(
182           PROTOCOL_ERROR, "HTTP upgrade must occur before HTTP/2 preface is received");
183     }
184 
185     // Create a local stream used for the HTTP cleartext upgrade.
186     connection().local().createStream(HTTP_UPGRADE_STREAM_ID, true);
187   }
188 
189   /**
190    * Handles the server-side (cleartext) upgrade from HTTP to HTTP/2.
191    *
192    * @param settings the settings for the remote endpoint.
193    */
194   public void onHttpServerUpgrade(Http2Settings settings) throws Http2Exception {
195     if (!connection().isServer()) {
196       throw connectionError(PROTOCOL_ERROR, "Server-side HTTP upgrade requested for a client");
197     }
198     if (!prefaceSent()) {
199       // If the preface was not sent yet it most likely means the handler was not added to the
200       // pipeline before
201       // calling this method.
202       throw connectionError(INTERNAL_ERROR, "HTTP upgrade must occur after preface was sent");
203     }
204     if (decoder.prefaceReceived()) {
205       throw connectionError(
206           PROTOCOL_ERROR, "HTTP upgrade must occur before HTTP/2 preface is received");
207     }
208 
209     // Apply the settings but no ACK is necessary.
210     encoder.remoteSettings(settings);
211 
212     // Create a stream in the half-closed state.
213     connection().remote().createStream(HTTP_UPGRADE_STREAM_ID, true);
214   }

216   @Override
217   public void flush(ChannelHandlerContext ctx) {
218     try {
219       // Trigger pending writes in the remote flow controller.
220       Histogram.Timer writeTimer = flushFlowControlWriteDuration.startTimer();
221       encoder.flowController().writePendingBytes();
222       writeTimer.observeDuration();
223 
224       Histogram.Timer flushTimer = flushCtxFlushDuration.startTimer();
225       ctx.flush();
226       flushTimer.observeDuration();
227     } catch (Http2Exception e) {
228       onError(ctx, true, e);
229     } catch (Throwable cause) {
230       onError(ctx, true, connectionError(INTERNAL_ERROR, cause, "Error flushing"));
231     }
232   }

262     PrefaceDecoder(ChannelHandlerContext ctx) throws Exception {
263       clientPrefaceString = clientPrefaceString(encoder.connection());
264       // This handler was just added to the context. In case it was handled after
265       // the connection became active, send the connection preface now.

274     @Override
275     public void decode(ChannelHandlerContext ctx, ByteBuf in, List<Object> out) throws Exception {
276       try {
277         if (ctx.channel().isActive()
278             && readClientPrefaceString(in)
279             && verifyFirstFrameIsSettings(in)) {
280           // After the preface is read, it is time to hand over control to the post initialized
281           // decoder.
282           byteDecoder = new FrameDecoder();
283           byteDecoder.decode(ctx, in, out);
284         }
285       } catch (Throwable e) {
286         onError(ctx, false, e);
287       }
288     }

292       // The channel just became active - send the connection preface to the remote endpoint.
293       sendPreface(ctx);

308     /** Releases the {@code clientPrefaceString}. Any active streams will be left in the open. */
309     private void cleanup() {

311         clientPrefaceString.release();
312         clientPrefaceString = null;
313       }
314     }

316     /**
317      * Decodes the client connection preface string from the input buffer.
318      *
319      * @return {@code true} if processing of the client preface string is complete. Since client
320      *     preface strings can only be received by servers, returns true immediately for client
321      *     endpoints.
322      */
323     private boolean readClientPrefaceString(ByteBuf in) throws Http2Exception {
324       if (clientPrefaceString == null) {
325         return true;
326       }
327 
328       int prefaceRemaining = clientPrefaceString.readableBytes();
329       int bytesRead = min(in.readableBytes(), prefaceRemaining);
330 
331       // If the input so far doesn't match the preface, break the connection.
332       if (bytesRead == 0
333           || !ByteBufUtil.equals(
334               in,
335               in.readerIndex(),
336               clientPrefaceString,
337               clientPrefaceString.readerIndex(),
338               bytesRead)) {
339         int maxSearch = 1024; // picked because 512 is too little, and 2048 too much
340         int http1Index =
341             ByteBufUtil.indexOf(
342                 HTTP_1_X_BUF, in.slice(in.readerIndex(), min(in.readableBytes(), maxSearch)));
343         if (http1Index != -1) {
344           String chunk =
345               in.toString(in.readerIndex(), http1Index - in.readerIndex(), CharsetUtil.US_ASCII);
346           throw connectionError(PROTOCOL_ERROR, "Unexpected HTTP/1.x request: %s", chunk);
347         }
348         String receivedBytes =
349             hexDump(
350                 in, in.readerIndex(), min(in.readableBytes(), clientPrefaceString.readableBytes()));
351         throw connectionError(
352             PROTOCOL_ERROR,
353             "HTTP/2 client preface string missing or corrupt. " + "Hex dump for received bytes: %s",
354             receivedBytes);
355       }
356       in.skipBytes(bytesRead);
357       clientPrefaceString.skipBytes(bytesRead);
358 
359       if (!clientPrefaceString.isReadable()) {
360         // Entire preface has been read.
361         clientPrefaceString.release();
362         clientPrefaceString = null;
363         return true;
364       }
365       return false;
366     }
367 
368     /**
369      * Peeks at that the next frame in the buffer and verifies that it is a non-ack {@code SETTINGS}
370      * frame.
371      *
372      * @param in the inbound buffer.
373      * @return {@code true} if the next frame is a non-ack {@code SETTINGS} frame, {@code false} if
374      *     more data is required before we can determine the next frame type.
375      * @throws Http2Exception thrown if the next frame is NOT a non-ack {@code SETTINGS} frame.
376      */
377     private boolean verifyFirstFrameIsSettings(ByteBuf in) throws Http2Exception {
378       if (in.readableBytes() < 5) {
379         // Need more data before we can see the frame type for the first frame.
380         return false;
381       }
382 
383       short frameType = in.getUnsignedByte(in.readerIndex() + 3);
384       short flags = in.getUnsignedByte(in.readerIndex() + 4);
385       if (frameType != SETTINGS || (flags & Http2Flags.ACK) != 0) {
386         throw connectionError(
387             PROTOCOL_ERROR,
388             "First received frame was not SETTINGS. " + "Hex dump for first 5 bytes: %s",
389             hexDump(in, in.readerIndex(), 5));
390       }
391       return true;
392     }

394     /**
395      * Sends the HTTP/2 connection preface upon establishment of the connection, if not already
396      * sent.
397      */
398     private void sendPreface(ChannelHandlerContext ctx) throws Exception {
399       if (prefaceSent || !ctx.channel().isActive()) {
400         return;
401       }
402 
403       prefaceSent = true;
404 
405       final boolean isClient = !connection().isServer();
406       if (isClient) {
407         // Clients must send the preface string as the first bytes on the connection.
408         ctx.write(connectionPrefaceBuf()).addListener(ChannelFutureListener.CLOSE_ON_FAILURE);
409       }
410 
411       // Both client and server must send their initial settings.
412       encoder
413           .writeSettings(ctx, initialSettings, ctx.newPromise())
414           .addListener(ChannelFutureListener.CLOSE_ON_FAILURE);
415 
416       if (isClient) {
417         // If this handler is extended by the user and we directly fire the userEvent from this
418         // context then
419         // the user will not see the event. We should fire the event starting with this handler so
420         // this class
421         // (and extending classes) have a chance to process the event.
422         userEventTriggered(ctx, Http2ConnectionPrefaceAndSettingsFrameWrittenEvent.INSTANCE);
423       }
424     }
425   }

477     // Writability is expected to change while we are writing. We cannot allow this event to trigger

517   @Override
518   public void close(ChannelHandlerContext ctx, ChannelPromise promise) throws Exception {
519     if (decoupleCloseAndGoAway) {
520       ctx.close(promise);
521       return;
522     }
523     promise = promise.unvoid();
524     // Avoid NotYetConnectedException
525     if (!ctx.channel().isActive()) {
526       ctx.close(promise);
527       return;
528     }
529 
530     // If the user has already sent a GO_AWAY frame they may be attempting to do a graceful shutdown
531     // which requires
532     // sending multiple GO_AWAY frames. We should only send a GO_AWAY here if one has not already
533     // been sent. If
534     // a GO_AWAY has been sent we send a empty buffer just so we can wait to close until all other
535     // data has been
536     // flushed to the OS.
537     // https://github.com/netty/netty/issues/5307
538     ChannelFuture f =
539         connection().goAwaySent() ? ctx.write(EMPTY_BUFFER) : goAway(ctx, null, ctx.newPromise());
540     ctx.flush();
541     doGracefulShutdown(ctx, f, promise);
542   }

544   private ChannelFutureListener newClosingChannelFutureListener(
545       ChannelHandlerContext ctx, ChannelPromise promise) {
546     long gracefulShutdownTimeoutMillis = this.gracefulShutdownTimeoutMillis;
547     return gracefulShutdownTimeoutMillis < 0
548         ? new ClosingChannelFutureListener(ctx, promise)
549         : new ClosingChannelFutureListener(
550             ctx, promise, gracefulShutdownTimeoutMillis, MILLISECONDS);
551   }

553   private void doGracefulShutdown(
554       ChannelHandlerContext ctx, ChannelFuture future, final ChannelPromise promise) {
555     final ChannelFutureListener listener = newClosingChannelFutureListener(ctx, promise);
556     if (isGracefulShutdownComplete()) {
557       // If there are no active streams, close immediately after the GO_AWAY write completes or the
558       // timeout
559       // elapsed.
560       future.addListener(listener);
561     } else {
562       // If there are active streams we should wait until they are all closed before closing the
563       // connection.
564 
565       // The ClosingChannelFutureListener will cascade promise completion. We need to always notify
566       // the
567       // new ClosingChannelFutureListener when the graceful close completes if the promise is not
568       // null.
569       if (closeListener == null) {
570         closeListener = listener;
571       } else if (promise != null) {
572         final ChannelFutureListener oldCloseListener = closeListener;
573         closeListener =
574             new ChannelFutureListener() {
575               @Override
576               public void operationComplete(ChannelFuture future) throws Exception {
577                 try {
578                   oldCloseListener.operationComplete(future);
579                 } finally {
580                   listener.operationComplete(future);
581                 }
582               }
583             };
584       }
585     }
586   }

604   @Override
605   public void channelReadComplete(ChannelHandlerContext ctx) throws Exception {
606     // Trigger flush after read on the assumption that flush is cheap if there is nothing to write
607     // and that
608     // for flow-control the read may release window that causes data to be written that can now be
609     // flushed.
610     try {
611       // First call channelReadComplete0(...) as this may produce more data that we want to flush
612       channelReadComplete0(ctx);
613     } finally {
614       flush(ctx);
615     }
616   }
617 
618   final void channelReadComplete0(ChannelHandlerContext ctx) {
619     // Discard bytes of the cumulation buffer if needed.
620     discardSomeReadBytes();
621 
622     // Ensure we never stale the HTTP/2 Channel. Flow-control is enforced by HTTP/2.
623     //
624     // See https://tools.ietf.org/html/rfc7540#section-5.2.2
625     if (!ctx.channel().config().isAutoRead()) {
626       ctx.read();
627     }
628 
629     ctx.fireChannelReadComplete();
630   }

686   @Override
687   public void closeStream(final Http2Stream stream, ChannelFuture future) {
688     stream.close();
689 
690     if (future.isDone()) {
691       checkCloseConnection(future);
692     } else {
693       future.addListener(
694           new ChannelFutureListener() {
695             @Override
696             public void operationComplete(ChannelFuture future) throws Exception {
697               checkCloseConnection(future);
698             }
699           });

703   /** Central handler for all exceptions caught during HTTP/2 processing. */
704   @Override
705   public void onError(ChannelHandlerContext ctx, boolean outbound, Throwable cause) {
706     Http2Exception embedded = getEmbeddedHttp2Exception(cause);
707     if (isStreamError(embedded)) {
708       onStreamError(ctx, outbound, cause, (StreamException) embedded);
709     } else if (embedded instanceof CompositeStreamException) {
710       CompositeStreamException compositException = (CompositeStreamException) embedded;
711       for (StreamException streamException : compositException) {
712         onStreamError(ctx, outbound, cause, streamException);
713       }
714     } else {
715       onConnectionError(ctx, outbound, cause, embedded);
716     }
717     ctx.flush();
718   }

720   /**
721    * Called by the graceful shutdown logic to determine when it is safe to close the connection.
722    * Returns {@code true} if the graceful shutdown has completed and the connection can be safely
723    * closed. This implementation just guarantees that there are no active streams. Subclasses may
724    * override to provide additional checks.
725    */
726   protected boolean isGracefulShutdownComplete() {
727     return connection().numActiveStreams() == 0;
728   }

730   /**
731    * Handler for a connection error. Sends a GO_AWAY frame to the remote endpoint. Once all streams
732    * are closed, the connection is shut down.
733    *
734    * @param ctx the channel context
735    * @param outbound {@code true} if the error was caused by an outbound operation.
736    * @param cause the exception that was caught
737    * @param http2Ex the {@link Http2Exception} that is embedded in the causality chain. This may be
738    *     {@code null} if it's an unknown exception.
739    */
740   protected void onConnectionError(
741       ChannelHandlerContext ctx, boolean outbound, Throwable cause, Http2Exception http2Ex) {
742     if (http2Ex == null) {
743       http2Ex = new Http2Exception(INTERNAL_ERROR, cause.getMessage(), cause);
744     }
745 
746     ChannelPromise promise = ctx.newPromise();
747     ChannelFuture future = goAway(ctx, http2Ex, ctx.newPromise());
748     if (http2Ex.shutdownHint() == Http2Exception.ShutdownHint.GRACEFUL_SHUTDOWN) {
749       doGracefulShutdown(ctx, future, promise);
750     } else {
751       future.addListener(newClosingChannelFutureListener(ctx, promise));
752     }
753   }

755   /**
756    * Handler for a stream error. Sends a {@code RST_STREAM} frame to the remote endpoint and closes
757    * the stream.
758    *
759    * @param ctx the channel context
760    * @param outbound {@code true} if the error was caused by an outbound operation.
761    * @param cause the exception that was caught
762    * @param http2Ex the {@link StreamException} that is embedded in the causality chain.
763    */
764   protected void onStreamError(
765       ChannelHandlerContext ctx,
766       boolean outbound,
767       @SuppressWarnings("unused") Throwable cause,
768       StreamException http2Ex) {
769     final int streamId = http2Ex.streamId();
770     Http2Stream stream = connection().stream(streamId);
771 
772     // if this is caused by reading headers that are too large, send a header with status 431
773     if (http2Ex instanceof Http2Exception.HeaderListSizeException
774         && ((Http2Exception.HeaderListSizeException) http2Ex).duringDecode()
775         && connection().isServer()) {
776 
777       // NOTE We have to check to make sure that a stream exists before we send our reply.
778       // We likely always create the stream below as the stream isn't created until the
779       // header block is completely processed.
780 
781       // The case of a streamId referring to a stream which was already closed is handled
782       // by createStream and will land us in the catch block below
783       if (stream == null) {
784         try {
785           stream = encoder.connection().remote().createStream(streamId, true);
786         } catch (Http2Exception e) {
787           resetUnknownStream(ctx, streamId, http2Ex.error().code(), ctx.newPromise());
788           return;
789         }
790       }
791 
792       // ensure that we have not already sent headers on this stream
793       if (stream != null && !stream.isHeadersSent()) {
794         try {
795           handleServerHeaderDecodeSizeError(ctx, stream);
796         } catch (Throwable cause2) {
797           onError(ctx, outbound, connectionError(INTERNAL_ERROR, cause2, "Error DecodeSizeError"));
798         }
799       }
800     }
801 
802     if (stream == null) {
803       if (!outbound || connection().local().mayHaveCreatedStream(streamId)) {
804         resetUnknownStream(ctx, streamId, http2Ex.error().code(), ctx.newPromise());
805       }
806     } else {
807       resetStream(ctx, stream, http2Ex.error().code(), ctx.newPromise());
808     }
809   }

812    * Notifies client that this server has received headers that are larger than what it is willing
813    * to accept. Override to change behavior.
814    *
815    * @param ctx the channel context

817    */
818   protected void handleServerHeaderDecodeSizeError(ChannelHandlerContext ctx, Http2Stream stream) {
819     encoder().writeHeaders(ctx, stream.id(), HEADERS_TOO_LARGE_HEADERS, 0, true, ctx.newPromise());

826   /**
827    * Sends a {@code RST_STREAM} frame even if we don't know about the stream. This error condition
828    * is most likely triggered by the first frame of a stream being invalid. That is, there was an
829    * error reading the frame before we could create a new stream.
830    */
831   private ChannelFuture resetUnknownStream(
832       final ChannelHandlerContext ctx, int streamId, long errorCode, ChannelPromise promise) {
833     ChannelFuture future = frameWriter().writeRstStream(ctx, streamId, errorCode, promise);
834     if (future.isDone()) {
835       closeConnectionOnError(ctx, future);
836     } else {
837       future.addListener(
838           new ChannelFutureListener() {
839             @Override
840             public void operationComplete(ChannelFuture future) throws Exception {
841               closeConnectionOnError(ctx, future);
842             }
843           });
844     }
845     return future;
846   }

848   @Override
849   public ChannelFuture resetStream(
850       final ChannelHandlerContext ctx, int streamId, long errorCode, ChannelPromise promise) {
851     final Http2Stream stream = connection().stream(streamId);
852     if (stream == null) {
853       return resetUnknownStream(ctx, streamId, errorCode, promise.unvoid());
854     }
855 
856     return resetStream(ctx, stream, errorCode, promise);
857   }
858 
859   private ChannelFuture resetStream(
860       final ChannelHandlerContext ctx,
861       final Http2Stream stream,
862       long errorCode,
863       ChannelPromise promise) {
864     promise = promise.unvoid();
865     if (stream.isResetSent()) {
866       // Don't write a RST_STREAM frame if we have already written one.
867       return promise.setSuccess();
868     }
869     // Synchronously set the resetSent flag to prevent any subsequent calls
870     // from resulting in multiple reset frames being sent.
871     //
872     // This needs to be done before we notify the promise as the promise may have a listener
873     // attached that
874     // call resetStream(...) again.
875     stream.resetSent();
876 
877     final ChannelFuture future;
878     // If the remote peer is not aware of the steam, then we are not allowed to send a RST_STREAM
879     // https://tools.ietf.org/html/rfc7540#section-6.4.
880     if (stream.state() == IDLE
881         || connection().local().created(stream)
882             && !stream.isHeadersSent()
883             && !stream.isPushPromiseSent()) {
884       future = promise.setSuccess();
885     } else {
886       future = frameWriter().writeRstStream(ctx, stream.id(), errorCode, promise);
887     }
888     if (future.isDone()) {
889       processRstStreamWriteResult(ctx, stream, future);
890     } else {
891       future.addListener(
892           new ChannelFutureListener() {
893             @Override
894             public void operationComplete(ChannelFuture future) throws Exception {
895               processRstStreamWriteResult(ctx, stream, future);
896             }
897           });
898     }
899 
900     return future;
901   }

903   @Override
904   public ChannelFuture goAway(
905       final ChannelHandlerContext ctx,
906       final int lastStreamId,
907       final long errorCode,
908       final ByteBuf debugData,
909       ChannelPromise promise) {
910     promise = promise.unvoid();
911     final Http2Connection connection = connection();
912     try {
913       if (!connection.goAwaySent(lastStreamId, errorCode, debugData)) {
914         debugData.release();
915         promise.trySuccess();
916         return promise;
917       }
918     } catch (Throwable cause) {
919       debugData.release();
920       promise.tryFailure(cause);
921       return promise;
922     }
923 
924     // Need to retain before we write the buffer because if we do it after the refCnt could already
925     // be 0 and
926     // result in an IllegalRefCountException.
927     debugData.retain();
928     ChannelFuture future =
929         frameWriter().writeGoAway(ctx, lastStreamId, errorCode, debugData, promise);
930 
931     if (future.isDone()) {
932       processGoAwayWriteResult(ctx, lastStreamId, errorCode, debugData, future);
933     } else {
934       future.addListener(
935           new ChannelFutureListener() {
936             @Override
937             public void operationComplete(ChannelFuture future) throws Exception {
938               processGoAwayWriteResult(ctx, lastStreamId, errorCode, debugData, future);
939             }
940           });
941     }
942 
943     return future;
944   }

946   /**
947    * Closes the connection if the graceful shutdown process has completed.
948    *
949    * @param future Represents the status that will be passed to the {@link #closeListener}.
950    */
951   private void checkCloseConnection(ChannelFuture future) {
952     // If this connection is closing and the graceful shutdown has completed, close the connection
953     // once this operation completes.
954     if (closeListener != null && isGracefulShutdownComplete()) {
955       ChannelFutureListener closeListener = this.closeListener;
956       // This method could be called multiple times
957       // and we don't want to notify the closeListener multiple times.
958       this.closeListener = null;
959       try {
960         closeListener.operationComplete(future);
961       } catch (Exception e) {
962         throw new IllegalStateException("Close listener threw an unexpected exception", e);
963       }
964     }
965   }

967   /**
968    * Close the remote endpoint with with a {@code GO_AWAY} frame. Does <strong>not</strong> flush
969    * immediately, this is the responsibility of the caller.
970    */
971   private ChannelFuture goAway(
972       ChannelHandlerContext ctx, Http2Exception cause, ChannelPromise promise) {
973     long errorCode = cause != null ? cause.error().code() : NO_ERROR.code();
974     int lastKnownStream = connection().remote().lastStreamCreated();
975     return goAway(ctx, lastKnownStream, errorCode, Http2CodecUtil.toByteBuf(ctx, cause), promise);
976   }

978   private void processRstStreamWriteResult(
979       ChannelHandlerContext ctx, Http2Stream stream, ChannelFuture future) {
980     if (future.isSuccess()) {
981       closeStream(stream, future);
982     } else {
983       // The connection will be closed and so no need to change the resetSent flag to false.
984       onConnectionError(ctx, true, future.cause(), null);
985     }
986   }
987 
988   private void closeConnectionOnError(ChannelHandlerContext ctx, ChannelFuture future) {
989     if (!future.isSuccess()) {
990       onConnectionError(ctx, true, future.cause(), null);
991     }
992   }

1002   private static void processGoAwayWriteResult(
1003       final ChannelHandlerContext ctx,
1004       final int lastStreamId,
1005       final long errorCode,
1006       final ByteBuf debugData,
1007       ChannelFuture future) {
1008     try {
1009       if (future.isSuccess()) {
1010         if (errorCode != NO_ERROR.code()) {
1011           if (logger.isDebugEnabled()) {
1012             logger.debug(
1013                 "{} Sent GOAWAY: lastStreamId '{}', errorCode '{}', "
1014                     + "debugData '{}'. Forcing shutdown of the connection.",
1015                 ctx.channel(),
1016                 lastStreamId,
1017                 errorCode,
1018                 debugData.toString(UTF_8),
1019                 future.cause());
1020           }
1021           ctx.close();
1022         }
1023       } else {
1024         if (logger.isDebugEnabled()) {
1025           logger.debug(
1026               "{} Sending GOAWAY failed: lastStreamId '{}', errorCode '{}', "
1027                   + "debugData '{}'. Forcing shutdown of the connection.",
1028               ctx.channel(),
1029               lastStreamId,
1030               errorCode,
1031               debugData.toString(UTF_8),
1032               future.cause());
1033         }
1034         ctx.close();
1035       }
1036     } finally {
1037       // We're done with the debug data now.
1038       debugData.release();
1039     }
1040   }

1042   /** Closes the channel when the future completes. */
1043   private static final class ClosingChannelFutureListener implements ChannelFutureListener {
1044     private final ChannelHandlerContext ctx;
1045     private final ChannelPromise promise;
1046     private final ScheduledFuture<?> timeoutTask;
1047     private boolean closed;
1048 
1049     ClosingChannelFutureListener(ChannelHandlerContext ctx, ChannelPromise promise) {
1050       this.ctx = ctx;
1051       this.promise = promise;
1052       timeoutTask = null;
1053     }
1054 
1055     ClosingChannelFutureListener(
1056         final ChannelHandlerContext ctx,
1057         final ChannelPromise promise,
1058         long timeout,
1059         TimeUnit unit) {
1060       this.ctx = ctx;
1061       this.promise = promise;
1062       timeoutTask =
1063           ctx.executor()
1064               .schedule(
1065                   new Runnable() {
1066                     @Override
1067                     public void run() {
1068                       doClose();
1069                     }
1070                   },
1071                   timeout,
1072                   unit);
1073     }
1074 
1075     @Override
1076     public void operationComplete(ChannelFuture sentGoAwayFuture) {
1077       if (timeoutTask != null) {
1078         timeoutTask.cancel(false);
1079       }
1080       doClose();
1081     }
1082 
1083     private void doClose() {
1084       // We need to guard against multiple calls as the timeout may trigger close() first and then
1085       // it will be
1086       // triggered again because of operationComplete(...) is called.
1087       if (closed) {
1088         // This only happens if we also scheduled a timeout task.
1089         assert timeoutTask != null;
1090         return;
1091       }
1092       closed = true;
1093       if (promise == null) {
1094         ctx.close();
1095       } else {
1096         ctx.close(promise);
1097       }
1098     }
1099   }
