78    public DefaultHttp2RemoteFlowController(
79        Http2Connection connection,
80        StreamByteDistributor streamByteDistributor,
81        final Listener listener) {
82      this.connection = checkNotNull(connection, "connection");
83      this.streamByteDistributor = checkNotNull(streamByteDistributor, "streamWriteDistributor");
84  
85      // Add a flow state for the connection.
86      stateKey = connection.newKey();
87      connectionState = new FlowState(connection.connectionStream());
88      connection.connectionStream().setProperty(stateKey, connectionState);
89  
90      // Monitor may depend upon connectionState, and so initialize after connectionState
91      listener(listener);
92      monitor.windowSize(connectionState, initialWindowSize);
93  
94      // Register for notification of new streams.
95      connection.addListener(
96          new Http2ConnectionAdapter() {
97            @Override
98            public void onStreamAdded(Http2Stream stream) {
99              // If the stream state is not open then the stream is not yet eligible for flow
100             // controlled frames and
101             // only requires the ReducedFlowState. Otherwise the full amount of memory is required.
102             stream.setProperty(stateKey, new FlowState(stream));
103           }
104 
105           @Override
106           public void onStreamActive(Http2Stream stream) {
107             // If the object was previously created, but later activated then we have to ensure the
108             // proper
109             // initialWindowSize is used.
110             monitor.windowSize(state(stream), initialWindowSize);
111           }
112 
113           @Override
114           public void onStreamClosed(Http2Stream stream) {
115             // Any pending frames can never be written, cancel and
116             // write errors for any pending frames.
117             state(stream).cancel(STREAM_CLOSED, null);
118           }
119 
120           @Override
121           public void onStreamHalfClosed(Http2Stream stream) {
122             if (HALF_CLOSED_LOCAL == stream.state()) {
123               /**
124                * When this method is called there should not be any pending frames left if the API
125                * is used correctly. However, it is possible that a erroneous application can sneak
126                * in a frame even after having already written a frame with the END_STREAM flag set,
127                * as the stream state might not transition immediately to HALF_CLOSED_LOCAL / CLOSED
128                * due to flow control delaying the write.
129                *
130                * <p>This is to cancel any such illegal writes.
131                */
132               state(stream).cancel(STREAM_CLOSED, null);
133             }
134           }
135         });
136   }

143   @Override
144   public void channelHandlerContext(ChannelHandlerContext ctx) throws Http2Exception {
145     this.ctx = checkNotNull(ctx, "ctx");
146 
147     // Writing the pending bytes will not check writability change and instead a writability change
148     // notification
149     // to be provided by an explicit call.

153     // streams will be
154     // closed and the queue cleanup will occur when the stream state transitions occur.
155 
156     // If any frames have been queued up, we should send them now that we have a channel context.
157     if (isChannelWritable()) {
158       writePendingBytes();
159     }

252   private int minUsableChannelBytes() {
253     // The current allocation algorithm values "fairness" and doesn't give any consideration to
254     // "goodput". It
255     // is possible that 1 byte will be allocated to many streams. In an effort to try to make
256     // "goodput"
257     // reasonable with the current allocation algorithm we have this "cheap" check up front to
258     // ensure there is
259     // an "adequate" amount of connection window before allocation is attempted. This is not
260     // foolproof as if the
261     // number of streams is >= this minimal number then we may still have the issue, but the idea is
262     // to narrow the
263     // circumstances in which this can happen without rewriting the allocation algorithm.
264     return max(ctx.channel().config().getWriteBufferLowWaterMark(), MIN_WRITABLE_CHUNK);
265   }
266 
267   private int maxUsableChannelBytes() {
268     // If the channel isWritable, allow at least minUsableChannelBytes.
269     int channelWritableBytes = (int) min(Integer.MAX_VALUE, ctx.channel().bytesBeforeUnwritable());
270     int usableBytes =
271         channelWritableBytes > 0 ? max(channelWritableBytes, minUsableChannelBytes()) : 0;
272 
273     // Clip the usable bytes by the connection window.
274     return min(connectionState.windowSize(), usableBytes);
275   }

290   /** The remote flow control state for a single stream. */
291   private final class FlowState implements StreamByteDistributor.StreamState {
292     private final Http2Stream stream;
293     private final Deque<FlowControlled> pendingWriteQueue;
294     private int window;
295     private long pendingBytes;
296     private boolean markedWritable;
297 
298     /** Set to true while a frame is being written, false otherwise. */
299     private boolean writing;
300     /** Set to true if cancel() was called. */
301     private boolean cancelled;
302 
303     FlowState(Http2Stream stream) {
304       this.stream = stream;
305       pendingWriteQueue = new ArrayDeque<FlowControlled>(2);
306     }
307 
308     /**
309      * Determine if the stream associated with this object is writable.
310      *
311      * @return {@code true} if the stream associated with this object is writable.
312      */
313     boolean isWritable() {
314       return windowSize() > pendingBytes() && !cancelled;
315     }
316 
317     /** The stream this state is associated with. */
318     @Override
319     public Http2Stream stream() {
320       return stream;
321     }
322 
323     /** Returns the parameter from the last call to {@link #markedWritability(boolean)}. */
324     boolean markedWritability() {
325       return markedWritable;
326     }
327 
328     /** Save the state of writability. */
329     void markedWritability(boolean isWritable) {
330       this.markedWritable = isWritable;
331     }
332 
333     @Override
334     public int windowSize() {
335       return window;
336     }
337 
338     /** Reset the window size for this stream. */
339     void windowSize(int initialWindowSize) {
340       window = initialWindowSize;
341     }
342 
343     /**
344      * Write the allocated bytes for this stream.
345      *
346      * @return the number of bytes written for a stream or {@code -1} if no write occurred.
347      */
348     int writeAllocatedBytes(int allocated) {
349       final int initialAllocated = allocated;
350       int writtenBytes;
351       // In case an exception is thrown we want to remember it and pass it to cancel(Throwable).
352       Throwable cause = null;
353       FlowControlled frame;
354       try {
355         assert !writing;
356         writing = true;
357 
358         // Write the remainder of frames that we are allowed to
359         boolean writeOccurred = false;
360         while (!cancelled && (frame = peek()) != null) {
361           int maxBytes = min(allocated, writableWindow());
362           if (maxBytes <= 0 && frame.size() > 0) {
363             // The frame still has data, but the amount of allocated bytes has been exhausted.
364             // Don't write needless empty frames.
365             break;
366           }
367           writeOccurred = true;
368           int initialFrameSize = frame.size();
369           try {
370             frame.write(ctx, max(0, maxBytes));
371             if (frame.size() == 0) {
372               // This frame has been fully written, remove this frame and notify it.
373               // Since we remove this frame first, we're guaranteed that its error
374               // method will not be called when we call cancel.
375               pendingWriteQueue.remove();
376               frame.writeComplete();
377             }
378           } finally {
379             // Decrement allocated by how much was actually written.
380             allocated -= initialFrameSize - frame.size();
381           }
382         }
383 
384         if (!writeOccurred) {
385           // Either there was no frame, or the amount of allocated bytes has been exhausted.
386           return -1;
387         }
388 
389       } catch (Throwable t) {
390         // Mark the state as cancelled, we'll clear the pending queue via cancel() below.
391         cancelled = true;
392         cause = t;
393       } finally {
394         writing = false;
395         // Make sure we always decrement the flow control windows
396         // by the bytes written.
397         writtenBytes = initialAllocated - allocated;
398 
399         decrementPendingBytes(writtenBytes, false);
400         decrementFlowControlWindow(writtenBytes);
401 
402         // If a cancellation occurred while writing, call cancel again to
403         // clear and error all of the pending writes.
404         if (cancelled) {
405           cancel(INTERNAL_ERROR, cause);
406         }
407       }
408       return writtenBytes;
409     }
410 
411     /**
412      * Increments the flow control window for this stream by the given delta and returns the new
413      * value.
414      */
415     int incrementStreamWindow(int delta) throws Http2Exception {
416       if (delta > 0 && Integer.MAX_VALUE - delta < window) {
417         throw streamError(
418             stream.id(), FLOW_CONTROL_ERROR, "Window size overflow for stream: %d", stream.id());
419       }
420       window += delta;
421 
422       streamByteDistributor.updateStreamableBytes(this);
423       return window;
424     }
425 
426     /** Returns the maximum writable window (minimum of the stream and connection windows). */
427     private int writableWindow() {
428       return min(window, connectionWindowSize());
429     }
430 
431     @Override
432     public long pendingBytes() {
433       return pendingBytes;
434     }
435 
436     /** Adds the {@code frame} to the pending queue and increments the pending byte count. */
437     void enqueueFrame(FlowControlled frame) {
438       FlowControlled last = pendingWriteQueue.peekLast();
439       if (last == null) {
440         enqueueFrameWithoutMerge(frame);
441         return;
442       }
443 
444       int lastSize = last.size();
445       if (last.merge(ctx, frame)) {
446         incrementPendingBytes(last.size() - lastSize, true);
447         return;
448       }
449       enqueueFrameWithoutMerge(frame);
450     }
451 
452     private void enqueueFrameWithoutMerge(FlowControlled frame) {
453       pendingWriteQueue.offer(frame);
454       // This must be called after adding to the queue in order so that hasFrame() is
455       // updated before updating the stream state.
456       incrementPendingBytes(frame.size(), true);
457     }
458 
459     @Override
460     public boolean hasFrame() {
461       return !pendingWriteQueue.isEmpty();
462     }
463 
464     /** Returns the head of the pending queue, or {@code null} if empty. */
465     private FlowControlled peek() {
466       return pendingWriteQueue.peek();
467     }
468 
469     /**
470      * Clears the pending queue and writes errors for each remaining frame.
471      *
472      * @param error the {@link Http2Error} to use.
473      * @param cause the {@link Throwable} that caused this method to be invoked.
474      */
475     void cancel(Http2Error error, Throwable cause) {
476       cancelled = true;
477       // Ensure that the queue can't be modified while we are writing.
478       if (writing) {
479         return;
480       }
481 
482       FlowControlled frame = pendingWriteQueue.poll();
483       if (frame != null) {
484         // Only create exception once and reuse to reduce overhead of filling in the stacktrace.
485         final Http2Exception exception =
486             streamError(stream.id(), error, cause, "Stream closed before write could take place");
487         do {
488           writeError(frame, exception);
489           frame = pendingWriteQueue.poll();
490         } while (frame != null);
491       }
492 
493       streamByteDistributor.updateStreamableBytes(this);
494 
495       monitor.stateCancelled(this);
496     }
497 
498     /**
499      * Increments the number of pending bytes for this node and optionally updates the {@link
500      * StreamByteDistributor}.
501      */
502     private void incrementPendingBytes(int numBytes, boolean updateStreamableBytes) {
503       pendingBytes += numBytes;
504       monitor.incrementPendingBytes(numBytes);
505       if (updateStreamableBytes) {
506         streamByteDistributor.updateStreamableBytes(this);
507       }
508     }
509 
510     /**
511      * If this frame is in the pending queue, decrements the number of pending bytes for the stream.
512      */
513     private void decrementPendingBytes(int bytes, boolean updateStreamableBytes) {
514       incrementPendingBytes(-bytes, updateStreamableBytes);
515     }
516 
517     /** Decrement the per stream and connection flow control window by {@code bytes}. */
518     private void decrementFlowControlWindow(int bytes) {
519       try {
520         int negativeBytes = -bytes;
521         connectionState.incrementStreamWindow(negativeBytes);
522         incrementStreamWindow(negativeBytes);
523       } catch (Http2Exception e) {
524         // Should never get here since we're decrementing.
525         throw new IllegalStateException(
526             "Invalid window state when writing frame: " + e.getMessage(), e);
527       }
528     }
529 
530     /**
531      * Discards this {@link FlowControlled}, writing an error. If this frame is in the pending
532      * queue, the unwritten bytes are removed from this branch of the priority tree.
533      */
534     private void writeError(FlowControlled frame, Http2Exception cause) {
535       assert ctx != null;
536       decrementPendingBytes(frame.size(), true);
537       frame.error(ctx, cause);
538     }
539   }

541   /** Abstract class which provides common functionality for writability monitor implementations. */
542   private class WritabilityMonitor implements StreamByteDistributor.Writer {
543     private boolean inWritePendingBytes;
544     private long totalPendingBytes;
545 
546     @Override
547     public final void write(Http2Stream stream, int numBytes) {
548       state(stream).writeAllocatedBytes(numBytes);
549     }

565     /**

568      * @param state the state to change the initial window size for.
569      * @param initialWindowSize the size of the window in bytes.
570      */
571     void windowSize(FlowState state, int initialWindowSize) {
572       state.windowSize(initialWindowSize);
573     }
574 
575     /**
576      * Increment the window size for a particular stream.
577      *
578      * @param state the state associated with the stream whose window is being incremented.
579      * @param delta The amount to increment by.
580      * @throws Http2Exception If this operation overflows the window for {@code state}.
581      */
582     void incrementWindowSize(FlowState state, int delta) throws Http2Exception {
583       state.incrementStreamWindow(delta);
584     }
585 
586     /**
587      * Add a frame to be sent via flow control.
588      *
589      * @param state The state associated with the stream which the {@code frame} is associated with.
590      * @param frame the frame to enqueue.
591      * @throws Http2Exception If a writability error occurs.
592      */
593     void enqueueFrame(FlowState state, FlowControlled frame) throws Http2Exception {
594       state.enqueueFrame(frame);
595     }
596 
597     /**
598      * Increment the total amount of pending bytes for all streams. When any stream's pending bytes
599      * changes method should be called.
600      *
601      * @param delta The amount to increment by.
602      */
603     final void incrementPendingBytes(int delta) {
604       totalPendingBytes += delta;
605 
606       // Notification of writibilty change should be delayed until the end of the top level event.
607       // This is to ensure the flow controller is more consistent state before calling external
608       // listener methods.
609     }
610 
611     /**
612      * Determine if the stream associated with {@code state} is writable.
613      *
614      * @param state The state which is associated with the stream to test writability for.
615      * @return {@code true} if {@link FlowState#stream()} is writable. {@code false} otherwise.
616      */
617     final boolean isWritable(FlowState state) {
618       return isWritableConnection() && state.isWritable();
619     }
620 
621     final void writePendingBytes() throws Http2Exception {
622       // Reentry is not permitted during the byte distribution process. It may lead to undesirable
623       // distribution of
624       // bytes and even infinite loops. We protect against reentry and make sure each call has an
625       // opportunity to
626       // cause a distribution to occur. This may be useful for example if the channel's writability
627       // changes from
628       // Writable -> Not Writable (because we are writing) -> Writable (because the user flushed to
629       // make more room
630       // in the channel outbound buffer).
631       if (inWritePendingBytes) {
632         return;
633       }
634       inWritePendingBytes = true;
635       try {
636         int bytesToWrite = writableBytes();
637         // Make sure we always write at least once, regardless if we have bytesToWrite or not.
638         // This ensures that zero-length frames will always be written.
639         for (; ; ) {
640           Histogram.Timer distributedTimer = byteDistributedDuration.startTimer();
641           boolean distributed = streamByteDistributor.distribute(bytesToWrite, this);
642           distributedTimer.observeDuration();
643           if (!distributed || (bytesToWrite = writableBytes()) <= 0 || !isChannelWritable0()) {
644             break;
645           }
646         }
647       } finally {
648         inWritePendingBytes = false;
649       }
650     }
651 
652     void initialWindowSize(int newWindowSize) throws Http2Exception {
653       checkPositiveOrZero(newWindowSize, "newWindowSize");
654 
655       final int delta = newWindowSize - initialWindowSize;
656       initialWindowSize = newWindowSize;
657       connection.forEachActiveStream(
658           new Http2StreamVisitor() {
659             @Override
660             public boolean visit(Http2Stream stream) throws Http2Exception {
661               state(stream).incrementStreamWindow(delta);
662               return true;
663             }
664           });
665 
666       if (delta > 0 && isChannelWritable()) {
667         // The window size increased, send any pending frames for all streams.
668         writePendingBytes();
669       }
670     }
671 
672     final boolean isWritableConnection() {
673       return connectionState.windowSize() - totalPendingBytes > 0 && isChannelWritable();
674     }
675   }

677   /**
678    * Writability of a {@code stream} is calculated using the following:
679    *
680    * <pre>
681    * Connection Window - Total Queued Bytes > 0 &&
682    * Stream Window - Bytes Queued for Stream > 0 &&
683    * isChannelWritable()
684    * </pre>
685    */
686   private final class ListenerWritabilityMonitor extends WritabilityMonitor
687       implements Http2StreamVisitor {
688     private final Listener listener;
689 
690     ListenerWritabilityMonitor(Listener listener) {
691       this.listener = listener;
692     }
693 
694     @Override
695     public boolean visit(Http2Stream stream) throws Http2Exception {
696       FlowState state = state(stream);
697       if (isWritable(state) != state.markedWritability()) {
698         notifyWritabilityChanged(state);
699       }
700       return true;
701     }
702 
703     @Override
704     void windowSize(FlowState state, int initialWindowSize) {
705       super.windowSize(state, initialWindowSize);
706       try {
707         checkStateWritability(state);
708       } catch (Http2Exception e) {
709         throw new RuntimeException("Caught unexpected exception from window", e);
710       }
711     }
712 
713     @Override
714     void incrementWindowSize(FlowState state, int delta) throws Http2Exception {
715       super.incrementWindowSize(state, delta);
716       checkStateWritability(state);
717     }
718 
719     @Override
720     void initialWindowSize(int newWindowSize) throws Http2Exception {
721       super.initialWindowSize(newWindowSize);
722       if (isWritableConnection()) {
723         // If the write operation does not occur we still need to check all streams because they
724         // may have transitioned from writable to not writable.
725         checkAllWritabilityChanged();
726       }
727     }
728 
729     @Override
730     void enqueueFrame(FlowState state, FlowControlled frame) throws Http2Exception {
731       super.enqueueFrame(state, frame);
732       checkConnectionThenStreamWritabilityChanged(state);
733     }
734 
735     @Override
736     void stateCancelled(FlowState state) {
737       try {
738         checkConnectionThenStreamWritabilityChanged(state);
739       } catch (Http2Exception e) {
740         throw new RuntimeException(
741             "Caught unexpected exception from checkAllWritabilityChanged", e);
742       }
743     }
744 
745     @Override
746     void channelWritabilityChange() throws Http2Exception {
747       if (connectionState.markedWritability() != isChannelWritable()) {
748         checkAllWritabilityChanged();
749       }
750     }
751 
752     private void checkStateWritability(FlowState state) throws Http2Exception {
753       if (isWritable(state) != state.markedWritability()) {
754         if (state == connectionState) {
755           checkAllWritabilityChanged();
756         } else {
757           notifyWritabilityChanged(state);
758         }
759       }
760     }
761 
762     private void notifyWritabilityChanged(FlowState state) {
763       state.markedWritability(!state.markedWritability());
764       try {
765         listener.writabilityChanged(state.stream);
766       } catch (Throwable cause) {
767         logger.error("Caught Throwable from listener.writabilityChanged", cause);
768       }
769     }
770 
771     private void checkConnectionThenStreamWritabilityChanged(FlowState state)
772         throws Http2Exception {
773       // It is possible that the connection window and/or the individual stream writability could
774       // change.
775       if (isWritableConnection() != connectionState.markedWritability()) {
776         checkAllWritabilityChanged();
777       } else if (isWritable(state) != state.markedWritability()) {
778         notifyWritabilityChanged(state);
779       }
780     }
781 
782     private void checkAllWritabilityChanged() throws Http2Exception {
783       // Make sure we mark that we have notified as a result of this change.
784       connectionState.markedWritability(isWritableConnection());
785       connection.forEachActiveStream(this);
786     }
787   }
