203   static NettyClientHandler newHandler(
204       final Http2Connection connection,
205       Http2FrameReader frameReader,
206       Http2FrameWriter frameWriter,
207       ClientTransportLifecycleManager lifecycleManager,
208       KeepAliveManager keepAliveManager,
209       boolean autoFlowControl,

213       Runnable tooManyPingsRunnable,

215       Attributes eagAttributes,
216       String authority,

218     Preconditions.checkNotNull(connection, "connection");
219     Preconditions.checkNotNull(frameReader, "frameReader");
220     Preconditions.checkNotNull(lifecycleManager, "lifecycleManager");
221     Preconditions.checkArgument(flowControlWindow > 0, "flowControlWindow must be positive");
222     Preconditions.checkArgument(maxHeaderListSize > 0, "maxHeaderListSize must be positive");
223     Preconditions.checkNotNull(stopwatchFactory, "stopwatchFactory");
224     Preconditions.checkNotNull(tooManyPingsRunnable, "tooManyPingsRunnable");
225     Preconditions.checkNotNull(eagAttributes, "eagAttributes");
226     Preconditions.checkNotNull(authority, "authority");
227 
228     Http2FrameLogger frameLogger = new Http2FrameLogger(LogLevel.DEBUG, NettyClientHandler.class);
229     frameReader = new Http2InboundFrameLogger(frameReader, frameLogger);
230     frameWriter = new Http2OutboundFrameLogger(frameWriter, frameLogger);
231 
232     PingCountingFrameWriter pingCounter;
233     frameWriter = pingCounter = new PingCountingFrameWriter(frameWriter);
234 
235     StreamBufferingEncoder encoder =
236         new StreamBufferingEncoder(new DefaultHttp2ConnectionEncoder(connection, frameWriter));

242             new DefaultHttp2LocalFlowController(connection, DEFAULT_WINDOW_UPDATE_RATIO, true));

244     Http2ConnectionDecoder decoder =
245         new DefaultHttp2ConnectionDecoder(connection, encoder, frameReader);

248         new TransportTracer.FlowControlReader() {

250           final Http2FlowController remote = connection.remote().flowController();

253           public TransportTracer.FlowControlWindows read() {
254             return new TransportTracer.FlowControlWindows(
255                 local.windowSize(connection.connectionStream()),
256                 remote.windowSize(connection.connectionStream()));
257           }
258         });
259

261     settings.pushEnabled(false);
262     settings.initialWindowSize(flowControlWindow);
263     settings.maxConcurrentStreams(0);
264     settings.maxHeaderListSize(maxHeaderListSize);
265 
266     return new NettyClientHandler(
267         decoder,
268         encoder,
269         settings,
270         negotiationLogger,

282   private NettyClientHandler(
283       Http2ConnectionDecoder decoder,
284       Http2ConnectionEncoder encoder,
285       Http2Settings settings,
286       ChannelLogger negotiationLogger,
287       ClientTransportLifecycleManager lifecycleManager,
288       KeepAliveManager keepAliveManager,
289       Supplier<Stopwatch> stopwatchFactory,
290       final Runnable tooManyPingsRunnable,
291       TransportTracer transportTracer,
292       Attributes eagAttributes,
293       String authority,
294       boolean autoFlowControl,
295       PingLimiter pingLimiter) {
296     super(
297         /* channelUnused= */ null,
298         decoder,
299         encoder,
300         settings,
301         negotiationLogger,
302         autoFlowControl,
303         pingLimiter);
304     this.lifecycleManager = lifecycleManager;
305     this.keepAliveManager = keepAliveManager;
306     this.stopwatchFactory = stopwatchFactory;
307     this.transportTracer = Preconditions.checkNotNull(transportTracer);
308     this.eagAttributes = eagAttributes;
309     this.authority = authority;
310     this.attributes =
311         Attributes.newBuilder().set(GrpcAttributes.ATTR_CLIENT_EAG_ATTRS, eagAttributes).build();
312 
313     // Set the frame listener on the decoder.
314     decoder().frameListener(new FrameListener());
315 
316     Http2Connection connection = encoder.connection();
317     streamKey = connection.newKey();
318 
319     connection.addListener(
320         new Http2ConnectionAdapter() {
321           @Override
322           public void onGoAwayReceived(int lastStreamId, long errorCode, ByteBuf debugData) {
323             byte[] debugDataBytes = ByteBufUtil.getBytes(debugData);
324             goingAway(errorCode, debugDataBytes);
325             if (errorCode == Http2Error.ENHANCE_YOUR_CALM.code()) {
326               String data = new String(debugDataBytes, UTF_8);
327               logger.log(
328                   Level.WARNING, "Received GOAWAY with ENHANCE_YOUR_CALM. Debug data: {0}", data);
329               if ("too_many_pings".equals(data)) {
330                 tooManyPingsRunnable.run();
331               }
332             }
333           }
334 
335           @Override
336           public void onStreamActive(Http2Stream stream) {
337             if (connection().numActiveStreams() == 1
338                 && NettyClientHandler.this.keepAliveManager != null) {
339               NettyClientHandler.this.keepAliveManager.onTransportActive();
340             }
341           }
342 
343           @Override
344           public void onStreamClosed(Http2Stream stream) {
345             // Although streams with CALL_OPTIONS_RPC_OWNED_BY_BALANCER are not marked as "in-use"
346             // in
347             // the first place, we don't propagate that option here, and it's safe to reset the
348             // in-use
349             // state for them, which will be a cheap no-op.
350             inUseState.updateObjectInUse(stream, false);
351             if (connection().numActiveStreams() == 0
352                 && NettyClientHandler.this.keepAliveManager != null) {
353               NettyClientHandler.this.keepAliveManager.onTransportIdle();
354             }
355           }
356         });
357   }

367   /** Handler for commands sent from the stream. */
368   @Override
369   public void write(ChannelHandlerContext ctx, Object msg, ChannelPromise promise)
370       throws Exception {
371     if (msg instanceof CreateStreamCommand) {
372       createStream((CreateStreamCommand) msg, promise);
373     } else if (msg instanceof SendGrpcFrameCommand) {
374       sendGrpcFrame(ctx, (SendGrpcFrameCommand) msg, promise);
375     } else if (msg instanceof CancelClientStreamCommand) {
376       cancelStream(ctx, (CancelClientStreamCommand) msg, promise);
377     } else if (msg instanceof SendPingCommand) {
378       sendPingFrame(ctx, (SendPingCommand) msg, promise);
379     } else if (msg instanceof GracefulCloseCommand) {
380       gracefulClose(ctx, (GracefulCloseCommand) msg, promise);
381     } else if (msg instanceof ForcefulCloseCommand) {
382       forcefulClose(ctx, (ForcefulCloseCommand) msg, promise);
383     } else if (msg == NOOP_MESSAGE) {
384       ctx.write(Unpooled.EMPTY_BUFFER, promise);
385     } else {
386       throw new AssertionError("Write called for unexpected type: " + msg.getClass().getName());
387     }
388   }

411   private void onHeadersRead(int streamId, Http2Headers headers, boolean endStream) {
412     // Stream 1 is reserved for the Upgrade response, so we should ignore its headers here:
413     if (streamId != Http2CodecUtil.HTTP_UPGRADE_STREAM_ID) {
414       NettyClientStream.TransportState stream = clientStream(requireHttp2Stream(streamId));
415       PerfMark.event("NettyClientHandler.onHeadersRead", stream.tag());
416       stream.transportHeadersReceived(headers, endStream);
417     }
418 
419     if (keepAliveManager != null) {
420       keepAliveManager.onDataReceived();
421     }
422   }

424   /** Handler for an inbound HTTP/2 DATA frame. */
425   private void onDataRead(int streamId, ByteBuf data, int padding, boolean endOfStream) {
426     flowControlPing().onDataRead(data.readableBytes(), padding);
427     NettyClientStream.TransportState stream = clientStream(requireHttp2Stream(streamId));
428     PerfMark.event("NettyClientHandler.onDataRead", stream.tag());
429     stream.transportDataReceived(data, endOfStream);
430     if (keepAliveManager != null) {
431       keepAliveManager.onDataReceived();
432     }
433   }

435   /** Handler for an inbound HTTP/2 RST_STREAM frame, terminating a stream. */
436   private void onRstStreamRead(int streamId, long errorCode) {
437     NettyClientStream.TransportState stream = clientStream(connection().stream(streamId));
438     if (stream != null) {
439       PerfMark.event("NettyClientHandler.onRstStreamRead", stream.tag());
440       Status status = statusFromH2Error(null, "RST_STREAM closed stream", errorCode, null);
441       stream.transportReportStatus(
442           status,
443           errorCode == Http2Error.REFUSED_STREAM.code()
444               ? RpcProgress.REFUSED
445               : RpcProgress.PROCESSED,
446           false /*stop delivery*/,
447           new Metadata());
448       if (keepAliveManager != null) {
449         keepAliveManager.onDataReceived();
450       }
451     }

454   @Override
455   public void close(ChannelHandlerContext ctx, ChannelPromise promise) throws Exception {
456     logger.fine("Network channel being closed by the application.");
457     if (ctx.channel().isActive()) { // Ignore notification that the socket was closed
458       lifecycleManager.notifyShutdown(
459           Status.UNAVAILABLE.withDescription("Transport closed for unknown reason"));
460     }
461     super.close(ctx, promise);
462   }

464   /** Handler for the Channel shutting down. */
465   @Override
466   public void channelInactive(ChannelHandlerContext ctx) throws Exception {
467     try {
468       logger.fine("Network channel is closed");
469       Status status = Status.UNAVAILABLE.withDescription("Network closed for unknown reason");
470       lifecycleManager.notifyShutdown(status);
471       final Status streamStatus;
472       if (channelInactiveReason != null) {
473         streamStatus = channelInactiveReason;
474       } else {
475         streamStatus = lifecycleManager.getShutdownStatus();
476       }
477       try {
478         cancelPing(lifecycleManager.getShutdownThrowable());
479         // Report status to the application layer for any open streams
480         connection()
481             .forEachActiveStream(
482                 new Http2StreamVisitor() {
483                   @Override
484                   public boolean visit(Http2Stream stream) throws Http2Exception {
485                     NettyClientStream.TransportState clientStream = clientStream(stream);
486                     if (clientStream != null) {
487                       clientStream.transportReportStatus(streamStatus, false, new Metadata());
488                     }
489                     return true;
490                   }
491                 });
492       } finally {
493         lifecycleManager.notifyTerminated(status);
494       }
495     } finally {
496       // Close any open streams
497       super.channelInactive(ctx);
498       if (keepAliveManager != null) {
499         keepAliveManager.onTransportTermination();
500       }
501     }
502   }

546   @Override
547   protected void onStreamError(
548       ChannelHandlerContext ctx,
549       boolean outbound,
550       Throwable cause,
551       Http2Exception.StreamException http2Ex) {
552     // Close the stream with a status that contains the cause.
553     NettyClientStream.TransportState stream = clientStream(connection().stream(http2Ex.streamId()));
554     if (stream != null) {
555       stream.transportReportStatus(Utils.statusFromThrowable(cause), false, new Metadata());
556     } else {
557       logger.log(Level.FINE, "Stream error for unknown stream " + http2Ex.streamId(), cause);
558     }
559 
560     // Delegate to the base class to send a RST_STREAM.
561     super.onStreamError(ctx, outbound, cause, http2Ex);
562   }

571   /**
572    * Attempts to create a new stream from the given command. If there are too many active streams,
573    * the creation request is queued.
574    */
575   private void createStream(CreateStreamCommand command, ChannelPromise promise) throws Exception {
576     if (lifecycleManager.getShutdownThrowable() != null) {
577       command.stream().setNonExistent();
578       // The connection is going away (it is really the GOAWAY case),
579       // just terminate the stream now.
580       command
581           .stream()
582           .transportReportStatus(
583               lifecycleManager.getShutdownStatus(), RpcProgress.REFUSED, true, new Metadata());
584       promise.setFailure(lifecycleManager.getShutdownThrowable());
585       return;
586     }
587 
588     // Get the stream ID for the new stream.
589     int streamId;
590     try {
591       streamId = incrementAndGetNextStreamId();
592     } catch (StatusException e) {
593       command.stream().setNonExistent();
594       // Stream IDs have been exhausted for this connection. Fail the promise immediately.
595       promise.setFailure(e);
596 
597       // Initiate a graceful shutdown if we haven't already.
598       if (!connection().goAwaySent()) {
599         logger.fine(
600             "Stream IDs have been exhausted for this connection. "
601                 + "Initiating graceful shutdown of the connection.");
602         lifecycleManager.notifyShutdown(e.getStatus());
603         close(ctx(), ctx().newPromise());
604       }
605       return;
606     }
607     if (connection().goAwayReceived()) {
608       Status s = abruptGoAwayStatus;
609       int maxActiveStreams = connection().local().maxActiveStreams();
610       int lastStreamId = connection().local().lastStreamKnownByPeer();
611       if (s == null) {
612         // Should be impossible, but handle pseudo-gracefully
613         s =
614             Status.INTERNAL.withDescription(
615                 "Failed due to abrupt GOAWAY, but can't find GOAWAY details");
616       } else if (streamId > lastStreamId) {
617         s =
618             s.augmentDescription(
619                 "stream id: " + streamId + ", GOAWAY Last-Stream-ID:" + lastStreamId);
620       } else if (connection().local().numActiveStreams() == maxActiveStreams) {
621         s = s.augmentDescription("At MAX_CONCURRENT_STREAMS limit. limit: " + maxActiveStreams);
622       }
623       if (streamId > lastStreamId || connection().local().numActiveStreams() == maxActiveStreams) {
624         // This should only be reachable during onGoAwayReceived, as otherwise
625         // getShutdownThrowable() != null
626         command.stream().setNonExistent();
627         command.stream().transportReportStatus(s, RpcProgress.REFUSED, true, new Metadata());
628         promise.setFailure(s.asRuntimeException());
629         return;
630       }
631     }
632 
633     NettyClientStream.TransportState stream = command.stream();
634     Http2Headers headers = command.headers();
635     stream.setId(streamId);
636 
637     PerfMark.startTask("NettyClientHandler.createStream", stream.tag());
638     Histogram.Timer createStream =
639         perfmarkNettyClientHandlerDuration.labels("NettyClientHandler.createStream").startTimer();
640     PerfMark.linkIn(command.getLink());
641     try {
642       createStreamTraced(
643           streamId, stream, headers, command.isGet(), command.shouldBeCountedForInUse(), promise);

645       PerfMark.stopTask("NettyClientHandler.createStream", stream.tag());
646       createStream.observeDuration();
647     }

650   private void createStreamTraced(
651       final int streamId,
652       final NettyClientStream.TransportState stream,
653       final Http2Headers headers,
654       boolean isGet,
655       final boolean shouldBeCountedForInUse,
656       final ChannelPromise promise) {
657     // Create an intermediate promise so that we can intercept the failure reported back to the
658     // application.
659     Histogram.Timer createFutureTimer = createStreamCreateNewFuture.startTimer();
660     ChannelPromise tempPromise = ctx().newPromise();
661     createFutureTimer.observeDuration();
662 
663     Histogram.Timer writeHeaderTimer = createStreamWriteHeaderDuration.startTimer();
664     ChannelFuture future = encoder().writeHeaders(ctx(), streamId, headers, 0, isGet, tempPromise);
665     writeHeaderTimer.observeDuration();
666 
667     Histogram.Timer addListenerTimer = createStreamAddListenerDuration.startTimer();
668     future.addListener(
669         new ChannelFutureListener() {
670           @Override
671           public void operationComplete(ChannelFuture future) throws Exception {
672             if (future.isSuccess()) {
673               // The http2Stream will be null in case a stream buffered in the encoder
674               // was canceled via RST_STREAM.
675               Http2Stream http2Stream = connection().stream(streamId);
676               if (http2Stream != null) {
677                 stream.getStatsTraceContext().clientOutboundHeaders();
678                 http2Stream.setProperty(streamKey, stream);
679 
680                 // This delays the in-use state until the I/O completes, which technically may
681                 // be later than we would like.
682                 if (shouldBeCountedForInUse) {
683                   inUseState.updateObjectInUse(http2Stream, true);
684                 }
685 
686                 // Attach the client stream to the HTTP/2 stream object as user data.
687                 stream.setHttp2Stream(http2Stream);
688               }
689               // Otherwise, the stream has been cancelled and Netty is sending a
690               // RST_STREAM frame which causes it to purge pending writes from the
691               // flow-controller and delete the http2Stream. The stream listener has already
692               // been notified of cancellation so there is nothing to do.
693 
694               // Just forward on the success status to the original promise.
695               promise.setSuccess();
696             } else {
697               final Throwable cause = future.cause();
698               if (cause instanceof StreamBufferingEncoder.Http2GoAwayException) {
699                 StreamBufferingEncoder.Http2GoAwayException e =
700                     (StreamBufferingEncoder.Http2GoAwayException) cause;
701                 Status status =
702                     statusFromH2Error(
703                         Status.Code.UNAVAILABLE,
704                         "GOAWAY closed buffered stream",
705                         e.errorCode(),
706                         e.debugData());
707                 stream.transportReportStatus(status, RpcProgress.REFUSED, true, new Metadata());
708                 promise.setFailure(status.asRuntimeException());
709               } else {
710                 promise.setFailure(cause);
711               }
712             }
713           }
714         });
715     addListenerTimer.observeDuration();
716   }

719   private void cancelStream(
720       ChannelHandlerContext ctx, CancelClientStreamCommand cmd, ChannelPromise promise) {
721     NettyClientStream.TransportState stream = cmd.stream();
722     PerfMark.startTask("NettyClientHandler.cancelStream", stream.tag());
723     Histogram.Timer cancelStream =
724         perfmarkNettyClientHandlerDuration.labels("NettyClientHandler.cancelStream").startTimer();
725     PerfMark.linkIn(cmd.getLink());
726     try {
727       Status reason = cmd.reason();
728       if (reason != null) {

730       }
731       if (!cmd.stream().isNonExistent()) {
732         encoder().writeRstStream(ctx, stream.id(), Http2Error.CANCEL.code(), promise);
733       } else {
734         promise.setSuccess();
735       }
736     } finally {
737       PerfMark.stopTask("NettyClientHandler.cancelStream", stream.tag());
738       cancelStream.observeDuration();
739     }

773   /**
774    * Sends a PING frame. If a ping operation is already outstanding, the callback in the message is
775    * registered to be called when the existing operation completes, and no new frame is sent.
776    */
777   private void sendPingFrameTraced(
778       ChannelHandlerContext ctx, SendPingCommand msg, ChannelPromise promise) {
779     // Don't check lifecycleManager.getShutdownStatus() since we want to allow pings after shutdown
780     // but before termination. After termination, messages will no longer arrive because the
781     // pipeline clears all handlers on channel close.
782 
783     PingCallback callback = msg.callback();
784     Executor executor = msg.executor();
785     // we only allow one outstanding ping at a time, so just add the callback to
786     // any outstanding operation
787     if (ping != null) {
788       promise.setSuccess();
789       ping.addCallback(callback, executor);
790       return;
791     }
792 
793     // Use a new promise to prevent calling the callback twice on write failure: here and in
794     // NettyClientTransport.ping(). It may appear strange, but it will behave the same as if
795     // ping != null above.
796     promise.setSuccess();
797     promise = ctx().newPromise();
798     // set outstanding operation
799     long data = USER_PING_PAYLOAD;
800     Stopwatch stopwatch = stopwatchFactory.get();
801     stopwatch.start();
802     ping = new Http2Ping(data, stopwatch);
803     ping.addCallback(callback, executor);
804     // and then write the ping
805     encoder().writePing(ctx, false, USER_PING_PAYLOAD, promise);
806     ctx.flush();
807     final Http2Ping finalPing = ping;
808     promise.addListener(
809         new ChannelFutureListener() {
810           @Override
811           public void operationComplete(ChannelFuture future) throws Exception {
812             if (future.isSuccess()) {
813               transportTracer.reportKeepAliveSent();
814             } else {
815               Throwable cause = future.cause();
816               if (cause instanceof ClosedChannelException) {
817                 cause = lifecycleManager.getShutdownThrowable();
818                 if (cause == null) {
819                   cause =
820                       Status.UNKNOWN
821                           .withDescription("Ping failed but for unknown reason.")
822                           .withCause(future.cause())
823                           .asException();
824                 }
825               }
826               finalPing.failed(cause);
827               if (ping == finalPing) {
828                 ping = null;
829               }
830             }
831           }
832         });
833   }

845   private void forcefulClose(
846       final ChannelHandlerContext ctx, final ForcefulCloseCommand msg, ChannelPromise promise)
847       throws Exception {
848     connection()
849         .forEachActiveStream(
850             new Http2StreamVisitor() {
851               @Override
852               public boolean visit(Http2Stream stream) throws Http2Exception {
853                 NettyClientStream.TransportState clientStream = clientStream(stream);
854                 Tag tag = clientStream != null ? clientStream.tag() : PerfMark.createTag();
855                 PerfMark.startTask("NettyClientHandler.forcefulClose", tag);
856                 Histogram.Timer forcefulClose =
857                     perfmarkNettyClientHandlerDuration
858                         .labels("NettyClientHandler.forcefulClose")
859                         .startTimer();
860                 PerfMark.linkIn(msg.getLink());
861                 try {
862                   if (clientStream != null) {
863                     clientStream.transportReportStatus(msg.getStatus(), true, new Metadata());
864                     resetStream(ctx, stream.id(), Http2Error.CANCEL.code(), ctx.newPromise());
865                   }
866                   stream.close();
867                   return true;
868                 } finally {
869                   PerfMark.stopTask("NettyClientHandler.forcefulClose", tag);
870                   forcefulClose.observeDuration();
871                 }
872               }
873             });
874     close(ctx, promise);
875   }

877   /**
878    * Handler for a GOAWAY being received. Fails any streams created after the last known stream. May
879    * only be called during a read.
880    */
881   private void goingAway(long errorCode, byte[] debugData) {
882     Status finalStatus =
883         statusFromH2Error(
884             Status.Code.UNAVAILABLE, "GOAWAY shut down transport", errorCode, debugData);
885     lifecycleManager.notifyGracefulShutdown(finalStatus);
886     abruptGoAwayStatus =
887         statusFromH2Error(
888             Status.Code.UNAVAILABLE, "Abrupt GOAWAY closed unsent stream", errorCode, debugData);
889     // While this _should_ be UNAVAILABLE, Netty uses the wrong stream id in the GOAWAY when it
890     // fails streams due to HPACK failures (e.g., header list too large). To be more conservative,
891     // we assume any sent streams may be related to the GOAWAY. This should rarely impact users
892     // since the main time servers should use abrupt GOAWAYs is if there is a protocol error, and if
893     // there wasn't a protocol error the error code was probably NO_ERROR which is mapped to
894     // UNAVAILABLE. https://github.com/netty/netty/issues/10670
895     final Status abruptGoAwayStatusConservative =
896         statusFromH2Error(null, "Abrupt GOAWAY closed sent stream", errorCode, debugData);
897     // Try to allocate as many in-flight streams as possible, to reduce race window of
898     // https://github.com/grpc/grpc-java/issues/2562 . To be of any help, the server has to
899     // gracefully shut down the connection with two GOAWAYs. gRPC servers generally send a PING
900     // after the first GOAWAY, so they can very precisely detect when the GOAWAY has been
901     // processed and thus this processing must be in-line before processing additional reads.
902 
903     // This can cause reentrancy, but should be minor since it is normal to handle writes in
904     // response to a read. Also, the call stack is rather shallow at this point
905     clientWriteQueue.drainNow();
906     if (lifecycleManager.notifyShutdown(finalStatus)) {
907       // This is for the only RPCs that are actually covered by the GOAWAY error code. All other
908       // RPCs were not observed by the remote and so should be UNAVAILABLE.
909       channelInactiveReason =
910           statusFromH2Error(null, "Connection closed after GOAWAY", errorCode, debugData);
911     }
912 
913     final int lastKnownStream = connection().local().lastStreamKnownByPeer();
914     try {
915       connection()
916           .forEachActiveStream(
917               new Http2StreamVisitor() {
918                 @Override
919                 public boolean visit(Http2Stream stream) throws Http2Exception {
920                   if (stream.id() > lastKnownStream) {
921                     NettyClientStream.TransportState clientStream = clientStream(stream);
922                     if (clientStream != null) {
923                       // RpcProgress _should_ be REFUSED, but are being conservative. See comment
924                       // for
925                       // abruptGoAwayStatusConservative. This does reduce our ability to perform
926                       // transparent
927                       // retries, but our main goal of transporent retries is to resolve the local
928                       // race. We
929                       // still hope/expect servers to use the graceful double-GOAWAY when closing
930                       // connections.
931                       clientStream.transportReportStatus(
932                           abruptGoAwayStatusConservative,
933                           RpcProgress.PROCESSED,
934                           false,
935                           new Metadata());
936                     }
937                     stream.close();
938                   }
939                   return true;
940                 }
941               });
942     } catch (Http2Exception e) {
943       throw new RuntimeException(e);
944     }
945   }

954   /** If {@code statusCode} is non-null, it will be used instead of the http2 error code mapping. */
955   private Status statusFromH2Error(
956       Status.Code statusCode, String context, long errorCode, byte[] debugData) {
957     Status status = GrpcUtil.Http2Error.statusForCode((int) errorCode);
958     if (statusCode == null) {
959       statusCode = status.getCode();
960     }
961     String debugString = "";
962     if (debugData != null && debugData.length > 0) {
963       // If a debug message was provided, use it.
964       debugString = ", debug data: " + new String(debugData, UTF_8);
965     }
966     return statusCode
967         .toStatus()
968         .withDescription(context + ". " + status.getDescription() + debugString);
969   }
