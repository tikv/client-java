18  package org.tikv.common.parser;
19  
20  import static org.tikv.common.types.IntegerType.BOOLEAN;
21  
22  import com.google.common.primitives.Doubles;
23  import org.antlr.v4.runtime.tree.TerminalNode;
24  import org.tikv.common.exception.UnsupportedSyntaxException;
25  import org.tikv.common.expression.*;
26  import org.tikv.common.expression.FuncCallExpr.Type;
27  import org.tikv.common.meta.TiTableInfo;
28  import org.tikv.common.parser.MySqlParser.ExpressionContext;
29  import org.tikv.common.parser.MySqlParser.FunctionNameBaseContext;
30  import org.tikv.common.types.IntegerType;
31  import org.tikv.common.types.RealType;
32  
33  // AstBuilder will convert ParseTree into Ast Node.
34  // In tikv java client, we only need to parser expression
35  // which is used by partition pruning.
36  public class AstBuilder extends MySqlParserBaseVisitor<Expression> {
37    private TiTableInfo tableInfo;
38  
39    public AstBuilder() {}
40  
41    public AstBuilder(TiTableInfo tableInfo) {
42      this.tableInfo = tableInfo;
43    }
44  
45    public Expression visitSimpleId(MySqlParser.SimpleIdContext ctx) {
46      if (ctx.ID() != null) {
47        return createColRef(ctx.ID().getSymbol().getText());
48      }
49  
50      if (ctx.functionNameBase() != null) {
51        return createColRef(ctx.functionNameBase().getText());
52      }
53  
54      throw new UnsupportedSyntaxException(ctx.getParent().toString() + ": it is not supported");
55    }
56  
57    private Expression createColRef(String id) {
58      if (tableInfo != null) {
59        return ColumnRef.create(id, tableInfo);
60      } else {
61        return ColumnRef.create(id);
62      }
63    }
64  
65    @Override
66    public Expression visitUid(MySqlParser.UidContext ctx) {
67      if (ctx.REVERSE_QUOTE_ID() != null) {
68        return createColRef(ctx.REVERSE_QUOTE_ID().getSymbol().getText());
69      }
70      return visitSimpleId(ctx.simpleId());
71    }
72  
73    @Override
74    public Expression visitScalarFunctionCall(MySqlParser.ScalarFunctionCallContext ctx) {
75      FunctionNameBaseContext fnNameCtx = ctx.scalarFunctionName().functionNameBase();
76      if (fnNameCtx != null) {
77        if (fnNameCtx.YEAR() != null) {
78          Expression args = visitFunctionArgs(ctx.functionArgs());
79          return new FuncCallExpr(args, Type.YEAR);
80        }
81      }
82      return visitChildren(ctx);
83    }
84  
85    @Override
86    public Expression visitFullColumnName(MySqlParser.FullColumnNameContext ctx) {
87      return visitUid(ctx.uid());
88    }
89  
90    private Expression parseIntOrLongOrDec(String val) {
91      try {
92        return Constant.create(Integer.parseInt(val), IntegerType.INT);
93      } catch (Exception e) {
94        try {
95          return Constant.create(Long.parseLong(val), IntegerType.BIGINT);
96        } catch (Exception e2) {
97          return Constant.create(Double.parseDouble(val), RealType.DOUBLE);
98        }
99      }
100   }
101 
102   @Override
103   public Expression visitDecimalLiteral(MySqlParser.DecimalLiteralContext ctx) {
104     if (ctx.ONE_DECIMAL() != null) {
105       String val = ctx.ONE_DECIMAL().getSymbol().getText();
106       return parseIntOrLongOrDec(val);
107     }
108     if (ctx.TWO_DECIMAL() != null) {
109       String val = ctx.TWO_DECIMAL().getSymbol().getText();
110       return parseIntOrLongOrDec(val);
111     }
112     if (ctx.DECIMAL_LITERAL() != null) {
113       String val = ctx.DECIMAL_LITERAL().getSymbol().getText();
114       return parseIntOrLongOrDec(val);
115     }
116 
117     if (ctx.ZERO_DECIMAL() != null) {
118       String val = ctx.ZERO_DECIMAL().getSymbol().getText();
119       return parseIntOrLongOrDec(val);
120     }
121 
122     throw new UnsupportedSyntaxException(ctx + ": it is not supported.");
123   }
124 
125   @Override
126   public Expression visitBooleanLiteral(MySqlParser.BooleanLiteralContext ctx) {
127     if (ctx.FALSE() != null) return Constant.create(0);
128     return Constant.create(1, BOOLEAN);
129   }
130 
131   @Override
132   public Expression visitStringLiteral(MySqlParser.StringLiteralContext ctx) {
133     if (ctx.STRING_LITERAL() != null) {
134       StringBuilder sb = new StringBuilder();
135       for (TerminalNode str : ctx.STRING_LITERAL()) {
136         sb.append(str.getSymbol().getText());
137       }
138       return Constant.create(sb.toString().replace("\"", ""));
139     }
140     throw new UnsupportedSyntaxException(ctx + " is not supported yet");
141   }
142 
143   @Override
144   public Expression visitConstant(MySqlParser.ConstantContext ctx) {
145     if (ctx.nullLiteral != null) {
146       return Constant.create(null);
147     }
148 
149     if (ctx.booleanLiteral() != null) {
150       return visitBooleanLiteral(ctx.booleanLiteral());
151     }
152 
153     if (ctx.decimalLiteral() != null) {
154       return visitDecimalLiteral(ctx.decimalLiteral());
155     }
156 
157     if (ctx.stringLiteral() != null) {
158       return visitStringLiteral(ctx.stringLiteral());
159     }
160 
161     if (ctx.REAL_LITERAL() != null) {
162       return Constant.create(
163           Doubles.tryParse(ctx.REAL_LITERAL().getSymbol().getText()), RealType.REAL);
164     }
165 
166     throw new UnsupportedSyntaxException(ctx + "not supported constant");
167   }
168 
169   @Override
170   public Expression visitConstantExpressionAtom(MySqlParser.ConstantExpressionAtomContext ctx) {
171     return visitChildren(ctx);
172   }
173 
174   @Override
175   public Expression visitBinaryComparisonPredicate(
176       MySqlParser.BinaryComparisonPredicateContext ctx) {
177     Expression left = visitChildren(ctx.left);
178     Expression right = visitChildren(ctx.right);
179     switch (ctx.comparisonOperator().getText()) {
180       case "<":
181         return ComparisonBinaryExpression.lessThan(left, right);
182       case "<=":
183         return ComparisonBinaryExpression.lessEqual(left, right);
184       case "=":
185         return ComparisonBinaryExpression.equal(left, right);
186       case ">":
187         return ComparisonBinaryExpression.greaterThan(left, right);
188       case ">=":
189         return ComparisonBinaryExpression.greaterEqual(left, right);
190     }
191 
192     throw new UnsupportedSyntaxException(ctx + ": it is not possible reach to this line of code");
193   }
194 
195   public Expression visitLogicalExpression(MySqlParser.LogicalExpressionContext ctx) {
196     ExpressionContext left = ctx.expression(0);
197     ExpressionContext right = ctx.expression(1);
198     switch (ctx.logicalOperator().getText()) {
199       case "and":
200         return LogicalBinaryExpression.and(visitChildren(left), visitChildren(right));
201       case "or":
202         return LogicalBinaryExpression.or(visitChildren(left), visitChildren(right));
203       case "xor":
204         return LogicalBinaryExpression.xor(visitChildren(left), visitChildren(right));
205     }
206 
207     throw new UnsupportedSyntaxException(ctx + ": it is not possible reach to this line of code");
208   }
209 
210   @Override
211   public Expression visitMathExpressionAtom(MySqlParser.MathExpressionAtomContext ctx) {
212     Expression left = visitChildren(ctx.left);
213     Expression right = visitChildren(ctx.right);
214     switch (ctx.mathOperator().getText()) {
215       case "+":
216         return ArithmeticBinaryExpression.plus(left, right);
217       case "-":
218         return ArithmeticBinaryExpression.minus(left, right);
219       case "*":
220         return ArithmeticBinaryExpression.multiply(left, right);
221       case "/":
222       case "div":
223         return ArithmeticBinaryExpression.divide(left, right);
224     }
225     throw new UnsupportedSyntaxException(ctx + ": it is not supported right now");
226   }
227 }
