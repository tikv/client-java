18  package org.tikv.common.meta;
19  
20  import com.fasterxml.jackson.annotation.JsonCreator;
21  import com.fasterxml.jackson.annotation.JsonProperty;
22  import java.io.Serializable;
23  
24  // TiUserIdentity represents username and hostname.
25  public class TiUserIdentity implements Serializable {
26    private final String username;
27    private final String hostname;
28    private final boolean currentUser;
29    private final String authUsername;
30    private final String authHostname;
31  
32    @JsonCreator
33    public TiUserIdentity(
34        @JsonProperty("Username") String userName,
35        @JsonProperty("Hostname") String hostName,
36        @JsonProperty("CurrentUser") boolean currentUser,
37        @JsonProperty("AuthUsername") String authUserName,
38        @JsonProperty("AuthHostname") String authHostName) {
39      this.authHostname = authHostName;
40      this.authUsername = authUserName;
41      this.hostname = hostName;
42      this.username = userName;
43      this.currentUser = currentUser;
44    }
45  }
