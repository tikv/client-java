68    @VisibleForTesting
69    public static ScanSpec extractConditions(
70        List<Expression> conditions, TiTableInfo table, TiIndexInfo index) {
71      // 0. Different than TiDB implementation, here logic has been unified for TableScan and
72      // IndexScan by
73      // adding fake index on clustered table's pk
74      // 1. Generate access point based on equal conditions
75      // 2. Cut access point condition if index is not continuous
76      // 3. Push back prefix index conditions since prefix index retrieve more result than needed
77      // 4. For remaining indexes (since access conditions consume some index, and they will
78      // not be used in filter push down later), find continuous matching index until first unmatched
79      // 5. Push back index related filter if prefix index, for remaining filters
80      // Equal conditions needs to be process first according to index sequence
81      // When index is null, no access condition can be applied
82      ScanSpec.Builder specBuilder = new ScanSpec.Builder(table, index);
83      if (index != null) {
84        Set<Expression> visited = new HashSet<>();
85        IndexMatchingLoop:
86        for (int i = 0; i < index.getIndexColumns().size(); i++) {
87          // for each index column try matches an equal condition
88          // and push remaining back
89          // TODO: if more than one equal conditions match an
90          // index, it likely yields nothing. Maybe a check needed
91          // to simplify it to a false condition
92          TiIndexColumn col = index.getIndexColumns().get(i);
93          IndexMatcher eqMatcher = IndexMatcher.equalOnlyMatcher(col);
94          boolean found = false;
95          // For first prefix index encountered, it equals to a range
96          // and we cannot push equal conditions further
97          for (Expression cond : conditions) {
98            if (visited.contains(cond)) {
99              continue;
100           }
101           if (eqMatcher.match(cond)) {
102             specBuilder.addPointPredicate(col, cond);
103             if (col.isPrefixIndex()) {
104               specBuilder.addResidualPredicate(cond);
105               break IndexMatchingLoop;
106             }
107             visited.add(cond);
108             found = true;
109             break;
110           }
111         }
112         if (!found) {
113           // For first "broken index chain piece"
114           // search for a matching range condition
115           IndexMatcher matcher = IndexMatcher.matcher(col);
116           for (Expression cond : conditions) {
117             if (visited.contains(cond)) {
118               continue;
119             }
120             if (matcher.match(cond)) {
121               specBuilder.addRangePredicate(col, cond);
122               if (col.isPrefixIndex()) {
123                 specBuilder.addResidualPredicate(cond);
124                 break;
125               }
126             }
127           }
128           break;
129         }
130       }
131     }
132 
133     specBuilder.addAllPredicates(conditions);
134     return specBuilder.build();
135   }

147   // Build scan plan picking access path with lowest cost by estimation
148   public TiDAGRequest buildTiDAGReq(
149       boolean allowIndexScan,
150       boolean canUseTiKV,
151       boolean canUseTiFlash,
152       List<TiColumnInfo> columnList,
153       List<Expression> conditions,
154       TiTableInfo table,
155       TableStatistics tableStatistics,
156       TiTimestamp ts,
157       TiDAGRequest dagRequest) {
158 
159     TiKVScanPlan minPlan = null;
160     if (canUseTiKV) {
161       minPlan = buildTableScan(conditions, table, tableStatistics);
162     }
163     if (canUseTiFlash) {
164       // it is possible that only TiFlash plan exists due to isolation read.
165       TiKVScanPlan plan = buildTiFlashScan(columnList, conditions, table, tableStatistics);
166       if (minPlan == null || plan.getCost() < minPlan.getCost()) {
167         minPlan = plan;
168       }
169     } else if (canUseTiKV && allowIndexScan) {
170       minPlan.getFilters().forEach(dagRequest::addDowngradeFilter);
171       double minCost = minPlan.getCost();
172       for (TiIndexInfo index : table.getIndices()) {
173         TiKVScanPlan plan =
174             buildIndexScan(columnList, conditions, index, table, tableStatistics, false);
175         if (plan.getCost() < minCost) {
176           minPlan = plan;
177           minCost = plan.getCost();
178         }
179       }
180     }
181     if (minPlan == null) {
182       throw new TiClientInternalException(
183           "No valid plan found for table '" + table.getName() + "'");
184     }
185 
186     TiStoreType minPlanStoreType = minPlan.getStoreType();
187     // TiKV should not use CHBlock as Encode Type.
188     if (minPlanStoreType == TiStoreType.TiKV
189         && dagRequest.getEncodeType() == EncodeType.TypeCHBlock) {
190       dagRequest.setEncodeType(EncodeType.TypeChunk);
191     }
192     // Set DAG Request's store type as minPlan's store type.
193     dagRequest.setStoreType(minPlanStoreType);
194 
195     dagRequest.addRanges(minPlan.getKeyRanges());
196     dagRequest.setPrunedParts(minPlan.getPrunedParts());
197     dagRequest.addFilters(new ArrayList<>(minPlan.getFilters()));
198     if (minPlan.isIndexScan()) {
199       dagRequest.setIndexInfo(minPlan.getIndex());
200       // need to set isDoubleRead to true for dagRequest in case of double read
201       dagRequest.setIsDoubleRead(minPlan.isDoubleRead());
202     }
203 
204     dagRequest.setTableInfo(table);
205     dagRequest.setStartTs(ts);
206     dagRequest.setEstimatedCount(minPlan.getEstimatedRowCount());
207     return dagRequest;
208   }

217       List<TiColumnInfo> columnList,
218       List<Expression> conditions,
219       TiTableInfo table,

225   TiKVScanPlan buildIndexScan(
226       List<TiColumnInfo> columnList,
227       List<Expression> conditions,
228       TiIndexInfo index,
229       TiTableInfo table,
230       TableStatistics tableStatistics,
231       boolean useTiFlash) {
232     requireNonNull(table, "Table cannot be null to encoding keyRange");
233     requireNonNull(conditions, "conditions cannot be null to encoding keyRange");
234 
235     TiKVScanPlan.Builder planBuilder = TiKVScanPlan.Builder.newBuilder(table.getName());
236     ScanSpec result = extractConditions(conditions, table, index);
237 
238     double cost = SelectivityCalculator.calcPseudoSelectivity(result);
239     planBuilder.setCost(cost);
240 
241     List<IndexRange> irs =
242         expressionToIndexRanges(
243             result.getPointPredicates(), result.getRangePredicate(), table, index);
244 
245     List<TiPartitionDef> prunedParts = null;
246     // apply partition pruning here.
247     if (table.getPartitionInfo() != null) {
248       prunedParts = PartitionPruner.prune(table, conditions);
249     }
250     planBuilder.setFilters(result.getResidualPredicates()).setPrunedParts(prunedParts);
251 
252     // table name and columns
253     long tableColSize = table.getEstimatedRowSizeInByte() + TABLE_PREFIX_SIZE;
254 
255     if (index == null || index.isFakePrimaryKey()) {
256       planBuilder
257           .setDoubleRead(false)
258           .setKeyRanges(buildTableScanKeyRange(table, irs, prunedParts));
259       if (useTiFlash) {
260         // TiFlash is a columnar storage engine
261         long colSize =
262             columnList.stream().mapToLong(TiColumnInfo::getSize).sum() + TABLE_PREFIX_SIZE;
263         return planBuilder
264             .setStoreType(TiStoreType.TiFlash)
265             .calculateCostAndEstimateCount(tableStatistics, colSize)
266             .build();
267       } else {
268         return planBuilder.calculateCostAndEstimateCount(tableStatistics, tableColSize).build();
269       }
270     } else {
271       // TiFlash does not support index scan.
272       assert (!useTiFlash);
273       long indexSize = index.getIndexColumnSize() + TABLE_PREFIX_SIZE + INDEX_PREFIX_SIZE;
274       return planBuilder
275           .setIndex(index)
276           .setDoubleRead(!isCoveringIndex(columnList, index, table.isPkHandle()))
277           // table name, index and handle column
278           .calculateCostAndEstimateCount(tableStatistics, conditions, irs, indexSize, tableColSize)
279           .setKeyRanges(buildIndexScanKeyRange(table, index, irs, prunedParts))
280           .build();
281     }
282   }

284   private Pair<Key, Key> buildTableScanKeyRangePerId(long id, IndexRange ir) {
285     Key startKey;
286     Key endKey;
287     if (ir.hasAccessKey()) {
288       checkArgument(
289           !ir.hasRange(), "Table scan must have one and only one access condition / point");
290 
291       Key key = ir.getAccessKey();
292       checkArgument(key instanceof TypedKey, "Table scan key range must be typed key");
293       TypedKey typedKey = (TypedKey) key;
294       startKey = RowKey.toRowKey(id, typedKey);
295       endKey = startKey.next();
296     } else if (ir.hasRange()) {
297       checkArgument(
298           !ir.hasAccessKey(), "Table scan must have one and only one access condition / point");
299       Range<TypedKey> r = ir.getRange();
300 
301       if (!r.hasLowerBound()) {
302         // -INF
303         startKey = RowKey.createMin(id);
304       } else {
305         // Comparison with null should be filtered since it yields unknown always
306         startKey = RowKey.toRowKey(id, r.lowerEndpoint());
307         if (r.lowerBoundType().equals(BoundType.OPEN)) {
308           startKey = startKey.next();
309         }
310       }
311 
312       if (!r.hasUpperBound()) {
313         // INF
314         endKey = RowKey.createBeyondMax(id);
315       } else {
316         endKey = RowKey.toRowKey(id, r.upperEndpoint());
317         if (r.upperBoundType().equals(BoundType.CLOSED)) {
318           endKey = endKey.next();
319         }
320       }
321     } else {
322       throw new TiClientInternalException("Empty access conditions");
323     }
324     return new Pair<>(startKey, endKey);
325   }

327   private Map<Long, List<KeyRange>> buildTableScanKeyRangeWithIds(
328       List<Long> ids, List<IndexRange> indexRanges) {
329     Map<Long, List<KeyRange>> idRanges = new HashMap<>(ids.size());
330     for (Long id : ids) {
331       List<KeyRange> ranges = new ArrayList<>(indexRanges.size());
332       indexRanges.forEach(
333           (ir) -> {
334             Pair<Key, Key> pairKey = buildTableScanKeyRangePerId(id, ir);
335             Key startKey = pairKey.first;
336             Key endKey = pairKey.second;
337             // This range only possible when < MIN or > MAX
338             if (!startKey.equals(endKey)) {
339               ranges.add(makeCoprocRange(startKey.toByteString(), endKey.toByteString()));
340             }
341           });
342 
343       idRanges.put(id, ranges);
344     }
345     return idRanges;
346   }

348   @VisibleForTesting
349   Map<Long, List<KeyRange>> buildTableScanKeyRange(
350       TiTableInfo table, List<IndexRange> indexRanges, List<TiPartitionDef> prunedParts) {
351     requireNonNull(table, "Table is null");
352     requireNonNull(indexRanges, "indexRanges is null");
353 
354     if (table.isPartitionEnabled()) {
355       List<Long> ids = new ArrayList<>();
356       for (TiPartitionDef pDef : prunedParts) {
357         ids.add(pDef.getId());
358       }
359       return buildTableScanKeyRangeWithIds(ids, indexRanges);
360     } else {
361       return buildTableScanKeyRangeWithIds(ImmutableList.of(table.getId()), indexRanges);
362     }
363   }

365   @VisibleForTesting
366   private Map<Long, List<KeyRange>> buildIndexScanKeyRangeWithIds(
367       List<Long> ids, TiIndexInfo index, List<IndexRange> indexRanges) {
368     Map<Long, List<KeyRange>> idRanges = new HashMap<>();
369     for (long id : ids) {
370       List<KeyRange> ranges = new ArrayList<>(indexRanges.size());
371       for (IndexRange ir : indexRanges) {
372         IndexScanKeyRangeBuilder indexScanKeyRangeBuilder =
373             new IndexScanKeyRangeBuilder(id, index, ir);
374         ranges.add(indexScanKeyRangeBuilder.compute());
375       }
376 
377       idRanges.put(id, ranges);
378     }
379     return idRanges;
380   }

403   // If all the columns requested in the select list of query, are available in the index, then the
404   // query engine doesn't have to lookup the table again compared with double read.
405   boolean isCoveringIndex(
406       List<TiColumnInfo> columns, TiIndexInfo indexColumns, boolean pkIsHandle) {
407     if (columns.isEmpty()) {
408       return false;
409     }
410 
411     Map<String, TiIndexColumn> colInIndex =
412         indexColumns
413             .getIndexColumns()
414             .stream()
415             .collect(Collectors.toMap(TiIndexColumn::getName, col -> col));
416     for (TiColumnInfo colInfo : columns) {
417       if (pkIsHandle && colInfo.isPrimaryKey()) {
418         continue;
419       }
420       if (colInfo.getId() == -1) {
421         continue;
422       }
423       boolean colNotInIndex = false;
424       if (colInIndex.containsKey(colInfo.getName())) {
425         TiIndexColumn indexCol = colInIndex.get(colInfo.getName());
426         boolean isFullLength =
427             indexCol.isLengthUnspecified() || indexCol.getLength() == colInfo.getType().getLength();
428         if (!colInfo.getName().equalsIgnoreCase(indexCol.getName()) || !isFullLength) {
429           colNotInIndex = true;
430         }
431       } else {
432         colNotInIndex = true;
433       }
434       if (colNotInIndex) {
435         return false;
436       }
437     }
438     return true;
439   }

578       private void debug(IndexScanType scanType) {
579         String plan, desc;
580         switch (scanType) {
581           case TABLE_SCAN:
582             plan = "TableScan";
583             desc = storeType.toString();
584             break;
585           case INDEX_SCAN:
586             plan = "IndexScan";
587             desc = index.getName();
588             break;
589           case COVERING_INDEX_SCAN:
590             plan = "CoveringIndexScan";
591             desc = index.getName();
592             break;
593           default:
594             // should not reach
595             plan = "None";
596             desc = "";
597         }
598         logger.debug(
599             "[Table:"
600                 + tableName
601                 + "]["
602                 + plan
603                 + ":"
604                 + desc
605                 + "] cost="
606                 + cost
607                 + " estimated row count="
608                 + estimatedRowCount);
609       }

611       // TODO: Fine-grained statistics usage
612       Builder calculateCostAndEstimateCount(TableStatistics tableStatistics, long tableColSize) {
613         cost = 100.0;
614         cost *= tableColSize * TABLE_SCAN_COST_FACTOR;
615         if (tableStatistics != null) {
616           estimatedRowCount = tableStatistics.getCount();
617         }
618         debug(IndexScanType.TABLE_SCAN);
619         return this;
620       }
621 
622       Builder calculateCostAndEstimateCount(
623           TableStatistics tableStatistics,
624           List<Expression> conditions,
625           List<IndexRange> irs,
626           long indexSize,
627           long tableColSize) {
628         if (tableStatistics != null) {
629           double totalRowCount = tableStatistics.getCount();
630           IndexStatistics indexStatistics = tableStatistics.getIndexHistMap().get(index.getId());
631           if (indexStatistics != null) {
632             totalRowCount = indexStatistics.getHistogram().totalRowCount();
633           }
634           if (conditions.isEmpty()) {
635             cost = 100.0; // Full index scan cost
636             // TODO: Fine-grained statistics usage
637             estimatedRowCount = totalRowCount;
638           } else if (indexStatistics != null) {
639             double idxRangeRowCnt = indexStatistics.getRowCount(irs);
640             // guess the percentage of rows hit
641             cost = 100.0 * idxRangeRowCnt / totalRowCount;
642             estimatedRowCount = idxRangeRowCnt;
643           }
644 
645           if (isDoubleRead) {
646             cost *= tableColSize * DOUBLE_READ_COST_FACTOR + indexSize * INDEX_SCAN_COST_FACTOR;
647             debug(IndexScanType.INDEX_SCAN);
648           } else {
649             cost *= indexSize * INDEX_SCAN_COST_FACTOR;
650             debug(IndexScanType.COVERING_INDEX_SCAN);
651           }
652         }
653         return this;
654       }
655     }
