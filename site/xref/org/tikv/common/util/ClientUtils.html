34    /**
35     * Append batch to list and split them according to batch limit
36     *
37     * @param backOffer backOffer
38     * @param batches a grouped batch
39     * @param region region
40     * @param keys keys
41     * @param batchMaxSizeInBytes batch max limit
42     */
43    public static void appendBatches(
44        BackOffer backOffer,
45        List<Batch> batches,
46        TiRegion region,
47        List<ByteString> keys,
48        int batchMaxSizeInBytes,
49        int batchLimit) {
50      if (keys == null) {
51        return;
52      }
53      int len = keys.size();
54      for (int start = 0, end; start < len; start = end) {
55        int size = 0;
56        for (end = start;
57            end < len && size < batchMaxSizeInBytes && end - start < batchLimit;
58            end++) {
59          size += keys.get(end).size();
60        }
61        Batch batch = new Batch(backOffer, region, keys.subList(start, end));
62        batches.add(batch);
63      }
64    }

66    /**
67     * Append batch to list and split them according to batch limit
68     *
69     * @param backOffer backOffer
70     * @param batches a grouped batch
71     * @param region region
72     * @param keys keys
73     * @param values values
74     * @param batchMaxSizeInBytes batch max limit
75     */
76    public static void appendBatches(
77        BackOffer backOffer,
78        List<Batch> batches,
79        TiRegion region,
80        List<ByteString> keys,
81        List<ByteString> values,
82        int batchMaxSizeInBytes,
83        int batchLimit) {
84      if (keys == null) {
85        return;
86      }
87      int len = keys.size();
88      for (int start = 0, end; start < len; start = end) {
89        int size = 0;
90        for (end = start;
91            end < len && size < batchMaxSizeInBytes && end - start < batchLimit;
92            end++) {
93          size += keys.get(end).size();
94          size += values.get(end).size();
95        }
96        Batch batch =
97            new Batch(backOffer, region, keys.subList(start, end), values.subList(start, end));
98        batches.add(batch);
99      }
100   }

102   public static List<Batch> getBatches(
103       BackOffer backOffer,
104       List<ByteString> keys,
105       int batchSize,
106       int batchLimit,
107       RegionStoreClient.RegionStoreClientBuilder clientBuilder) {
108     Map<TiRegion, List<ByteString>> groupKeys =
109         groupKeysByRegion(clientBuilder.getRegionManager(), keys, backOffer);
110     List<Batch> retryBatches = new ArrayList<>();
111 
112     for (Map.Entry<TiRegion, List<ByteString>> entry : groupKeys.entrySet()) {
113       appendBatches(
114           backOffer, retryBatches, entry.getKey(), entry.getValue(), batchSize, batchLimit);
115     }
116 
117     return retryBatches;
118   }

130   /**
131    * Group by list of keys according to its region
132    *
133    * @param keys keys
134    * @return a mapping of keys and their region
135    */
136   public static Map<TiRegion, List<ByteString>> groupKeysByRegion(
137       RegionManager regionManager, List<ByteString> keys, BackOffer backoffer, boolean sorted) {
138     Map<TiRegion, List<ByteString>> groups = new HashMap<>();
139     if (!sorted) {
140       keys.sort((k1, k2) -> FastByteComparisons.compareTo(k1.toByteArray(), k2.toByteArray()));
141     }
142     TiRegion lastRegion = null;
143     for (ByteString key : keys) {
144       if (lastRegion == null || !lastRegion.contains(key)) {
145         lastRegion = regionManager.getRegionByKey(key, backoffer);
146       }
147       groups.computeIfAbsent(lastRegion, k -> new ArrayList<>()).add(key);
148     }
149     return groups;
150   }

152   public static List<Kvrpcpb.KvPair> getKvPairs(
153       ExecutorCompletionService<List<Kvrpcpb.KvPair>> completionService,
154       List<Batch> batches,
155       int backOff) {
156     try {
157       List<Kvrpcpb.KvPair> result = new ArrayList<>();
158       for (int i = 0; i < batches.size(); i++) {
159         result.addAll(completionService.take().get(backOff, TimeUnit.MILLISECONDS));
160       }
161       return result;
162     } catch (InterruptedException e) {
163       Thread.currentThread().interrupt();
164       throw new TiKVException("Current thread interrupted.", e);
165     } catch (TimeoutException e) {
166       throw new TiKVException("TimeOut Exceeded for current operation. ", e);
167     } catch (ExecutionException e) {
168       throw new TiKVException("Execution exception met.", e);
169     }
170   }

172   public static <T> void getTasks(
173       ExecutorCompletionService<List<T>> completionService,
174       Queue<List<T>> taskQueue,
175       List<T> batches,
176       long backOff) {
177     try {
178       for (int i = 0; i < batches.size(); i++) {
179         Future<List<T>> future = completionService.poll(backOff, TimeUnit.MILLISECONDS);
180         if (future == null) {
181           throw new TiKVException("TimeOut Exceeded for current operation.");
182         }
183         List<T> task = future.get();
184         if (!task.isEmpty()) {
185           taskQueue.offer(task);
186         }
187       }
188     } catch (InterruptedException e) {
189       Thread.currentThread().interrupt();
190       throw new TiKVException("Current thread interrupted.", e);
191     } catch (ExecutionException e) {
192       throw new TiKVException("Execution exception met.", e);
193     }
194   }

196   public static <T, U> List<U> getTasksWithOutput(
197       ExecutorCompletionService<Pair<List<T>, List<U>>> completionService,
198       Queue<List<T>> taskQueue,
199       List<T> batches,
200       long backOff) {
201     try {
202       List<U> result = new ArrayList<>();
203       for (int i = 0; i < batches.size(); i++) {
204         Future<Pair<List<T>, List<U>>> future =
205             completionService.poll(backOff, TimeUnit.MILLISECONDS);
206         if (future == null) {
207           throw new TiKVException("TimeOut Exceeded for current operation.");
208         }
209         Pair<List<T>, List<U>> task = future.get();
210         if (!task.first.isEmpty()) {
211           taskQueue.offer(task.first);
212         } else {
213           result.addAll(task.second);
214         }
215       }
216       return result;
217     } catch (InterruptedException e) {
218       Thread.currentThread().interrupt();
219       throw new TiKVException("Current thread interrupted.", e);
220     } catch (ExecutionException e) {
221       throw new TiKVException("Execution exception met.", e);
222     }
223   }

225   public static byte[] genUUID() {
226     UUID uuid = UUID.randomUUID();
227 
228     byte[] out = new byte[16];
229     long msb = uuid.getMostSignificantBits();
230     long lsb = uuid.getLeastSignificantBits();
231     for (int i = 0; i < 8; i++) {
232       out[i] = (byte) ((msb >> ((7 - i) * 8)) & 0xff);
233     }
234     for (int i = 8; i < 16; i++) {
235       out[i] = (byte) ((lsb >> ((15 - i) * 8)) & 0xff);
236     }
237     return out;
238   }
