18  package org.tikv.common.util;
19  
20  import io.grpc.ManagedChannel;
21  import io.grpc.netty.GrpcSslContexts;
22  import io.grpc.netty.NettyChannelBuilder;
23  import io.netty.handler.ssl.SslContext;
24  import io.netty.handler.ssl.SslContextBuilder;
25  import java.io.File;
26  import java.io.FileInputStream;
27  import java.net.URI;
28  import java.security.KeyStore;
29  import java.util.concurrent.ConcurrentHashMap;
30  import java.util.concurrent.TimeUnit;
31  import javax.net.ssl.KeyManagerFactory;
32  import javax.net.ssl.SSLException;
33  import javax.net.ssl.TrustManagerFactory;
34  import org.slf4j.Logger;
35  import org.slf4j.LoggerFactory;
36  import org.tikv.common.HostMapping;
37  import org.tikv.common.pd.PDUtils;
38  
39  public class ChannelFactory implements AutoCloseable {
40    private static final Logger logger = LoggerFactory.getLogger(ChannelFactory.class);
41  
42    private final int maxFrameSize;
43    private final int keepaliveTime;
44    private final int keepaliveTimeout;
45    private final int idleTimeout;
46    private final ConcurrentHashMap<String, ManagedChannel> connPool = new ConcurrentHashMap<>();
47    private final SslContextBuilder sslContextBuilder;
48    private static final String PUB_KEY_INFRA = "PKIX";
49  
50    public ChannelFactory(
51        int maxFrameSize, int keepaliveTime, int keepaliveTimeout, int idleTimeout) {
52      this.maxFrameSize = maxFrameSize;
53      this.keepaliveTime = keepaliveTime;
54      this.keepaliveTimeout = keepaliveTimeout;
55      this.idleTimeout = idleTimeout;
56      this.sslContextBuilder = null;
57    }
58  
59    public ChannelFactory(
60        int maxFrameSize,
61        int keepaliveTime,
62        int keepaliveTimeout,
63        int idleTimeout,
64        String trustCertCollectionFilePath,
65        String keyCertChainFilePath,
66        String keyFilePath) {
67      this.maxFrameSize = maxFrameSize;
68      this.keepaliveTime = keepaliveTime;
69      this.keepaliveTimeout = keepaliveTimeout;
70      this.idleTimeout = idleTimeout;
71      this.sslContextBuilder =
72          getSslContextBuilder(trustCertCollectionFilePath, keyCertChainFilePath, keyFilePath);
73    }
74  
75    public ChannelFactory(
76        int maxFrameSize,
77        int keepaliveTime,
78        int keepaliveTimeout,
79        int idleTimeout,
80        String jksKeyPath,
81        String jksKeyPassword,
82        String jkstrustPath,
83        String jksTrustPassword) {
84      this.maxFrameSize = maxFrameSize;
85      this.keepaliveTime = keepaliveTime;
86      this.keepaliveTimeout = keepaliveTimeout;
87      this.idleTimeout = idleTimeout;
88      this.sslContextBuilder =
89          getSslContextBuilder(jksKeyPath, jksKeyPassword, jkstrustPath, jksTrustPassword);
90    }
91  
92    private SslContextBuilder getSslContextBuilder(
93        String jksKeyPath, String jksKeyPassword, String jksTrustPath, String jksTrustPassword) {
94      SslContextBuilder builder = GrpcSslContexts.forClient();
95      try {
96        if (jksKeyPath != null && jksKeyPassword != null) {
97          KeyStore keyStore = KeyStore.getInstance("JKS");
98          keyStore.load(new FileInputStream(jksKeyPath), jksKeyPassword.toCharArray());
99          KeyManagerFactory keyManagerFactory =
100             KeyManagerFactory.getInstance(KeyManagerFactory.getDefaultAlgorithm());
101         keyManagerFactory.init(keyStore, jksKeyPassword.toCharArray());
102         builder.keyManager(keyManagerFactory);
103       }
104       if (jksTrustPath != null && jksTrustPassword != null) {
105         KeyStore trustStore = KeyStore.getInstance("JKS");
106         trustStore.load(new FileInputStream(jksTrustPath), jksTrustPassword.toCharArray());
107         TrustManagerFactory trustManagerFactory = TrustManagerFactory.getInstance(PUB_KEY_INFRA);
108         trustManagerFactory.init(trustStore);
109         builder.trustManager(trustManagerFactory);
110       }
111     } catch (Exception e) {
112       logger.error("JKS SSL context builder failed!", e);
113     }
114     return builder;
115   }
116 
117   private SslContextBuilder getSslContextBuilder(
118       String trustCertCollectionFilePath, String keyCertChainFilePath, String keyFilePath) {
119     SslContextBuilder builder = GrpcSslContexts.forClient();
120     if (trustCertCollectionFilePath != null) {
121       builder.trustManager(new File(trustCertCollectionFilePath));
122     }
123     if (keyCertChainFilePath != null && keyFilePath != null) {
124       builder.keyManager(new File(keyCertChainFilePath), new File(keyFilePath));
125     }
126     return builder;
127   }
128 
129   public ManagedChannel getChannel(String addressStr, HostMapping hostMapping) {
130     return connPool.computeIfAbsent(
131         addressStr,
132         key -> {
133           URI address;
134           URI mappedAddr;
135           try {
136             address = PDUtils.addrToUri(key);
137           } catch (Exception e) {
138             throw new IllegalArgumentException("failed to form address " + key, e);
139           }
140           try {
141             mappedAddr = hostMapping.getMappedURI(address);
142           } catch (Exception e) {
143             throw new IllegalArgumentException("failed to get mapped address " + address, e);
144           }
145 
146           // Channel should be lazy without actual connection until first call
147           // So a coarse grain lock is ok here
148           NettyChannelBuilder builder =
149               NettyChannelBuilder.forAddress(mappedAddr.getHost(), mappedAddr.getPort())
150                   .maxInboundMessageSize(maxFrameSize)
151                   .keepAliveTime(keepaliveTime, TimeUnit.SECONDS)
152                   .keepAliveTimeout(keepaliveTimeout, TimeUnit.SECONDS)
153                   .keepAliveWithoutCalls(true)
154                   .idleTimeout(idleTimeout, TimeUnit.SECONDS);
155 
156           if (sslContextBuilder == null) {
157             return builder.usePlaintext().build();
158           } else {
159             SslContext sslContext = null;
160             try {
161               sslContext = sslContextBuilder.build();
162             } catch (SSLException e) {
163               logger.error("create ssl context failed!", e);
164               return null;
165             }
166             return builder.sslContext(sslContext).build();
167           }
168         });
169   }
170 
171   public void close() {
172     for (ManagedChannel ch : connPool.values()) {
173       ch.shutdown();
174     }
175     connPool.clear();
176   }
177 }
