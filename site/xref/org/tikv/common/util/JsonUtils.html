30    private static final int KEY_ENTRY_LENGTH = 6;

33    // TypeCodeObject indicates the JSON is an object.
34    private static final byte TYPE_CODE_OBJECT = 0x01;
35    // TypeCodeArray indicates the JSON is an array.
36    private static final byte TYPE_CODE_ARRAY = 0x03;
37    // TypeCodeLiteral indicates the JSON is a literal.
38    private static final byte TYPE_CODE_LITERAL = 0x04;
39    // TypeCodeInt64 indicates the JSON is a signed integer.
40    private static final byte TYPE_CODE_INT64 = 0x09;
41    // TypeCodeUint64 indicates the JSON is a unsigned integer.
42    private static final byte TYPE_CODE_UINT64 = 0x0a;
43    // TypeCodeFloat64 indicates the JSON is a double float number.
44    private static final byte TYPE_CODE_FLOAT64 = 0x0b;
45    // TypeCodeString indicates the JSON is a string.
46    private static final byte TYPE_CODE_STRING = 0x0c;
47  
48    // LiteralNil represents JSON null.
49    private static final byte LITERAL_NIL = 0x00;
50    // LiteralTrue represents JSON true.
51    private static final byte LITERAL_TRUE = 0x01;
52    // LiteralFalse represents JSON false.
53    private static final byte LITERAL_FALSE = 0x02;
54    private static final JsonPrimitive JSON_FALSE = new JsonPrimitive(false);
55    private static final JsonPrimitive JSON_TRUE = new JsonPrimitive(true);

57    public static JsonElement parseJson(DataInput di) {
58      return parseValue(readByte(di), di);
59    }
60    /**
61     * The binary JSON format from MySQL 5.7 is as follows:
62     *
63     * <pre>{@code
64     * JsonFormat {
65     *      JSON doc ::= type value
66     *      type ::=
67     *          0x01 |       // large JSON object
68     *          0x03 |       // large JSON array
69     *          0x04 |       // literal (true/false/null)
70     *          0x05 |       // int16
71     *          0x06 |       // uint16
72     *          0x07 |       // int32
73     *          0x08 |       // uint32
74     *          0x09 |       // int64
75     *          0x0a |       // uint64
76     *          0x0b |       // double
77     *          0x0c |       // utf8mb4 string
78     *      value ::=
79     *          object  |
80     *          array   |
81     *          literal |
82     *          number  |
83     *          string  |
84     *      object ::= element-count size key-entry* value-entry* key* value*
85     *      array ::= element-count size value-entry* value*
86     *      // number of members in object or number of elements in array
87     *      element-count ::= uint32
88     *      // number of bytes in the binary representation of the object or array
89     *      size ::= uint32
90     *      key-entry ::= key-offset key-length
91     *      key-offset ::= uint32
92     *      key-length ::= uint16    // key length must be less than 64KB
93     *      value-entry ::= type offset-or-inlined-value
94     *      // This field holds either the offset to where the value is stored,
95     *      // or the value itself if it is small enough to be inlined (that is,
96     *      // if it is a JSON literal or a small enough [u]int).
97     *      offset-or-inlined-value ::= uint32
98     *      key ::= utf8mb4-data
99     *      literal ::=
100    *          0x00 |   // JSON null literal
101    *          0x01 |   // JSON true literal
102    *          0x02 |   // JSON false literal
103    *      number ::=  ....    // little-endian format for [u]int(16|32|64), whereas
104    *                          // double is stored in a platform-independent, eight-byte
105    *                          // format using float8store()
106    *      string ::= data-length utf8mb4-data
107    *      data-length ::= uint8*    // If the high bit of a byte is 1, the length
108    *                                // field is continued in the next byte,
109    *                                // otherwise it is the last byte of the length
110    *                                // field. So we need 1 byte to represent
111    *                                // lengths up to 127, 2 bytes to represent
112    *                                // lengths up to 16383, and so on...
113    * }
114    * }</pre>
115    *
116    * @param type type byte
117    * @param di codec data input
118    * @return Json element parsed
119    */
120   private static JsonElement parseValue(byte type, DataInput di) {
121     switch (type) {
122       case TYPE_CODE_OBJECT:
123         return parseObject(di);
124       case TYPE_CODE_ARRAY:
125         return parseArray(di);
126       case TYPE_CODE_LITERAL:
127         return parseLiteralJson(di);
128       case TYPE_CODE_INT64:
129         return new JsonPrimitive(parseInt64(di));
130       case TYPE_CODE_UINT64:
131         return new JsonPrimitive(parseUint64(di));
132       case TYPE_CODE_FLOAT64:
133         return new JsonPrimitive(parseDouble(di));
134       case TYPE_CODE_STRING:
135         long length = parseDataLength(di);
136         return new JsonPrimitive(parseString(di, length));
137       default:
138         throw new AssertionError("error type|type=" + (int) type);
139     }
140   }

142   // * notice use this as a unsigned long
143   private static long parseUint64(DataInput cdi) {
144     byte[] readBuffer = new byte[8];
145     readFully(cdi, readBuffer, 0, 8);
146 
147     return ((long) (readBuffer[7]) << 56)
148         + ((long) (readBuffer[6] & 255) << 48)
149         + ((long) (readBuffer[5] & 255) << 40)
150         + ((long) (readBuffer[4] & 255) << 32)
151         + ((long) (readBuffer[3] & 255) << 24)
152         + ((readBuffer[2] & 255) << 16)
153         + ((readBuffer[1] & 255) << 8)
154         + ((readBuffer[0] & 255));
155   }
156 
157   private static void readFully(DataInput di, byte[] buffer) {
158     try {
159       di.readFully(buffer);
160     } catch (IOException e) {
161       throw new UncheckedIOException(e);
162     }
163   }
164 
165   private static void readFully(DataInput cdi, byte[] readBuffer, final int off, final int len) {
166     try {
167       cdi.readFully(readBuffer, off, len);
168     } catch (IOException e) {
169       throw new UncheckedIOException(e);
170     }
171   }

173   private static long parseInt64(DataInput cdi) {
174     byte[] readBuffer = new byte[8];
175     readFully(cdi, readBuffer);
176     return bytesToLong(readBuffer);
177   }
178 
179   private static long parseUint32(DataInput cdi) {
180     byte[] readBuffer = new byte[4];
181     readFully(cdi, readBuffer);
182 
183     return ((long) (readBuffer[3] & 255) << 24)
184         + ((readBuffer[2] & 255) << 16)
185         + ((readBuffer[1] & 255) << 8)
186         + ((readBuffer[0] & 255));
187   }
188 
189   private static long bytesToLong(byte[] readBuffer) {
190     return ((long) readBuffer[7] << 56)
191         + ((long) (readBuffer[6] & 255) << 48)
192         + ((long) (readBuffer[5] & 255) << 40)
193         + ((long) (readBuffer[4] & 255) << 32)
194         + ((long) (readBuffer[3] & 255) << 24)
195         + ((readBuffer[2] & 255) << 16)
196         + ((readBuffer[1] & 255) << 8)
197         + ((readBuffer[0] & 255));
198   }
199 
200   private static double parseDouble(DataInput cdi) {
201     byte[] readBuffer = new byte[8];
202     readFully(cdi, readBuffer);
203     return Double.longBitsToDouble(bytesToLong(readBuffer));
204   }
205 
206   private static int parseUint16(DataInput cdi) {
207     byte[] readBuffer = new byte[2];
208     readFully(cdi, readBuffer);
209 
210     return ((readBuffer[1] & 255) << 8) + ((readBuffer[0] & 255));
211   }

213   private static String parseString(DataInput di, long length) {
214 
215     byte[] buffer = new byte[Math.toIntExact(length)];
216     readFully(di, buffer);
217     return new String(buffer, StandardCharsets.UTF_8);
218   }

220   /**
221    * @param di data input
222    * @return length of data.
223    */
224   private static long parseDataLength(DataInput di) {
225     long x = 0;
226     byte b;
227     int i = 0;
228     int s = 0;
229     while ((b = readByte(di)) < 0) {
230       if (i == 9) {
231         throw new IllegalArgumentException("overflow: found >=9 leading bytes");
232       }
233       x |= ((long) (b & 0x7f)) << s;
234       s += 7;
235       i++;
236     }
237 
238     if (i == 9 && b > 1) {
239       throw new IllegalArgumentException("overflow: 8 leading byte and last one > 1");
240     }
241     x |= ((long) b) << s;
242     return x;
243   }

245   private @Nullable static Boolean parseLiteral(DataInput cdi) {
246     byte type;
247     type = readByte(cdi);
248     switch (type) {
249       case LITERAL_FALSE:
250         return Boolean.FALSE;
251       case LITERAL_NIL:
252         return null;
253       case LITERAL_TRUE:
254         return Boolean.TRUE;
255       default:
256         throw new AssertionError("unknown literal type|" + (int) type);
257     }
258   }
259 
260   private static byte readByte(DataInput cdi) {
261     byte type;
262     try {
263       type = cdi.readByte();
264     } catch (IOException e) {
265       throw new UncheckedIOException(e);
266     }
267     return type;
268   }

270   private static JsonArray parseArray(DataInput di) {
271     long elementCount = parseUint32(di);
272     long size = parseUint32(di);
273     byte[] buffer = new byte[Math.toIntExact(size - 8)];
274     readFully(di, buffer);
275     JsonArray jsonArray = new JsonArray();
276     for (int i = 0; i < elementCount; i++) {
277       JsonElement value = parseValueEntry(buffer, VALUE_ENTRY_SIZE * i);
278       jsonArray.add(value);
279     }
280     return jsonArray;
281   }
282 
283   private static JsonObject parseObject(DataInput di) {
284     long elementCount = parseUint32(di);
285     long size = parseUint32(di);
286 
287     byte[] buffer = new byte[Math.toIntExact(size - 8)];
288     readFully(di, buffer);
289     JsonObject jsonObject = new JsonObject();
290     for (int i = 0; i < elementCount; i++) {
291       KeyEntry keyEntry = parseKeyEntry(ByteStreams.newDataInput(buffer, i * KEY_ENTRY_LENGTH));
292       String key =
293           parseString(
294               ByteStreams.newDataInput(buffer, Math.toIntExact(keyEntry.keyOffset - 8)),
295               keyEntry.keyLength);
296       long valueEntryOffset = elementCount * KEY_ENTRY_LENGTH + i * VALUE_ENTRY_SIZE;
297       JsonElement value = parseValueEntry(buffer, valueEntryOffset);
298       jsonObject.add(key, value);
299     }
300     return jsonObject;
301   }
302 
303   private static JsonElement parseValueEntry(byte[] buffer, long valueEntryOffset) {
304     byte valueType = buffer[Math.toIntExact(valueEntryOffset)];
305     JsonElement value;
306     ByteArrayDataInput bs = ByteStreams.newDataInput(buffer, Math.toIntExact(valueEntryOffset + 1));
307     switch (valueType) {
308       case TYPE_CODE_LITERAL:
309         value = parseLiteralJson(bs);
310         break;
311       default:
312         long valueOffset = parseUint32(bs);
313         value =
314             parseValue(
315                 valueType, ByteStreams.newDataInput(buffer, Math.toIntExact(valueOffset - 8)));
316     }
317     return value;
318   }

320   private static JsonElement parseLiteralJson(DataInput di) {
321     JsonElement value;
322     Boolean bool = parseLiteral(di);
323     if (bool == null) {
324       value = JsonNull.INSTANCE;
325     } else if (bool) {
326       value = JSON_TRUE;
327     } else {
328       value = JSON_FALSE;
329     }
330     return value;
331   }
332 
333   private static KeyEntry parseKeyEntry(DataInput di) {
334     return new KeyEntry(parseUint32(di), parseUint16(di));
335   }
336 
337   static class KeyEntry {
338     long keyOffset;
339     int keyLength;
340 
341     KeyEntry(long keyOffset, int keyLength) {
342       this.keyOffset = keyOffset;
343       this.keyLength = keyLength;
344     }
345   }
