18  package org.tikv.common.util;
19  
20  import io.prometheus.client.Histogram;
21  
22  public class HistogramUtils {
23    private static final double[] DURATION_BUCKETS =
24        new double[] {
25          0.001D, 0.002D, 0.003D, 0.004D, 0.005D,
26          0.008D, 0.010D, 0.012D, 0.015D, 0.020D,
27          0.025D, 0.030D, 0.035D, 0.040D, 0.045D,
28          0.050D, 0.060D, 0.07D, 0.080D, 0.090D,
29          0.10D, 0.120D, 0.150D, 0.170D, 0.200D,
30          0.4D, 0.5D, 0.6D, 0.7D, 0.8D,
31          1D, 2.5D, 5D, 7.5D, 10D,
32        };
33  
34    private static final double[] BYTES_BUCKETS;
35  
36    static {
37      BYTES_BUCKETS = new double[30];
38      for (int i = 0; i < 30; ++i) {
39        BYTES_BUCKETS[i] = 1 * Math.pow(1.7, i);
40      }
41    }
42  
43    public static Histogram.Builder buildDuration() {
44      return Histogram.build().buckets(DURATION_BUCKETS);
45    }
46  
47    public static Histogram.Builder buildBytes() {
48      return Histogram.build().buckets(BYTES_BUCKETS);
49    }
50  }
