18  package org.tikv.common.row;
19  
20  import java.math.BigDecimal;
21  import java.sql.Date;
22  import java.sql.Time;
23  import java.sql.Timestamp;
24  import org.tikv.common.codec.KeyUtils;
25  import org.tikv.common.types.Converter;
26  import org.tikv.common.types.DataType;
27  
28  // A dummy implementation of Row interface
29  // Using non-memory compact format
30  public class ObjectRowImpl implements Row {
31    private final Object[] values;
32  
33    private ObjectRowImpl(Object[] values) {
34      this.values = values;
35    }
36  
37    private ObjectRowImpl(int fieldCount) {
38      values = new Object[fieldCount];
39    }
40  
41    public static Row create(Object[] values) {
42      return new ObjectRowImpl(values);
43    }
44  
45    public static Row create(int fieldCount) {
46      return new ObjectRowImpl(fieldCount);
47    }
48  
49    @Override
50    public void setNull(int pos) {
51      values[pos] = null;
52    }
53  
54    @Override
55    public boolean isNull(int pos) {
56      return values[pos] == null;
57    }
58  
59    @Override
60    public void setFloat(int pos, float v) {
61      values[pos] = v;
62    }
63  
64    @Override
65    public float getFloat(int pos) {
66      return (float) values[pos];
67    }
68  
69    @Override
70    public void setInteger(int pos, int v) {
71      values[pos] = v;
72    }
73  
74    @Override
75    public int getInteger(int pos) {
76      return (int) values[pos];
77    }
78  
79    @Override
80    public void setShort(int pos, short v) {
81      values[pos] = v;
82    }
83  
84    @Override
85    public short getShort(int pos) {
86      return (short) values[pos];
87    }
88  
89    @Override
90    public void setDouble(int pos, double v) {
91      values[pos] = v;
92    }
93  
94    @Override
95    public double getDouble(int pos) {
96      // Null should be handled by client code with isNull
97      // below all get method behave the same
98      return (double) values[pos];
99    }
100 
101   @Override
102   public void setLong(int pos, long v) {
103     values[pos] = v;
104   }
105 
106   @Override
107   public long getLong(int pos) {
108     // Null should be handled by client code with isNull
109     // below all get method behave the same
110     return (long) values[pos];
111   }
112 
113   @Override
114   public long getUnsignedLong(int pos) {
115     return ((BigDecimal) values[pos]).longValue();
116   }
117 
118   @Override
119   public void setString(int pos, String v) {
120     values[pos] = v;
121   }
122 
123   @Override
124   public String getString(int pos) {
125     return Converter.convertToString(values[pos]);
126   }
127 
128   @Override
129   public void setTime(int pos, Time v) {
130     values[pos] = v;
131   }
132 
133   @Override
134   public Date getTime(int pos) {
135     return (Date) values[pos];
136   }
137 
138   @Override
139   public void setTimestamp(int pos, Timestamp v) {
140     values[pos] = v;
141   }
142 
143   @Override
144   public Timestamp getTimestamp(int pos) {
145     return (Timestamp) values[pos];
146   }
147 
148   @Override
149   public void setDate(int pos, Date v) {
150     values[pos] = v;
151   }
152 
153   @Override
154   public Date getDate(int pos) {
155     return (Date) values[pos];
156   }
157 
158   @Override
159   public void setBytes(int pos, byte[] v) {
160     values[pos] = v;
161   }
162 
163   @Override
164   public byte[] getBytes(int pos) {
165     return (byte[]) values[pos];
166   }
167 
168   @Override
169   public void set(int pos, DataType type, Object v) {
170     // Ignore type for this implementation since no serialization happens
171     values[pos] = v;
172   }
173 
174   @Override
175   public Object get(int pos, DataType type) {
176     // Ignore type for this implementation since no serialization happens
177     return values[pos];
178   }
179 
180   @Override
181   public int fieldCount() {
182     return values.length;
183   }
184 
185   @Override
186   public String toString() {
187     StringBuilder builder = new StringBuilder();
188     builder.append("(");
189     for (int i = 0; i < values.length; i++) {
190       if (values[i] instanceof byte[]) {
191         builder.append("[");
192         builder.append(KeyUtils.formatBytes(((byte[]) values[i])));
193         builder.append("]");
194       } else if (values[i] instanceof BigDecimal) {
195         builder.append(((BigDecimal) values[i]).toPlainString());
196       } else {
197         builder.append(values[i]);
198       }
199       if (i < values.length - 1) {
200         builder.append(",");
201       }
202     }
203     builder.append(")");
204     return builder.toString();
205   }
206 }
