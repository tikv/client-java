18  package org.tikv.common.key;
19  
20  import com.google.common.collect.BoundType;
21  import com.google.common.collect.Range;
22  import org.tikv.common.predicates.IndexRange;
23  
24  // An abstract class that computes key range from IndexRange
25  // After computing lower and upper boundaries, the builder should
26  // build a key range according to its own rules.
27  abstract class KeyRangeBuilder {
28  
29    private final IndexRange ir;
30    private final Key pointKey;
31    Key lPointKey;
32    Key uPointKey;
33    Key lKey;
34    Key uKey;
35  
36    KeyRangeBuilder(IndexRange ir) {
37      this.ir = ir;
38      pointKey = ir.hasAccessKey() ? Key.toRawKey(ir.getAccessKey().getBytes()) : Key.EMPTY;
39    }
40  
41    private void computeWithOutRange() {
42      lPointKey = pointKey;
43      uPointKey = pointKey.nextPrefix();
44  
45      lKey = Key.EMPTY;
46      uKey = Key.EMPTY;
47    }
48  
49    private void computeWithRange() {
50      Range<TypedKey> range = ir.getRange();
51      lPointKey = pointKey;
52      uPointKey = pointKey;
53  
54      if (!range.hasLowerBound()) {
55        // -INF
56        lKey = Key.NULL;
57      } else {
58        lKey = Key.toRawKey(range.lowerEndpoint().getBytes());
59        if (range.lowerBoundType().equals(BoundType.OPEN)) {
60          lKey = lKey.nextPrefix();
61        }
62      }
63  
64      if (!range.hasUpperBound()) {
65        // INF
66        uKey = Key.MAX;
67      } else {
68        uKey = Key.toRawKey(range.upperEndpoint().getBytes());
69        if (range.upperBoundType().equals(BoundType.CLOSED)) {
70          uKey = uKey.nextPrefix();
71        }
72      }
73    }
74  
75    void computeKeyRange() {
76      if (!ir.hasRange()) {
77        computeWithOutRange();
78      } else {
79        computeWithRange();
80      }
81    }
82  }
