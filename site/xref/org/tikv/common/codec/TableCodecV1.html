18  package org.tikv.common.codec;
19  
20  import java.util.HashMap;
21  import java.util.List;
22  import org.tikv.common.codec.Codec.IntegerCodec;
23  import org.tikv.common.meta.TiColumnInfo;
24  import org.tikv.common.meta.TiTableInfo;
25  import org.tikv.common.row.ObjectRowImpl;
26  import org.tikv.common.row.Row;
27  import org.tikv.common.types.DataType.EncodeType;
28  import org.tikv.common.types.IntegerType;
29  
30  public class TableCodecV1 {
31    /** Row layout: colID1, value1, colID2, value2, ..... */
32    protected static byte[] encodeRow(
33        List<TiColumnInfo> columnInfos, Object[] values, boolean isPkHandle) {
34      CodecDataOutput cdo = new CodecDataOutput();
35  
36      for (int i = 0; i < columnInfos.size(); i++) {
37        TiColumnInfo col = columnInfos.get(i);
38        // skip pk is handle case
39        if (col.isPrimaryKey() && isPkHandle) {
40          continue;
41        }
42        IntegerCodec.writeLongFully(cdo, col.getId(), false);
43        col.getType().encode(cdo, EncodeType.VALUE, values[i]);
44      }
45  
46      // We could not set nil value into kv.
47      if (cdo.toBytes().length == 0) {
48        return new byte[] {Codec.NULL_FLAG};
49      }
50  
51      return cdo.toBytes();
52    }
53  
54    protected static Object[] decodeObjects(byte[] value, Long handle, TiTableInfo tableInfo) {
55      if (handle == null && tableInfo.isPkHandle()) {
56        throw new IllegalArgumentException("when pk is handle, handle cannot be null");
57      }
58  
59      int colSize = tableInfo.getColumns().size();
60      HashMap<Long, TiColumnInfo> idToColumn = new HashMap<>(colSize);
61      for (TiColumnInfo col : tableInfo.getColumns()) {
62        idToColumn.put(col.getId(), col);
63      }
64  
65      // decode bytes to Map<ColumnID, Data>
66      HashMap<Long, Object> decodedDataMap = new HashMap<>(colSize);
67      CodecDataInput cdi = new CodecDataInput(value);
68      Object[] res = new Object[colSize];
69      while (!cdi.eof()) {
70        long colID = (long) IntegerType.BIGINT.decode(cdi);
71        Object colValue = idToColumn.get(colID).getType().decodeForBatchWrite(cdi);
72        decodedDataMap.put(colID, colValue);
73      }
74  
75      // construct Row with Map<ColumnID, Data> & handle
76      for (int i = 0; i < colSize; i++) {
77        // skip pk is handle case
78        TiColumnInfo col = tableInfo.getColumn(i);
79        if (col.isPrimaryKey() && tableInfo.isPkHandle()) {
80          res[i] = handle;
81        } else {
82          res[i] = decodedDataMap.get(col.getId());
83        }
84      }
85      return res;
86    }
87  
88    protected static Row decodeRow(byte[] value, Long handle, TiTableInfo tableInfo) {
89      return ObjectRowImpl.create(decodeObjects(value, handle, tableInfo));
90    }
91  }
