18  package org.tikv.common;
19  
20  import java.util.ArrayList;
21  import java.util.Comparator;
22  import java.util.List;
23  import java.util.stream.Collectors;
24  import org.tikv.common.key.IndexKey;
25  import org.tikv.common.key.Key;
26  import org.tikv.common.key.RowKey;
27  import org.tikv.common.meta.TiIndexInfo;
28  import org.tikv.common.meta.TiTableInfo;
29  import org.tikv.common.region.TiRegion;
30  
31  public class TiBatchWriteUtils {
32  
33    private static final Comparator<TiIndexInfo> tiIndexInfoComparator =
34        Comparator.comparing(TiIndexInfo::getId);
35  
36    public static List<TiRegion> getRegionByIndex(
37        TiSession session, TiTableInfo table, TiIndexInfo index) {
38      ArrayList<TiRegion> regionList = new ArrayList<>();
39      Key min = IndexKey.toIndexKey(table.getId(), index.getId());
40      Key max = min.nextPrefix();
41  
42      while (min.compareTo(max) < 0) {
43        TiRegion region = session.getRegionManager().getRegionByKey(min.toByteString());
44        regionList.add(region);
45        min = Key.toRawKey(region.getEndKey());
46      }
47      return regionList;
48    }
49  
50    public static List<TiRegion> getIndexRegions(TiSession session, TiTableInfo table) {
51      return table
52          .getIndices()
53          .stream()
54          .sorted(tiIndexInfoComparator)
55          .flatMap(index -> getRegionByIndex(session, table, index).stream())
56          .collect(Collectors.toList());
57    }
58  
59    public static List<TiRegion> getRecordRegions(TiSession session, TiTableInfo table) {
60      ArrayList<TiRegion> regionList = new ArrayList<>();
61      Key key = RowKey.createMin(table.getId());
62      RowKey endRowKey = RowKey.createBeyondMax(table.getId());
63  
64      while (key.compareTo(endRowKey) < 0) {
65        TiRegion region = session.getRegionManager().getRegionByKey(key.toByteString());
66        regionList.add(region);
67        key = Key.toRawKey(region.getEndKey());
68      }
69      return regionList;
70    }
71  
72    public static List<TiRegion> getRegionsByTable(TiSession session, TiTableInfo table) {
73      List<TiRegion> recordRegions = getIndexRegions(session, table);
74      recordRegions.addAll(getRecordRegions(session, table));
75      return recordRegions;
76    }
77  }
