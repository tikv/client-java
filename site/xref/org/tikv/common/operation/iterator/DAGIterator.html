44  public abstract class DAGIterator<T>
45      extends org.tikv.common.operation.iterator.CoprocessorIterator<T> {
46    private static final Logger logger = LoggerFactory.getLogger(DAGIterator.class.getName());
47    private final PushDownType pushDownType;
48    private final TiStoreType storeType;
49    private final long startTs;
50    protected EncodeType encodeType;
51    private ExecutorCompletionService<Iterator<SelectResponse>> streamingService;
52    private ExecutorCompletionService<SelectResponse> dagService;
53    private SelectResponse response;
54    private Iterator<SelectResponse> responseIterator;
55  
56    DAGIterator(
57        DAGRequest req,
58        List<RangeSplitter.RegionTask> regionTasks,
59        TiSession session,
60        SchemaInfer infer,
61        PushDownType pushDownType,
62        TiStoreType storeType,
63        long startTs) {
64      super(req, regionTasks, session, infer);
65      this.pushDownType = pushDownType;
66      this.storeType = storeType;
67      this.startTs = startTs;
68      switch (pushDownType) {
69        case NORMAL:
70          dagService = new ExecutorCompletionService<>(session.getThreadPoolForTableScan());
71          break;
72        case STREAMING:
73          streamingService = new ExecutorCompletionService<>(session.getThreadPoolForTableScan());
74          break;
75      }
76      submitTasks();
77    }
78  
79    @Override
80    void submitTasks() {
81      for (RangeSplitter.RegionTask task : regionTasks) {
82        switch (pushDownType) {
83          case STREAMING:
84            streamingService.submit(() -> processByStreaming(task));
85            break;
86          case NORMAL:
87            dagService.submit(() -> process(task));
88            break;
89        }
90      }
91    }
92  
93    @Override
94    public boolean hasNext() {
95      if (eof) {
96        return false;
97      }
98  
99      while (chunkList == null || chunkIndex >= chunkList.size() || dataInput.available() <= 0) {
100       // First we check if our chunk list has remaining chunk
101       if (tryAdvanceChunkIndex()) {
102         createDataInputReader();
103       }
104       // If not, check next region/response
105       else if (pushDownType == STREAMING) {
106         if (!advanceNextResponse() && !readNextRegionChunks()) {
107           return false;
108         }
109       } else if (!readNextRegionChunks()) {
110         return false;
111       }
112     }
113 
114     return true;
115   }
116 
117   private boolean hasMoreResponse() {
118     switch (pushDownType) {
119       case STREAMING:
120         return responseIterator != null && responseIterator.hasNext();
121       case NORMAL:
122         return response != null;
123     }
124 
125     throw new IllegalArgumentException("Invalid push down type:" + pushDownType);
126   }
127 
128   private boolean advanceNextResponse() {
129     if (!hasMoreResponse()) {
130       return false;
131     }
132 
133     switch (pushDownType) {
134       case STREAMING:
135         SelectResponse resp = responseIterator.next();
136         chunkList = resp.getChunksList();
137         this.encodeType = resp.getEncodeType();
138         break;
139       case NORMAL:
140         chunkList = response.getChunksList();
141         this.encodeType = this.response.getEncodeType();
142         break;
143     }
144 
145     if (chunkList == null || chunkList.isEmpty()) {
146       return false;
147     }
148 
149     chunkIndex = 0;
150     createDataInputReader();
151     return true;
152   }
153 
154   /**
155    * chunk maybe empty while there is still data transmitting from TiKV. In this case, {@code
156    * readNextRegionChunks} cannot just returns false because the iterator thinks there is no data to
157    * process. This while loop ensures we can drain all possible data transmitting from TiKV.
158    *
159    * @return
160    */
161   private boolean readNextRegionChunks() {
162     while (hasNextRegionTask()) {
163       if (doReadNextRegionChunks()) {
164         return true;
165       }
166     }
167     return false;
168   }
169 
170   private boolean hasNextRegionTask() {
171     return !(eof || regionTasks == null || taskIndex >= regionTasks.size());
172   }
173 
174   private boolean doReadNextRegionChunks() {
175     try {
176       switch (pushDownType) {
177         case STREAMING:
178           responseIterator = streamingService.take().get();
179           break;
180         case NORMAL:
181           response = dagService.take().get();
182           break;
183       }
184 
185     } catch (Exception e) {
186       throw new TiClientInternalException("Error reading region:", e);
187     }
188 
189     taskIndex++;
190     return advanceNextResponse();
191   }
192 
193   private SelectResponse process(RangeSplitter.RegionTask regionTask) {
194     Queue<RangeSplitter.RegionTask> remainTasks = new ArrayDeque<>();
195     Queue<SelectResponse> responseQueue = new ArrayDeque<>();
196     remainTasks.add(regionTask);
197     BackOffer backOffer = ConcreteBackOffer.newCopNextMaxBackOff();
198 
199     HashSet<Long> resolvedLocks = new HashSet<>();
200     // In case of one region task spilt into several others, we ues a queue to properly handle all
201     // the remaining tasks.
202     while (!remainTasks.isEmpty()) {
203       RangeSplitter.RegionTask task = remainTasks.poll();
204       if (task == null) {
205         continue;
206       }
207       List<Coprocessor.KeyRange> ranges = task.getRanges();
208       TiRegion region = task.getRegion();
209       TiStore store = task.getStore();
210 
211       try {
212         RegionStoreClient client =
213             session.getRegionStoreClientBuilder().build(region, store, storeType);
214         client.addResolvedLocks(startTs, resolvedLocks);
215         Collection<RangeSplitter.RegionTask> tasks =
216             client.coprocess(backOffer, dagRequest, ranges, responseQueue, startTs);
217         if (tasks != null) {
218           remainTasks.addAll(tasks);
219         }
220         resolvedLocks.addAll(client.getResolvedLocks(startTs));
221       } catch (Throwable e) {
222         // Handle region task failed
223         logger.error(
224             "Process region tasks failed, remain "
225                 + remainTasks.size()
226                 + " tasks not executed due to",
227             e);
228         // Rethrow to upper levels
229         throw new RegionTaskException("Handle region task failed:", e);
230       }
231     }
232 
233     // Add all chunks to the final result
234     List<Chunk> resultChunk = new ArrayList<>();
235     EncodeType encodeType = null;
236     while (!responseQueue.isEmpty()) {
237       SelectResponse response = responseQueue.poll();
238       if (response != null) {
239         encodeType = response.getEncodeType();
240         resultChunk.addAll(response.getChunksList());
241       }
242     }
243 
244     return SelectResponse.newBuilder().addAllChunks(resultChunk).setEncodeType(encodeType).build();
245   }
246 
247   private Iterator<SelectResponse> processByStreaming(RangeSplitter.RegionTask regionTask) {
248     List<Coprocessor.KeyRange> ranges = regionTask.getRanges();
249     TiRegion region = regionTask.getRegion();
250     TiStore store = regionTask.getStore();
251 
252     RegionStoreClient client;
253     try {
254       client = session.getRegionStoreClientBuilder().build(region, store, storeType);
255       Iterator<SelectResponse> responseIterator =
256           client.coprocessStreaming(dagRequest, ranges, startTs);
257       if (responseIterator == null) {
258         eof = true;
259         return null;
260       }
261       return responseIterator;
262     } catch (Exception e) {
263       // TODO: Fix stale error handling in streaming
264       // see:https://github.com/pingcap/tikv-client-lib-java/pull/149
265       throw new TiClientInternalException("Error Closing Store client.", e);
266     }
267   }
