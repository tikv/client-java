42  public class RegionErrorHandler<RespT> implements ErrorHandler<RespT> {
43    private static final Logger logger = LoggerFactory.getLogger(RegionErrorHandler.class);
44    // if a store does not have leader currently, store id is set to 0
45    public static final int NO_LEADER_STORE_ID = 0;
46    private final Function<RespT, Errorpb.Error> getRegionError;
47    private final RegionManager regionManager;
48    private final RegionErrorReceiver recv;
49    private final List<Function<CacheInvalidateEvent, Void>> cacheInvalidateCallBackList;
50  
51    private final ExecutorService callBackThreadPool;
52    private final int INVALID_STORE_ID = 0;
53    private final int INVALID_REGION_ID = 0;
54  
55    public RegionErrorHandler(
56        RegionManager regionManager,
57        RegionErrorReceiver recv,
58        Function<RespT, Errorpb.Error> getRegionError) {
59      this.recv = recv;
60      this.regionManager = regionManager;
61      this.getRegionError = getRegionError;
62      this.cacheInvalidateCallBackList = regionManager.getCacheInvalidateCallbackList();
63      this.callBackThreadPool = regionManager.getCallBackThreadPool();
64    }
65  
66    @Override
67    public boolean handleResponseError(BackOffer backOffer, RespT resp) {
68      if (resp == null) {
69        String msg = String.format("Request Failed with unknown reason for [%s]", recv.getRegion());
70        return handleRequestError(backOffer, new GrpcException(msg));
71      }
72      // Region error handling logic
73      Errorpb.Error error = getRegionError(resp);
74      if (error != null) {
75        return handleRegionError(backOffer, error);
76      }
77      return false;
78    }

80    public boolean handleRegionError(BackOffer backOffer, Errorpb.Error error) {
81      if (error.hasNotLeader()) {
82        // this error is reported from raftstore:
83        // peer of current request is not leader, the following might be its causes:
84        // 1. cache is outdated, region has changed its leader, can be solved by re-fetching from PD
85        // 2. leader of current region is missing, need to wait and then fetch region info from PD
86        long newStoreId = error.getNotLeader().getLeader().getStoreId();
87        boolean retry;
88  
89        // update Leader here
90        logger.warn(
91            String.format(
92                "NotLeader Error with region id %d and store id %d, new store id %d",
93                recv.getRegion().getId(), recv.getRegion().getLeader().getStoreId(), newStoreId));
94  
95        BackOffFunction.BackOffFuncType backOffFuncType;
96        // if there's current no leader, we do not trigger update pd cache logic
97        // since issuing store = NO_LEADER_STORE_ID requests to pd will definitely fail.
98        if (newStoreId != NO_LEADER_STORE_ID) {
99          // If update leader fails, we need to fetch new region info from pd,
100         // and re-split key range for new region. Setting retry to false will
101         // stop retry and enter handleCopResponse logic, which would use RegionMiss
102         // backOff strategy to wait, fetch new region and re-split key range.
103         // onNotLeader is only needed when updateLeader succeeds, thus switch
104         // to a new store address.
105         TiRegion newRegion = this.regionManager.updateLeader(recv.getRegion(), newStoreId);
106         retry = newRegion != null && recv.onNotLeader(newRegion, backOffer);
107 
108         backOffFuncType = BackOffFunction.BackOffFuncType.BoUpdateLeader;
109       } else {
110         logger.info(
111             String.format(
112                 "Received zero store id, from region %d try next time", recv.getRegion().getId()));
113 
114         backOffFuncType = BackOffFunction.BackOffFuncType.BoRegionMiss;
115         retry = false;
116       }
117 
118       if (!retry) {
119         this.regionManager.invalidateRegion(recv.getRegion());
120         notifyRegionLeaderError(recv.getRegion());
121       }
122 
123       backOffer.doBackOff(backOffFuncType, new GrpcException(error.toString()));
124 
125       return retry;
126     } else if (error.hasStoreNotMatch()) {
127       // this error is reported from raftstore:
128       // store_id requested at the moment is inconsistent with that expected
129       // Solutionï¼šre-fetch from PD
130       long storeId = error.getStoreNotMatch().getRequestStoreId();
131       long actualStoreId = error.getStoreNotMatch().getActualStoreId();
132       logger.warn(
133           String.format(
134               "Store Not Match happened with region id %d, store id %d, actual store id %d",
135               recv.getRegion().getId(), storeId, actualStoreId));
136       // may request store which is not leader.
137       invalidateRegionStoreCache(recv.getRegion(), storeId);
138       // assume this is a low probability error, do not retry, just re-split the request by
139       // throwing it out.
140       return false;
141     } else if (error.hasEpochNotMatch()) {
142       logger.warn(
143           String.format("tikv reports `EpochNotMatch` retry later, region: %s", recv.getRegion()));
144       return onRegionEpochNotMatch(backOffer, error.getEpochNotMatch().getCurrentRegionsList());
145     } else if (error.hasServerIsBusy()) {
146       // this error is reported from kv:
147       // will occur when write pressure is high. Please try later.
148       logger.warn(
149           String.format(
150               "Server is busy for region [%s], reason: %s",
151               recv.getRegion(), error.getServerIsBusy().getReason()));
152       backOffer.doBackOff(
153           BackOffFunction.BackOffFuncType.BoServerBusy,
154           new StatusRuntimeException(
155               Status.fromCode(Status.Code.UNAVAILABLE).withDescription(error.toString())));
156       return true;
157     } else if (error.hasStaleCommand()) {
158       // this error is reported from raftstore:
159       // command outdated, please try later
160       logger.warn(String.format("Stale command for region [%s]", recv.getRegion()));
161       backOffer.doBackOff(
162           BackOffFunction.BackOffFuncType.BoRegionMiss, new GrpcException(error.getMessage()));
163       return true;
164     } else if (error.hasRaftEntryTooLarge()) {
165       logger.warn(String.format("Raft too large for region [%s]", recv.getRegion()));
166       throw new StatusRuntimeException(
167           Status.fromCode(Status.Code.UNAVAILABLE).withDescription(error.toString()));
168     } else if (error.hasKeyNotInRegion()) {
169       // this error is reported from raftstore:
170       // key requested is not in current region
171       // should not happen here.
172       ByteString invalidKey = error.getKeyNotInRegion().getKey();
173       logger.error(
174           String.format(
175               "Key not in region [%s] for key [%s], this error should not happen here.",
176               recv.getRegion(), KeyUtils.formatBytesUTF8(invalidKey)));
177       regionManager.clearRegionCache();
178       throw new StatusRuntimeException(Status.UNKNOWN.withDescription(error.toString()));
179     }
180     // The tso cache is used up in TiKV servers, we should backoff and wait its cache is renewed.
181     else if (error.getMessage().contains("TsoBatchUsedUp")) {
182       logger.warn(String.format("tso batch used up for region [%s]", recv.getRegion()));
183       backOffer.doBackOff(BackOffFuncType.BoTsoBatchUsedUp, new GrpcException(error.getMessage()));
184       return true;
185     }
186 
187     logger.warn(String.format("Unknown error %s for region [%s]", error, recv.getRegion()));
188     // For other errors, we only drop cache here.
189     // Upper level may split this task.
190     invalidateRegionStoreCache(recv.getRegion(), recv.getRegion().getLeader().getStoreId());
191     // retry if raft proposal is dropped, it indicates the store is in the middle of transition
192     if (error.getMessage().contains("Raft ProposalDropped")) {
193       backOffer.doBackOff(
194           BackOffFunction.BackOffFuncType.BoRegionMiss, new GrpcException(error.getMessage()));
195       return true;
196     }
197     return false;
198   }

200   // ref: https://github.com/tikv/client-go/blob/tidb-5.2/internal/locate/region_request.go#L985
201   // OnRegionEpochNotMatch removes the old region and inserts new regions into the cache.
202   // It returns whether retries the request because it's possible the region epoch is ahead of
203   // TiKV's due to slow appling.
204   private boolean onRegionEpochNotMatch(BackOffer backOffer, List<Metapb.Region> currentRegions) {
205     if (currentRegions.size() == 0) {
206       this.regionManager.onRegionStale(recv.getRegion());
207       notifyRegionCacheInvalidate(recv.getRegion());
208       return false;
209     }
210 
211     // Find whether the region epoch in `ctx` is ahead of TiKV's. If so, backoff.
212     for (Metapb.Region meta : currentRegions) {
213       if (meta.getId() == recv.getRegion().getId()
214           && (meta.getRegionEpoch().getConfVer() < recv.getRegion().getVerID().getConfVer()
215               || meta.getRegionEpoch().getVersion() < recv.getRegion().getVerID().getVer())) {
216         String errorMsg =
217             String.format(
218                 "region epoch is ahead of tikv, region: %s, currentRegions: %s",
219                 recv.getRegion(), currentRegions);
220         logger.info(errorMsg);
221         backOffer.doBackOff(
222             BackOffFunction.BackOffFuncType.BoRegionMiss, new TiKVException(errorMsg));
223         return true;
224       }
225     }
226 
227     boolean needInvalidateOld = true;
228     List<TiRegion> newRegions = new ArrayList<>(currentRegions.size());
229     // If the region epoch is not ahead of TiKV's, replace region meta in region cache.
230     for (Metapb.Region meta : currentRegions) {
231       // The region needs to be decoded to plain format.
232       meta = regionManager.getPDClient().getCodec().decodeRegion(meta);
233       TiRegion region = regionManager.createRegion(meta, backOffer);
234       newRegions.add(region);
235       if (recv.getRegion().getVerID() == region.getVerID()) {
236         needInvalidateOld = false;
237       }
238     }
239 
240     if (needInvalidateOld) {
241       notifyRegionCacheInvalidate(recv.getRegion());
242       this.regionManager.onRegionStale(recv.getRegion());
243     }
244 
245     for (TiRegion region : newRegions) {
246       regionManager.insertRegionToCache(region);
247     }
248 
249     return false;
250   }

252   @Override
253   public boolean handleRequestError(BackOffer backOffer, Exception e) {
254     if (recv.onStoreUnreachable(backOffer)) {
255       if (!backOffer.canRetryAfterSleep(BackOffFunction.BackOffFuncType.BoTiKVRPC)) {
256         regionManager.onRequestFail(recv.getRegion());
257         throw new GrpcException("retry is exhausted.", e);
258       }
259       return true;
260     }
261 
262     logger.warn("request failed because of: " + e.getMessage());
263     if (!backOffer.canRetryAfterSleep(BackOffFunction.BackOffFuncType.BoTiKVRPC)) {
264       regionManager.onRequestFail(recv.getRegion());
265       throw new GrpcException(
266           "send tikv request error: " + e.getMessage() + ", try next peer later", e);
267     }
268     // TiKV maybe down, so do not retry in `callWithRetry`
269     // should re-fetch the new leader from PD and send request to it
270     return false;
271   }

273   public Errorpb.Error getRegionError(RespT resp) {
274     if (getRegionError != null) {
275       return getRegionError.apply(resp);
276     }
277     return null;
278   }
279 
280   public TiRegion getRegion() {
281     return recv.getRegion();
282   }

284   private void notifyRegionRequestError(
285       TiRegion ctxRegion, long storeId, CacheInvalidateEvent.CacheType type) {
286     CacheInvalidateEvent event;
287     // When store(region) id is invalid,
288     // it implies that the error was not caused by store(region) error.
289     switch (type) {
290       case REGION:
291       case LEADER:
292         event = new CacheInvalidateEvent(ctxRegion.getId(), INVALID_STORE_ID, true, false, type);
293         break;
294       case REGION_STORE:
295         event = new CacheInvalidateEvent(ctxRegion.getId(), storeId, true, true, type);
296         break;
297       case REQ_FAILED:
298         event = new CacheInvalidateEvent(INVALID_REGION_ID, INVALID_STORE_ID, false, false, type);
299         break;
300       default:
301         throw new IllegalArgumentException("Unexpect invalid cache invalid type " + type);
302     }
303     if (cacheInvalidateCallBackList != null) {
304       for (Function<CacheInvalidateEvent, Void> cacheInvalidateCallBack :
305           cacheInvalidateCallBackList) {
306         callBackThreadPool.submit(
307             () -> {
308               try {
309                 cacheInvalidateCallBack.apply(event);
310               } catch (Exception e) {
311                 logger.error(String.format("CacheInvalidCallBack failed %s", e));
312               }
313             });
314       }
315     }
316   }
317 
318   private void invalidateRegionStoreCache(TiRegion ctxRegion, long storeId) {
319     regionManager.invalidateRegion(ctxRegion);
320     regionManager.invalidateStore(storeId);
321     notifyRegionRequestError(ctxRegion, storeId, CacheType.REGION_STORE);
322   }
323 
324   private void notifyRegionCacheInvalidate(TiRegion ctxRegion) {
325     notifyRegionRequestError(ctxRegion, 0, CacheType.REGION);
326   }
327 
328   private void notifyRegionLeaderError(TiRegion ctxRegion) {
329     notifyRegionRequestError(ctxRegion, 0, CacheType.LEADER);
330   }
