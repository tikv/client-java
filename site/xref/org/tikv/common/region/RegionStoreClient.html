154       TiConfiguration conf,
155       TiRegion region,
156       TiStore store,
157       TiStoreType storeType,
158       ChannelFactory channelFactory,
159       TikvBlockingStub blockingStub,
160       TikvFutureStub asyncStub,
161       RegionManager regionManager,
162       PDClient pdClient,
163       RegionStoreClient.RegionStoreClientBuilder clientBuilder) {
164     super(conf, region, store, channelFactory, blockingStub, asyncStub, regionManager);
165     this.storeType = storeType;
166 
167     if (this.storeType == TiStoreType.TiKV) {
168       this.lockResolverClient =
169           AbstractLockResolverClient.getInstance(
170               conf,
171               region,
172               store,
173               this.blockingStub,
174               this.asyncStub,
175               channelFactory,
176               regionManager,
177               pdClient,
178               clientBuilder);
179 
180     } else {
181       TiStore tikvStore =
182           regionManager.getRegionStorePairByKey(region.getStartKey(), TiStoreType.TiKV).second;
183 
184       String addressStr = tikvStore.getStore().getAddress();
185       if (logger.isDebugEnabled()) {
186         logger.debug(String.format("Create region store client on address %s", addressStr));
187       }
188       ManagedChannel channel = channelFactory.getChannel(addressStr, pdClient.getHostMapping());
189 
190       TikvBlockingStub tikvBlockingStub = TikvGrpc.newBlockingStub(channel);
191       TikvGrpc.TikvFutureStub tikvAsyncStub = TikvGrpc.newFutureStub(channel);
192 
193       this.lockResolverClient =
194           AbstractLockResolverClient.getInstance(
195               conf,
196               region,
197               tikvStore,
198               tikvBlockingStub,
199               tikvAsyncStub,
200               channelFactory,
201               regionManager,
202               pdClient,
203               clientBuilder);
204     }
205     this.pdClient = pdClient;
206   }

208   public synchronized boolean addResolvedLocks(Long version, Set<Long> locks) {
209     Set<Long> oldList = resolvedLocks.get(version);
210     if (oldList != null) {
211       oldList.addAll(locks);
212     } else {
213       resolvedLocks.put(version, new HashSet<>(locks));
214     }
215     return true;
216   }
217 
218   public synchronized Set<Long> getResolvedLocks(Long version) {
219     return resolvedLocks.getOrDefault(version, java.util.Collections.emptySet());
220   }

222   /**
223    * Fetch a value according to a key
224    *
225    * @param backOffer backOffer
226    * @param key key to fetch
227    * @param version key version
228    * @return value
229    * @throws TiClientInternalException TiSpark Client exception, unexpected
230    * @throws KeyException Key may be locked
231    */
232   public ByteString get(BackOffer backOffer, ByteString key, long version)
233       throws TiClientInternalException, KeyException {
234     boolean forWrite = false;
235     Supplier<GetRequest> factory =
236         () ->
237             GetRequest.newBuilder()
238                 .setContext(
239                     makeContext(getResolvedLocks(version), this.storeType, backOffer.getSlowLog()))
240                 .setKey(codec.encodeKey(key))
241                 .setVersion(version)
242                 .build();
243 
244     KVErrorHandler<GetResponse> handler =
245         new KVErrorHandler<>(
246             regionManager,
247             this,
248             lockResolverClient,
249             resp -> resp.hasRegionError() ? resp.getRegionError() : null,
250             resp -> resp.hasError() ? resp.getError() : null,
251             resolveLockResult -> addResolvedLocks(version, resolveLockResult.getResolvedLocks()),
252             version,
253             forWrite);
254 
255     GetResponse resp = callWithRetry(backOffer, TikvGrpc.getKvGetMethod(), factory, handler);
256 
257     handleGetResponse(resp);
258     return resp.getValue();
259   }

262    * @param resp GetResponse
263    * @throws TiClientInternalException TiSpark Client exception, unexpected
264    * @throws KeyException Key may be locked
265    */
266   private void handleGetResponse(GetResponse resp) throws TiClientInternalException, KeyException {
267     if (resp == null) {
268       this.regionManager.onRequestFail(region);
269       throw new TiClientInternalException("GetResponse failed without a cause");
270     }
271     if (resp.hasRegionError()) {
272       throw new RegionException(resp.getRegionError());
273     }
274     if (resp.hasError()) {
275       throw new KeyException(resp.getError());
276     }
277   }

304   private List<KvPair> handleBatchGetResponse(
305       BackOffer backOffer, BatchGetResponse resp, long version) {
306     boolean forWrite = false;
307     if (resp == null) {
308       this.regionManager.onRequestFail(region);
309       throw new TiClientInternalException("BatchGetResponse failed without a cause");
310     }
311     if (resp.hasRegionError()) {
312       throw new RegionException(resp.getRegionError());
313     }
314     List<Lock> locks = new ArrayList<>();
315 
316     for (KvPair pair : resp.getPairsList()) {
317       if (pair.hasError()) {
318         if (pair.getError().hasLocked()) {
319           Lock lock = new Lock(pair.getError().getLocked(), codec);
320           locks.add(lock);
321         } else {
322           throw new KeyException(pair.getError());
323         }
324       }
325     }
326 
327     if (!locks.isEmpty()) {
328       ResolveLockResult resolveLockResult =
329           lockResolverClient.resolveLocks(backOffer, version, locks, forWrite);
330       addResolvedLocks(version, resolveLockResult.getResolvedLocks());
331       // resolveLocks already retried, just throw error to upper logic.
332       throw new TiKVException("locks not resolved, retry");
333     }
334 
335     return codec.decodeKvPairs(resp.getPairsList());
336   }

338   public List<KvPair> scan(
339       BackOffer backOffer, ByteString startKey, long version, boolean keyOnly) {
340     boolean forWrite = false;
341     while (true) {
342       Supplier<ScanRequest> request =
343           () ->
344               ScanRequest.newBuilder()
345                   .setContext(
346                       makeContext(
347                           getResolvedLocks(version), this.storeType, backOffer.getSlowLog()))
348                   .setStartKey(codec.encodeKey(startKey))
349                   .setVersion(version)
350                   .setKeyOnly(keyOnly)
351                   .setLimit(getConf().getScanBatchSize())
352                   .build();
353 
354       KVErrorHandler<ScanResponse> handler =
355           new KVErrorHandler<>(
356               regionManager,
357               this,
358               lockResolverClient,
359               resp -> resp.hasRegionError() ? resp.getRegionError() : null,
360               resp -> null,
361               resolveLockResult -> addResolvedLocks(version, resolveLockResult.getResolvedLocks()),
362               version,
363               forWrite);
364       ScanResponse resp = callWithRetry(backOffer, TikvGrpc.getKvScanMethod(), request, handler);
365       // retry may refresh region info
366       // we need to update region after retry
367       region = regionManager.getRegionByKey(startKey, backOffer);
368 
369       if (isScanSuccess(backOffer, resp)) {
370         return doScan(resp);
371       }
372     }
373   }
374 
375   private boolean isScanSuccess(BackOffer backOffer, ScanResponse resp) {
376     if (resp == null) {
377       this.regionManager.onRequestFail(region);
378       throw new TiClientInternalException("ScanResponse failed without a cause");
379     }
380     if (resp.hasRegionError()) {
381       backOffer.doBackOff(BoRegionMiss, new RegionException(resp.getRegionError()));
382       return false;
383     }
384     return true;
385   }

387   // TODO: resolve locks after scan
388   private List<KvPair> doScan(ScanResponse resp) {
389     // Check if kvPair contains error, it should be a Lock if hasError is true.
390     List<KvPair> kvPairs = resp.getPairsList();
391     List<KvPair> newKvPairs = new ArrayList<>();
392     for (KvPair kvPair : kvPairs) {
393       if (kvPair.hasError()) {
394         Lock lock = AbstractLockResolverClient.extractLockFromKeyErr(kvPair.getError(), codec);
395         newKvPairs.add(
396             KvPair.newBuilder()
397                 .setError(kvPair.getError())
398                 .setValue(kvPair.getValue())
399                 .setKey(lock.getKey())
400                 .build());
401       } else {
402         newKvPairs.add(codec.decodeKvPair(kvPair));
403       }
404     }
405     return Collections.unmodifiableList(newKvPairs);
406   }

430   /**
431    * Prewrite batch keys
432    *
433    * @param skipConstraintCheck whether to skip constraint check
434    */
435   public void prewrite(
436       BackOffer bo,
437       ByteString primaryLock,
438       List<Mutation> mutations,
439       long startTs,
440       long ttl,
441       boolean skipConstraintCheck)
442       throws TiClientInternalException, KeyException, RegionException {
443     boolean forWrite = true;
444     while (true) {
445       Supplier<PrewriteRequest> factory =
446           () ->
447               getIsV4()
448                   ? PrewriteRequest.newBuilder()
449                       .setContext(makeContext(storeType, bo.getSlowLog()))
450                       .setStartVersion(startTs)
451                       .setPrimaryLock(codec.encodeKey(primaryLock))
452                       .addAllMutations(codec.encodeMutations(mutations))
453                       .setLockTtl(ttl)
454                       .setSkipConstraintCheck(skipConstraintCheck)
455                       .setMinCommitTs(startTs)
456                       .setTxnSize(16)
457                       .build()
458                   : PrewriteRequest.newBuilder()
459                       .setContext(makeContext(storeType, bo.getSlowLog()))
460                       .setStartVersion(startTs)
461                       .setPrimaryLock(primaryLock)
462                       .addAllMutations(mutations)
463                       .setLockTtl(ttl)
464                       .setSkipConstraintCheck(skipConstraintCheck)
465                       // v3 does not support setMinCommitTs(startTs)
466                       .setTxnSize(16)
467                       .build();
468       KVErrorHandler<PrewriteResponse> handler =
469           new KVErrorHandler<>(
470               regionManager,
471               this,
472               lockResolverClient,
473               resp -> resp.hasRegionError() ? resp.getRegionError() : null,
474               resp -> null,
475               resolveLockResult -> null,
476               startTs,
477               forWrite);
478       PrewriteResponse resp = callWithRetry(bo, TikvGrpc.getKvPrewriteMethod(), factory, handler);
479       if (isPrewriteSuccess(bo, resp, startTs)) {
480         return;
481       }
482     }
483   }

485   /**
486    * @param backOffer backOffer
487    * @param resp response
488    * @return Return true means the rpc call success. Return false means the rpc call fail,
489    *     RegionStoreClient should retry. Throw an Exception means the rpc call fail,
490    *     RegionStoreClient cannot handle this kind of error
491    * @throws TiClientInternalException
492    * @throws RegionException
493    * @throws KeyException
494    */
495   private boolean isPrewriteSuccess(BackOffer backOffer, PrewriteResponse resp, long startTs)
496       throws TiClientInternalException, KeyException, RegionException {
497     boolean forWrite = true;
498     if (resp == null) {
499       this.regionManager.onRequestFail(region);
500       throw new TiClientInternalException("Prewrite Response failed without a cause");
501     }
502     if (resp.hasRegionError()) {
503       throw new RegionException(resp.getRegionError());
504     }
505 
506     boolean isSuccess = true;
507     List<Lock> locks = new ArrayList<>();
508     for (KeyError err : resp.getErrorsList()) {
509       if (err.hasLocked()) {
510         isSuccess = false;
511         Lock lock = new Lock(err.getLocked(), codec);
512         locks.add(lock);
513       } else {
514         throw new KeyException(err.toString());
515       }
516     }
517     if (isSuccess) {
518       return true;
519     }
520 
521     ResolveLockResult resolveLockResult =
522         lockResolverClient.resolveLocks(backOffer, startTs, locks, forWrite);
523     addResolvedLocks(startTs, resolveLockResult.getResolvedLocks());
524     long msBeforeExpired = resolveLockResult.getMsBeforeTxnExpired();
525     if (msBeforeExpired > 0) {
526       backOffer.doBackOffWithMaxSleep(
527           BoTxnLock, msBeforeExpired, new KeyException(resp.getErrorsList().get(0)));
528     }
529     return false;
530   }

612     CommitResponse resp = callWithRetry(backOffer, TikvGrpc.getKvCommitMethod(), factory, handler);
613     handleCommitResponse(resp);
614   }

616   /**
617    * @param resp CommitResponse
618    * @throws TiClientInternalException
619    * @throws RegionException
620    * @throws KeyException
621    */
622   private void handleCommitResponse(CommitResponse resp)
623       throws TiClientInternalException, RegionException, KeyException {
624     if (resp == null) {
625       this.regionManager.onRequestFail(region);
626       throw new TiClientInternalException("CommitResponse failed without a cause");
627     }
628     if (resp.hasRegionError()) {
629       // bo.doBackOff(BoRegionMiss, new RegionException(resp.getRegionError()));
630       // return false;
631       // Caller method should restart commit
632       throw new RegionException(resp.getRegionError());
633     }
634     // If we find locks, we first resolve and let its caller retry.
635     if (resp.hasError()) {

640   /**
641    * Execute and retrieve the response from TiKV server.
642    *
643    * @param req Select request to process
644    * @param ranges Key range list
645    * @return Remaining tasks of this request, if task split happens, null otherwise
646    */
647   public List<RangeSplitter.RegionTask> coprocess(
648       BackOffer backOffer,
649       DAGRequest req,
650       List<Coprocessor.KeyRange> ranges,
651       Queue<SelectResponse> responseQueue,
652       long startTs) {
653     boolean forWrite = false;
654     if (req == null || ranges == null || req.getExecutorsCount() < 1) {
655       throw new IllegalArgumentException("Invalid coprocessor argument!");
656     }
657 
658     Supplier<Coprocessor.Request> reqToSend =
659         () ->
660             Coprocessor.Request.newBuilder()
661                 .setContext(
662                     makeContext(getResolvedLocks(startTs), this.storeType, backOffer.getSlowLog()))
663                 .setTp(REQ_TYPE_DAG.getValue())
664                 .setStartTs(startTs)
665                 .setData(req.toByteString())
666                 .addAllRanges(ranges)
667                 .build();
668 
669     // we should handle the region error ourselves
670     KVErrorHandler<Coprocessor.Response> handler =
671         new KVErrorHandler<>(
672             regionManager,
673             this,
674             lockResolverClient,
675             resp -> resp.hasRegionError() ? resp.getRegionError() : null,
676             resp -> null,
677             resolveLockResult -> addResolvedLocks(startTs, resolveLockResult.getResolvedLocks()),
678             startTs,
679             forWrite);
680     Coprocessor.Response resp =
681         callWithRetry(backOffer, TikvGrpc.getCoprocessorMethod(), reqToSend, handler);
682     return handleCopResponse(backOffer, resp, ranges, responseQueue, startTs);
683   }

685   // handleCopResponse checks coprocessor Response for region split and lock,
686   // returns more tasks when that happens, or handles the response if no error.
687   // if we're handling streaming coprocessor response, lastRange is the range of last
688   // successful response, otherwise it's nil.
689   private List<RangeSplitter.RegionTask> handleCopResponse(
690       BackOffer backOffer,
691       Coprocessor.Response response,
692       List<Coprocessor.KeyRange> ranges,
693       Queue<SelectResponse> responseQueue,
694       long startTs) {
695     boolean forWrite = false;
696     if (response == null) {
697       // Send request failed, reasons may:
698       // 1. TiKV down
699       // 2. Network partition
700       backOffer.doBackOff(
701           BackOffFunction.BackOffFuncType.BoRegionMiss,
702           new GrpcException("TiKV down or Network partition"));
703       logger.warn("Re-splitting region task due to region error: TiKV down or Network partition");
704       // Split ranges
705       return RangeSplitter.newSplitter(this.regionManager).splitRangeByRegion(ranges, storeType);
706     }
707 
708     if (response.hasRegionError()) {
709       Errorpb.Error regionError = response.getRegionError();
710       backOffer.doBackOff(
711           BackOffFunction.BackOffFuncType.BoRegionMiss, new GrpcException(regionError.toString()));
712       logger.warn("Re-splitting region task due to region error:" + regionError.getMessage());
713       // Split ranges
714       return RangeSplitter.newSplitter(this.regionManager).splitRangeByRegion(ranges, storeType);
715     }
716 
717     if (response.hasLocked()) {
718       Lock lock = new Lock(response.getLocked(), codec);
719       logger.debug(String.format("coprocessor encounters locks: %s", lock));
720       ResolveLockResult resolveLockResult =
721           lockResolverClient.resolveLocks(
722               backOffer, startTs, Collections.singletonList(lock), forWrite);
723       addResolvedLocks(startTs, resolveLockResult.getResolvedLocks());
724       long msBeforeExpired = resolveLockResult.getMsBeforeTxnExpired();
725       if (msBeforeExpired > 0) {
726         backOffer.doBackOffWithMaxSleep(BoTxnLockFast, msBeforeExpired, new LockException(lock));
727       }
728       // Split ranges
729       return RangeSplitter.newSplitter(this.regionManager).splitRangeByRegion(ranges, storeType);
730     }
731 
732     String otherError = response.getOtherError();
733     if (!otherError.isEmpty()) {
734       logger.warn(String.format("Other error occurred, message: %s", otherError));
735       throw new GrpcException(otherError);
736     }
737 
738     responseQueue.offer(doCoprocessor(response));
739     return null;
740   }

742   private Iterator<SelectResponse> doCoprocessor(StreamingResponse response) {
743     Iterator<Coprocessor.Response> responseIterator = response.iterator();
744     // If we got nothing to handle, return null
745     if (!responseIterator.hasNext()) {
746       return null;
747     }
748 
749     // Simply wrap it
750     return new Iterator<SelectResponse>() {
751       @Override
752       public boolean hasNext() {
753         return responseIterator.hasNext();
754       }
755 
756       @Override
757       public SelectResponse next() {
758         return doCoprocessor(responseIterator.next());
759       }
760     };
761   }

768       }
769       return selectResp;
770     } catch (InvalidProtocolBufferException e) {
771       throw new TiClientInternalException("Error parsing protobuf for coprocessor response.", e);
772     }
773   }
