74    protected AbstractRegionStoreClient(
75        TiConfiguration conf,
76        TiRegion region,
77        TiStore store,
78        ChannelFactory channelFactory,
79        TikvGrpc.TikvBlockingStub blockingStub,
80        TikvGrpc.TikvFutureStub asyncStub,
81        RegionManager regionManager) {
82      super(conf, channelFactory, blockingStub, asyncStub);
83      checkNotNull(region, "Region is empty");
84      checkNotNull(region.getLeader(), "Leader Peer is null");
85      checkArgument(region.getLeader() != null, "Leader Peer is null");
86      this.region = region;
87      this.regionManager = regionManager;
88      this.store = store;
89      this.codec = regionManager.getPDClient().getCodec();
90      if (this.store.getProxyStore() != null) {
91        this.timeout = conf.getForwardTimeout();
92      }
93    }

113   /**
114    * onNotLeader deals with NotLeaderError and returns whether re-splitting key range is needed
115    *
116    * @param newRegion the new region presented by NotLeader Error
117    * @return false when re-split is needed.
118    */
119   @Override
120   public boolean onNotLeader(TiRegion newRegion, BackOffer backOffer) {
121     if (logger.isDebugEnabled()) {
122       logger.debug(region + ", new leader = " + newRegion.getLeader().getStoreId());
123     }
124     // When switch leader fails or the region changed its region epoch,
125     // it would be necessary to re-split task's key range for new region.
126     if (!region.getRegionEpoch().equals(newRegion.getRegionEpoch())) {
127       return false;
128     }
129 
130     // If we try one peer but find the leader has not changed, we do not need to try other peers.
131     if (region.getLeader().getStoreId() == newRegion.getLeader().getStoreId()) {
132       store = null;
133     }
134     region = newRegion;
135     store = regionManager.getStoreById(region.getLeader().getStoreId(), backOffer);
136     updateClientStub();
137     return true;
138   }
139 
140   @Override
141   public boolean onStoreUnreachable(BackOffer backOffer) {
142     if (!store.isValid()) {
143       logger.warn(String.format("store [%d] has been invalid", store.getId()));
144       store = regionManager.getStoreById(store.getId(), backOffer);
145       updateClientStub();
146       return true;
147     }
148 
149     // seek an available leader store to send request
150     backOffer.checkTimeout();
151     Boolean result = seekLeaderStore(backOffer);
152     if (result != null) {
153       return result;
154     }
155     if (conf.getEnableGrpcForward()) {
156       // seek an available proxy store to forward request
157       backOffer.checkTimeout();
158       return seekProxyStore(backOffer);
159     }
160     return false;
161   }

163   private Kvrpcpb.Context addTraceId(Kvrpcpb.Context context, SlowLog slowLog) {
164     if (slowLog.getThresholdMS() < 0) {
165       // disable tikv tracing
166       return context;
167     }
168     long traceId = slowLog.getTraceId();
169     return Kvrpcpb.Context.newBuilder(context)
170         .setTraceContext(
171             Tracepb.TraceContext.newBuilder()
172                 .setDurationThresholdMs(
173                     (int) (slowLog.getThresholdMS() * conf.getRawKVServerSlowLogFactor()))
174                 .addRemoteParentSpans(Tracepb.RemoteParentSpan.newBuilder().setTraceId(traceId)))
175         .build();
176   }

178   protected Kvrpcpb.Context makeContext(TiStoreType storeType, SlowLog slowLog) {
179     Kvrpcpb.Context context = region.getReplicaContext(java.util.Collections.emptySet(), storeType);
180     return addTraceId(context, slowLog);
181   }

185     Kvrpcpb.Context context = region.getReplicaContext(resolvedLocks, storeType);

197   private void updateClientStub() {
198     String addressStr = store.getStore().getAddress();
199     long deadline = timeout;
200     if (store.getProxyStore() != null) {
201       addressStr = store.getProxyStore().getAddress();
202       deadline = conf.getForwardTimeout();
203     }
204     ManagedChannel channel =
205         channelFactory.getChannel(addressStr, regionManager.getPDClient().getHostMapping());
206     blockingStub =
207         TikvGrpc.newBlockingStub(channel).withDeadlineAfter(deadline, TimeUnit.MILLISECONDS);
208     asyncStub = TikvGrpc.newFutureStub(channel).withDeadlineAfter(deadline, TimeUnit.MILLISECONDS);
209     if (store.getProxyStore() != null) {
210       Metadata header = new Metadata();
211       header.put(TiConfiguration.FORWARD_META_DATA_KEY, store.getStore().getAddress());
212       blockingStub = MetadataUtils.attachHeaders(blockingStub, header);
213       asyncStub = MetadataUtils.attachHeaders(asyncStub, header);
214     }
215   }

217   private Boolean seekLeaderStore(BackOffer backOffer) {
218     Histogram.Timer switchLeaderDurationTimer =
219         SEEK_LEADER_STORE_DURATION
220             .labels(regionManager.getPDClient().getClusterId().toString())
221             .startTimer();
222     SlowLogSpan slowLogSpan = backOffer.getSlowLog().start("seekLeaderStore");
223     try {
224       List<Metapb.Peer> peers = region.getFollowerList();
225       if (peers.isEmpty()) {
226         // no followers available, retry
227         logger.warn(String.format("no followers of region[%d] available, retry", region.getId()));
228         regionManager.onRequestFail(region);
229         return false;
230       }
231 
232       logger.info(String.format("try switch leader: region[%d]", region.getId()));
233 
234       Metapb.Peer peer = switchLeaderStore(backOffer);
235       if (peer != null) {
236         // we found a leader
237         TiStore currentLeaderStore = regionManager.getStoreById(peer.getStoreId(), backOffer);
238         if (currentLeaderStore.isReachable()) {
239           logger.info(
240               String.format(
241                   "update leader using switchLeader logic from store[%d] to store[%d]",
242                   region.getLeader().getStoreId(), peer.getStoreId()));
243           // update region cache
244           TiRegion result = regionManager.updateLeader(region, peer.getStoreId());
245           if (result != null) {
246             region = result;
247             // switch to leader store
248             store = currentLeaderStore;
249             updateClientStub();
250             return true;
251           }
252           return false;
253         }
254       } else {
255         // no leader found, some response does not return normally, there may be network partition.
256         logger.warn(
257             String.format(
258                 "leader for region[%d] is not found, it is possible that network partition occurred",
259                 region.getId()));
260       }
261     } finally {
262       switchLeaderDurationTimer.observeDuration();
263       slowLogSpan.end();
264     }
265     return null;
266   }

268   private boolean seekProxyStore(BackOffer backOffer) {
269     SlowLogSpan slowLogSpan = backOffer.getSlowLog().start("seekProxyStore");
270     Histogram.Timer grpcForwardDurationTimer =
271         SEEK_PROXY_STORE_DURATION
272             .labels(regionManager.getPDClient().getClusterId().toString())
273             .startTimer();
274     try {
275       logger.info(String.format("try grpc forward: region[%d]", region.getId()));
276       // when current leader cannot be reached
277       TiStore storeWithProxy = switchProxyStore(backOffer);
278       if (storeWithProxy == null) {
279         // no store available, retry
280         logger.warn(String.format("No store available, retry: region[%d]", region.getId()));
281         return false;
282       }
283       // use proxy store to forward requests
284       regionManager.updateStore(store, storeWithProxy);
285       store = storeWithProxy;
286       updateClientStub();
287       return true;
288     } finally {
289       grpcForwardDurationTimer.observeDuration();
290       slowLogSpan.end();
291     }
292   }

294   // first: leader peer, second: true if any responses returned with grpc error
295   private Metapb.Peer switchLeaderStore(BackOffer backOffer) {
296     List<SwitchLeaderTask> responses = new LinkedList<>();
297     for (Metapb.Peer peer : region.getFollowerList()) {
298       ByteString key = region.getStartKey();
299       try {
300         TiStore peerStore = regionManager.getStoreById(peer.getStoreId(), backOffer);
301         ManagedChannel channel =
302             channelFactory.getChannel(
303                 peerStore.getAddress(), regionManager.getPDClient().getHostMapping());
304         TikvGrpc.TikvFutureStub stub =
305             TikvGrpc.newFutureStub(channel).withDeadlineAfter(timeout, TimeUnit.MILLISECONDS);
306         Kvrpcpb.RawGetRequest rawGetRequest =
307             Kvrpcpb.RawGetRequest.newBuilder()
308                 .setContext(makeContext(peer))
309                 .setKey(codec.encodeKey(key))
310                 .build();
311         ListenableFuture<Kvrpcpb.RawGetResponse> task = stub.rawGet(rawGetRequest);
312         responses.add(new SwitchLeaderTask(task, peer));
313       } catch (Exception e) {
314         logger.warn(
315             "switch region[{}] leader store to {} failed: {}",
316             region.getId(),
317             peer.getStoreId(),
318             e);
319       }
320     }
321     while (true) {
322       try {
323         Thread.sleep(2);
324       } catch (InterruptedException e) {
325         throw new GrpcException(e);
326       }
327       List<SwitchLeaderTask> unfinished = new LinkedList<>();
328       for (SwitchLeaderTask task : responses) {
329         if (!task.task.isDone()) {
330           unfinished.add(task);
331           continue;
332         }
333         try {
334           Kvrpcpb.RawGetResponse resp = task.task.get();
335           if (resp != null) {
336             if (!resp.hasRegionError()) {
337               // the peer is leader
338               logger.info(
339                   String.format("rawGet response indicates peer[%d] is leader", task.peer.getId()));
340               return task.peer;
341             }
342           }
343         } catch (Exception ignored) {
344         }
345       }
346       if (unfinished.isEmpty()) {
347         return null;
348       }
349       responses = unfinished;
350     }
351   }

353   private TiStore switchProxyStore(BackOffer backOffer) {
354     long forwardTimeout = conf.getForwardTimeout();
355     List<ForwardCheckTask> responses = new LinkedList<>();
356     for (Metapb.Peer peer : region.getFollowerList()) {
357       ByteString key = region.getStartKey();
358       try {
359         TiStore peerStore = regionManager.getStoreById(peer.getStoreId(), backOffer);
360         ManagedChannel channel =
361             channelFactory.getChannel(
362                 peerStore.getAddress(), regionManager.getPDClient().getHostMapping());
363         TikvGrpc.TikvFutureStub stub =
364             TikvGrpc.newFutureStub(channel)
365                 .withDeadlineAfter(forwardTimeout, TimeUnit.MILLISECONDS);
366         Metadata header = new Metadata();
367         header.put(TiConfiguration.FORWARD_META_DATA_KEY, store.getStore().getAddress());
368         Kvrpcpb.RawGetRequest rawGetRequest =
369             Kvrpcpb.RawGetRequest.newBuilder()
370                 .setContext(makeContext())
371                 .setKey(codec.encodeKey(key))
372                 .build();
373         ListenableFuture<Kvrpcpb.RawGetResponse> task =
374             MetadataUtils.attachHeaders(stub, header).rawGet(rawGetRequest);
375         responses.add(new ForwardCheckTask(task, peerStore.getStore()));
376       } catch (Exception e) {
377         logger.warn(
378             "switch region[{}] leader store to {} failed: {}",
379             region.getId(),
380             peer.getStoreId(),
381             e);
382       }
383     }
384     while (true) {
385       try {
386         Thread.sleep(2);
387       } catch (InterruptedException e) {
388         throw new GrpcException(e);
389       }
390       List<ForwardCheckTask> unfinished = new LinkedList<>();
391       for (ForwardCheckTask task : responses) {
392         if (!task.task.isDone()) {
393           unfinished.add(task);
394           continue;
395         }
396         try {
397           // any answer will do
398           Kvrpcpb.RawGetResponse resp = task.task.get();
399           logger.info(
400               String.format(
401                   "rawGetResponse indicates forward from [%s] to [%s]",
402                   task.store.getAddress(), store.getAddress()));
403           return store.withProxy(task.store);
404         } catch (Exception ignored) {
405         }
406       }
407       if (unfinished.isEmpty()) {
408         return null;
409       }
410       responses = unfinished;
411     }
412   }
