34  public abstract class AbstractDateTimeType extends DataType {
35  
36    AbstractDateTimeType(InternalTypeHolder holder) {
37      super(holder);
38    }
39  
40    AbstractDateTimeType(MySQLType tp) {
41      super(tp);
42    }
43  
44    public abstract DateTimeZone getTimezone();
45    /**
46     * Decode DateTime from packed long value In TiDB / MySQL, timestamp type is converted to UTC and
47     * stored
48     */
49    long decodeDateTime(int flag, CodecDataInput cdi) {
50      ExtendedDateTime extendedDateTime;
51      if (flag == Codec.UVARINT_FLAG) {
52        extendedDateTime = DateTimeCodec.readFromUVarInt(cdi, getTimezone());
53      } else if (flag == Codec.UINT_FLAG) {
54        extendedDateTime = DateTimeCodec.readFromUInt(cdi, getTimezone());
55      } else {
56        throw new InvalidCodecFormatException(
57            "Invalid Flag type for " + getClass().getSimpleName() + ": " + flag);
58      }
59  
60      // Even though null is filtered out but data like 0000-00-00 exists
61      // according to MySQL JDBC behavior, it can chose the **ROUND** behavior converted to the
62      // nearest
63      // value which is 0001-01-01.
64      if (extendedDateTime == null) {
65        extendedDateTime = DateTimeCodec.createExtendedDateTime(getTimezone(), 1, 1, 1, 0, 0, 0, 0);
66      }
67      Timestamp ts = extendedDateTime.toTimeStamp();
68      return ts.getTime() / 1000 * 1000000 + ts.getNanos() / 1000;
69    }
70  
71    Timestamp decodeDateTimeForBatchWrite(int flag, CodecDataInput cdi) {
72      ExtendedDateTime extendedDateTime;
73      if (flag == Codec.UVARINT_FLAG) {
74        extendedDateTime = DateTimeCodec.readFromUVarInt(cdi, getTimezone());
75      } else if (flag == Codec.UINT_FLAG) {
76        extendedDateTime = DateTimeCodec.readFromUInt(cdi, getTimezone());
77      } else {
78        throw new InvalidCodecFormatException(
79            "Invalid Flag type for " + getClass().getSimpleName() + ": " + flag);
80      }
81      if (extendedDateTime == null) {
82        return DateTimeCodec.createExtendedDateTime(getTimezone(), 1, 1, 1, 0, 0, 0, 0).toTimeStamp();
83      }
84      return extendedDateTime.toTimeStamp();
85    }
86  
87    /** Decode Date from packed long value */
88    LocalDate decodeDate(int flag, CodecDataInput cdi) {
89      LocalDate date;
90      if (flag == Codec.UVARINT_FLAG) {
91        date = DateCodec.readFromUVarInt(cdi);
92      } else if (flag == Codec.UINT_FLAG) {
93        date = DateCodec.readFromUInt(cdi);
94      } else {
95        throw new InvalidCodecFormatException(
96            "Invalid Flag type for " + getClass().getSimpleName() + ": " + flag);
97      }
98      return date;
99    }
100 
101   /** {@inheritDoc} */
102   @Override
103   protected void encodeKey(CodecDataOutput cdo, Object value) {
104     ExtendedDateTime edt = Converter.convertToDateTime(value);
105     DateTimeCodec.writeDateTimeFully(cdo, edt, getTimezone());
106   }
107 
108   /** {@inheritDoc} */
109   @Override
110   protected void encodeValue(CodecDataOutput cdo, Object value) {
111     encodeKey(cdo, value);
112   }
113 
114   /** {@inheritDoc} */
115   @Override
116   protected void encodeProto(CodecDataOutput cdo, Object value) {
117     ExtendedDateTime edt = Converter.convertToDateTime(value);
118     DateTimeCodec.writeDateTimeProto(cdo, edt, getTimezone());
119   }
120 
121   @Override
122   public ExprType getProtoExprType() {
123     return ExprType.MysqlTime;
124   }
125 
126   Timestamp convertToMysqlDateTime(Object value) throws ConvertNotSupportException {
127     Timestamp result;
128     if (value instanceof String) {
129       result = Timestamp.valueOf((String) value);
130     } else if (value instanceof java.sql.Date) {
131       result = new Timestamp(((java.sql.Date) value).getTime());
132     } else if (value instanceof Timestamp) {
133       result = (Timestamp) value;
134     } else {
135       throw new ConvertNotSupportException(value.getClass().getName(), this.getClass().getName());
136     }
137     return result;
138   }
139 }
