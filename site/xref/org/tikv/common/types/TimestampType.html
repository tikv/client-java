34  /**
35   * Timestamp in TiDB is represented as packed long including year/month and etc. When stored, it is
36   * converted into UTC and when read it should be interpreted as UTC and convert to local When
37   * encoded as coprocessor request 1. all date time in key should be converted to UTC 2. all date
38   * time in proto should be converted into local timezone 3. all incoming data should be regarded as
39   * local timezone if no timezone indicated
40   *
41   * <p>For example, Encoding: If incoming literal is a string '2000-01-01 00:00:00' which has no
42   * timezone, it interpreted as text in local timezone and encoded with UTC; If incoming literal is a
43   * epoch millisecond, it interpreted as UTC epoch and encode with UTC if index / key or local
44   * timezone if proto
45   */
46  public class TimestampType extends AbstractDateTimeType {
47    public static final TimestampType TIMESTAMP = new TimestampType(MySQLType.TypeTimestamp);
48  
49    public static final MySQLType[] subTypes = new MySQLType[] {MySQLType.TypeTimestamp};
50  
51    TimestampType(MySQLType tp) {
52      super(tp);
53    }
54  
55    TimestampType(TiColumnInfo.InternalTypeHolder holder) {
56      super(holder);
57    }
58  
59    public DateTimeZone getTimezone() {
60      return DateTimeZone.UTC;
61    }
62  
63    @Override
64    protected Object doConvertToTiDBType(Object value)
65        throws ConvertNotSupportException, ConvertOverflowException {
66      return convertToMysqlLocalTimestamp(value);
67    }
68  
69    private Timestamp convertToMysqlLocalTimestamp(Object value) throws ConvertNotSupportException {
70      Timestamp result;
71      if (value instanceof Long) {
72        throw new ConvertNotSupportException(value.getClass().getName(), this.getClass().getName());
73        // result = new java.sql.Timestamp((Long) value);
74      } else {
75        return convertToMysqlDateTime(value);
76      }
77    }
78  
79    /**
80     * Decode timestamp from packed long value In TiDB / MySQL, timestamp type is converted to UTC and
81     * stored
82     */
83    @Override
84    protected Long decodeNotNull(int flag, CodecDataInput cdi) {
85      return decodeDateTime(flag, cdi);
86    }
87  
88    @Override
89    protected Timestamp decodeNotNullForBatchWrite(int flag, CodecDataInput cdi) {
90      return decodeDateTimeForBatchWrite(flag, cdi);
91    }
92  
93    @Override
94    public DateTime getOriginDefaultValueNonNull(String value, long version) {
95      if (version >= DataType.COLUMN_VERSION_FLAG) {
96        LocalDateTime localDateTime =
97            Converter.strToDateTime(value, UTC_TIME_FORMATTER)
98                .withZone(DateTimeZone.getDefault())
99                .toLocalDateTime();
100       return localDateTime.toDateTime();
101     }
102     return Converter.convertToDateTime(value).getDateTime();
103   }
104 
105   /** {@inheritDoc} */
106   @Override
107   protected void encodeProto(CodecDataOutput cdo, Object value) {
108     ExtendedDateTime localExtendedDateTime = Converter.convertToDateTime(value);
109     DateTime utcDateTime = localExtendedDateTime.getDateTime().toDateTime(DateTimeZone.UTC);
110     ExtendedDateTime utcExtendedDateTime =
111         new ExtendedDateTime(utcDateTime, localExtendedDateTime.getMicrosOfMillis());
112     DateTimeCodec.writeDateTimeProto(cdo, utcExtendedDateTime, Converter.getLocalTimezone());
113   }
114 
115   @Override
116   public String getName() {
117     return "TIMESTAMP";
118   }
119 }
