18  package org.tikv.common.types;
19  
20  public class Charset {
21    // CharsetBin is used for marking binary charset.
22    public static final String CharsetBin = "binary";
23    // CollationBin is the default collation for CharsetBin.
24    public static final String CollationBin = "binary";
25    // CharsetUTF8 is the default charset for string types.
26    public static final String CharsetUTF8 = "utf8";
27    // CollationUTF8 is the default collation for CharsetUTF8.
28    public static final String CollationUTF8 = "utf8_bin";
29    // CharsetUTF8MB4 represents 4 bytes utf8, which works the same way as utf8 in Go.
30    public static final String CharsetUTF8MB4 = "utf8mb4";
31    // CollationUTF8MB4 is the default collation for CharsetUTF8MB4.
32    public static final String CollationUTF8MB4 = "utf8mb4_bin";
33    // CharsetASCII is a subset of UTF8.
34    public static final String CharsetASCII = "ascii";
35    // CollationASCII is the default collation for CharsetACSII.
36    public static final String CollationASCII = "ascii_bin";
37    // CharsetLatin1 is a single byte charset.
38    public static final String CharsetLatin1 = "latin1";
39    // CollationLatin1 is the default collation for CharsetLatin1.
40    public static final String CollationLatin1 = "latin1_bin";
41  }
