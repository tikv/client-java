174   public synchronized void keepPauseChecker(PDChecker checker) {
175     if (!this.pauseCheckerService.containsKey(checker)) {
176       ScheduledExecutorService newService =
177           Executors.newSingleThreadScheduledExecutor(
178               new ThreadFactoryBuilder()
179                   .setNameFormat(String.format("PDClient-pause-%s-pool-%%d", checker.name()))
180                   .setDaemon(true)
181                   .build());
182       newService.scheduleAtFixedRate(
183           () -> pauseChecker(checker, PAUSE_CHECKER_TIMEOUT),
184           0,
185           KEEP_CHECKER_PAUSE_PERIOD,
186           TimeUnit.SECONDS);
187       this.pauseCheckerService.put(checker, newService);
188     }
189   }

191   public synchronized void stopKeepPauseChecker(PDChecker checker) {
192     if (this.pauseCheckerService.containsKey(checker)) {
193       this.pauseCheckerService.get(checker).shutdown();
194       this.pauseCheckerService.remove(checker);
195     }
196   }

202   private void pauseChecker(PDChecker checker, int timeout) {
203     String verb = timeout == 0 ? "resume" : "pause";
204     URI url = pdAddrs.get(0);
205     String api = url.toString() + "/pd/api/v1/checker/" + checker.apiName();
206     HashMap<String, Integer> arguments = new HashMap<>();
207     arguments.put("delay", timeout);
208     try (CloseableHttpClient client = HttpClients.createDefault()) {
209       JsonMapper jsonMapper = new JsonMapper();
210       byte[] body = jsonMapper.writeValueAsBytes(arguments);
211       HttpPost post = new HttpPost(api);
212       post.setEntity(new ByteArrayEntity(body));
213       try (CloseableHttpResponse resp = client.execute(post)) {
214         if (resp.getStatusLine().getStatusCode() != 200) {
215           logger.error("failed to {} checker.", verb);
216         }
217         logger.info("checker {} {}d", checker.apiName(), verb);
218       }
219     } catch (Exception e) {
220       logger.error(String.format("failed to %s checker.", verb), e);
221     }
222   }

262   /**
263    * wait scatter region until finish
264    *
265    * @param region
266    */
267   void waitScatterRegionFinish(Metapb.Region region, BackOffer backOffer) {
268     for (; ; ) {
269       GetOperatorResponse resp = getOperator(region.getId());
270       if (resp != null) {
271         if (isScatterRegionFinish(resp)) {
272           logger.info(String.format("wait scatter region on %d is finished", region.getId()));
273           return;
274         } else {
275           backOffer.doBackOff(
276               BackOffFuncType.BoRegionMiss, new GrpcException("waiting scatter region"));
277           logger.info(
278               String.format(
279                   "wait scatter region %d at key %s is %s",
280                   region.getId(),
281                   KeyUtils.formatBytes(resp.getDesc().toByteArray()),
282                   resp.getStatus()));
283         }
284       }
285     }
286   }

289     Supplier<GetOperatorRequest> request =
290         () -> GetOperatorRequest.newBuilder().setHeader(header).setRegionId(regionId).build();
291     // get operator no need to handle error and no need back offer.
292     return callWithRetry(
293         ConcreteBackOffer.newCustomBackOff(0, getClusterId()),
294         PDGrpc.getGetOperatorMethod(),
295         request,
296         new NoopHandler<>());
297   }

299   private boolean isScatterRegionFinish(GetOperatorResponse resp) {
300     // If the current operator of region is not `scatter-region`, we could assume
301     // that `scatter-operator` has finished or timeout.
302     boolean finished =
303         !resp.getDesc().equals(ByteString.copyFromUtf8("scatter-region"))
304             || resp.getStatus() != OperatorStatus.RUNNING;
305 
306     if (resp.hasHeader()) {
307       ResponseHeader header = resp.getHeader();
308       if (header.hasError()) {
309         Error error = header.getError();
310         // heartbeat may not send to PD
311         if (error.getType() == ErrorType.REGION_NOT_FOUND) {
312           finished = true;
313         }
314       }
315     }
316     return finished;
317   }

320   public Pair<Metapb.Region, Metapb.Peer> getRegionByKey(BackOffer backOffer, ByteString key) {
321     Histogram.Timer requestTimer =
322         PD_GET_REGION_BY_KEY_REQUEST_LATENCY.labels(getClusterId().toString()).startTimer();
323     try {

332           new PDErrorHandler<>(getRegionResponseErrorExtractor, this);

334       GetRegionResponse resp =
335           callWithRetry(backOffer, PDGrpc.getGetRegionMethod(), request, handler);
336       return new Pair<>(codec.decodeRegion(resp.getRegion()), resp.getLeader());
337     } finally {
338       requestTimer.observeDuration();
339     }

355   @Override
356   public List<Pdpb.Region> scanRegions(
357       BackOffer backOffer, ByteString startKey, ByteString endKey, int limit) {
358     // no need to backoff because ScanRegions is just for optimization
359     // introduce a warm-up timeout for ScanRegions requests
360     PDGrpc.PDBlockingStub stub =
361         getBlockingStub().withDeadlineAfter(conf.getWarmUpTimeout(), TimeUnit.MILLISECONDS);
362     Pair<ByteString, ByteString> range = codec.encodePdQueryRange(startKey, endKey);
363     Pdpb.ScanRegionsRequest request =
364         Pdpb.ScanRegionsRequest.newBuilder()
365             .setHeader(header)
366             .setStartKey(range.first)
367             .setEndKey(range.second)
368             .setLimit(limit)
369             .build();
370     Pdpb.ScanRegionsResponse resp = stub.scanRegions(request);
371     if (resp == null) {
372       return null;
373     }
374 
375     return codec.decodePdRegions(resp.getRegionsList());
376   }

448   private GetMembersResponse doGetMembers(BackOffer backOffer, URI uri) {
449     while (true) {
450       backOffer.checkTimeout();
451 
452       try {
453         ManagedChannel probChan = channelFactory.getChannel(uriToAddr(uri), hostMapping);
454         PDGrpc.PDBlockingStub stub =
455             PDGrpc.newBlockingStub(probChan).withDeadlineAfter(getTimeout(), TimeUnit.MILLISECONDS);
456         GetMembersRequest request =
457             GetMembersRequest.newBuilder().setHeader(RequestHeader.getDefaultInstance()).build();
458         GetMembersResponse resp = stub.getMembers(request);
459         // check if the response contains a valid leader
460         if (resp != null && resp.getLeader().getMemberId() == 0) {
461           return null;
462         }
463         return resp;
464       } catch (Exception e) {
465         logger.warn(
466             "failed to get member from pd server from {}, caused by: {}", uri, e.getMessage());
467         backOffer.doBackOff(BackOffFuncType.BoPDRPC, e);
468       }
469     }
470   }

480   // return whether the leader has changed to target address `leaderUrlStr`.
481   synchronized boolean trySwitchLeader(String leaderUrlStr) {
482     if (pdClientWrapper != null) {
483       if (leaderUrlStr.equals(pdClientWrapper.getLeaderInfo())) {
484         // The message to leader is not forwarded by follower.
485         if (leaderUrlStr.equals(pdClientWrapper.getStoreAddress())) {
486           return true;
487         }
488       }
489       // If leader has transferred to another member, we can create another leaderWrapper.
490     }
491     // switch leader
492     return createLeaderClientWrapper(leaderUrlStr);
493   }

495   private synchronized boolean createLeaderClientWrapper(String leaderUrlStr) {
496     try {
497       // create new Leader
498       ManagedChannel clientChannel = channelFactory.getChannel(leaderUrlStr, hostMapping);
499       pdClientWrapper =
500           new PDClientWrapper(leaderUrlStr, leaderUrlStr, clientChannel, System.nanoTime());
501       timeout = conf.getTimeout();
502     } catch (IllegalArgumentException e) {
503       return false;
504     }
505     logger.info(String.format("Switched to new leader: %s", pdClientWrapper));
506     return true;
507   }

509   synchronized boolean createFollowerClientWrapper(
510       BackOffer backOffer, String followerUrlStr, String leaderUrls) {
511     // TODO: Why not strip protocol info on server side since grpc does not need it
512 
513     try {
514       if (!checkHealth(backOffer, followerUrlStr, hostMapping)) {
515         return false;
516       }
517 
518       // create new Leader
519       ManagedChannel channel = channelFactory.getChannel(followerUrlStr, hostMapping);
520       pdClientWrapper = new PDClientWrapper(leaderUrls, followerUrlStr, channel, System.nanoTime());
521       timeout = conf.getForwardTimeout();
522     } catch (IllegalArgumentException e) {
523       return false;
524     }
525     logger.info(String.format("Switched to new leader by follower forward: %s", pdClientWrapper));
526     return true;
527   }

529   public void tryUpdateLeaderOrForwardFollower() {
530     if (updateLeaderNotify.compareAndSet(false, true)) {
531       try {
532         updateLeaderService.submit(
533             () -> {
534               try {
535                 updateLeaderOrForwardFollower();
536               } catch (Exception e) {
537                 logger.info("update leader or forward follower failed", e);
538                 throw e;
539               } finally {
540                 updateLeaderNotify.set(false);
541                 logger.info("updating leader finish");
542               }
543             });
544       } catch (RejectedExecutionException e) {
545         logger.error("PDClient is shutdown", e);
546         updateLeaderNotify.set(false);
547       }
548     }
549   }

551   private synchronized void updateLeaderOrForwardFollower() {
552     logger.warn("updating leader or forward follower");
553     if (System.currentTimeMillis() - lastUpdateLeaderTime < MIN_TRY_UPDATE_DURATION) {
554       return;
555     }
556     for (URI url : this.pdAddrs) {
557       BackOffer backOffer = this.probeBackOffer();
558       // since resp is null, we need update leader's address by walking through all pd server.
559       GetMembersResponse resp = getMembers(backOffer, url);
560       if (resp == null) {
561         continue;
562       }
563       if (resp.getLeader().getClientUrlsList().isEmpty()) {
564         continue;
565       }
566 
567       String leaderUrlStr = resp.getLeader().getClientUrlsList().get(0);
568       leaderUrlStr = uriToAddr(addrToUri(leaderUrlStr));
569 
570       // if leader is switched, just return.
571       if (checkHealth(backOffer, leaderUrlStr, hostMapping)
572           && createLeaderClientWrapper(leaderUrlStr)) {
573         lastUpdateLeaderTime = System.currentTimeMillis();
574         return;
575       }
576 
577       if (!conf.getEnableGrpcForward()) {
578         continue;
579       }
580 
581       logger.info(String.format("can not switch to new leader, try follower forward"));
582       List<Pdpb.Member> members = resp.getMembersList();
583 
584       // If we have not used follower forward, try the first follower.
585       boolean hasReachNextMember =
586           pdClientWrapper != null && pdClientWrapper.getStoreAddress().equals(leaderUrlStr);
587 
588       for (int i = 0; i < members.size() * 2; i++) {
589         Pdpb.Member member = members.get(i % members.size());
590         if (member.getMemberId() == resp.getLeader().getMemberId()) {
591           continue;
592         }
593         String followerUrlStr = member.getClientUrlsList().get(0);
594         followerUrlStr = uriToAddr(addrToUri(followerUrlStr));
595         if (pdClientWrapper != null && pdClientWrapper.getStoreAddress().equals(followerUrlStr)) {
596           hasReachNextMember = true;
597           continue;
598         }
599         if (hasReachNextMember
600             && createFollowerClientWrapper(backOffer, followerUrlStr, leaderUrlStr)) {
601           logger.warn(
602               String.format("forward request to pd [%s] by pd [%s]", leaderUrlStr, followerUrlStr));
603           return;
604         }
605       }
606     }
607     lastUpdateLeaderTime = System.currentTimeMillis();
608     if (pdClientWrapper == null) {
609       throw new TiClientInternalException(
610           "already tried all address on file, but not leader found yet.");
611     }
612   }

614   public void tryUpdateLeader() {
615     logger.info("try update leader");
616     for (URI url : this.pdAddrs) {
617       BackOffer backOffer = this.probeBackOffer();
618       // since resp is null, we need update leader's address by walking through all pd server.
619       GetMembersResponse resp = getMembers(backOffer, url);
620       if (resp == null) {
621         continue;
622       }
623       List<URI> urls =
624           resp.getMembersList()
625               .stream()
626               .map(mem -> addrToUri(mem.getClientUrls(0)))
627               .collect(Collectors.toList());
628       String leaderUrlStr = resp.getLeader().getClientUrlsList().get(0);
629       leaderUrlStr = uriToAddr(addrToUri(leaderUrlStr));
630 
631       // If leader is not change but becomes available, we can cancel follower forward.
632       if (checkHealth(backOffer, leaderUrlStr, hostMapping) && trySwitchLeader(leaderUrlStr)) {
633         if (!urls.equals(this.pdAddrs)) {
634           tryUpdateMembers(urls);
635         }
636         return;
637       }
638     }
639     lastUpdateLeaderTime = System.currentTimeMillis();
640     if (pdClientWrapper == null) {
641       throw new TiClientInternalException(
642           "already tried all address on file, but not leader found yet.");
643     }
644   }

650   public void updateTiFlashReplicaStatus() {
651     ByteSequence prefix =
652         ByteSequence.from(TIFLASH_TABLE_SYNC_PROGRESS_PATH, StandardCharsets.UTF_8);
653     for (int i = 0; i < 5; i++) {
654       CompletableFuture<GetResponse> resp;
655       try {
656         resp =
657             etcdClient.getKVClient().get(prefix, GetOption.newBuilder().withPrefix(prefix).build());
658       } catch (Exception e) {
659         logger.info("get tiflash table replica sync progress failed, continue checking.", e);
660         continue;
661       }
662       GetResponse getResp;
663       try {
664         getResp = resp.get();
665       } catch (InterruptedException e) {
666         Thread.currentThread().interrupt();
667         continue;
668       } catch (ExecutionException e) {
669         throw new GrpcException("failed to update tiflash replica", e);
670       }
671       ConcurrentMap<Long, Double> progressMap = new ConcurrentHashMap<>();
672       for (KeyValue kv : getResp.getKvs()) {
673         long tableId;
674         try {
675           tableId =
676               Long.parseLong(
677                   kv.getKey().toString().substring(TIFLASH_TABLE_SYNC_PROGRESS_PATH.length()));
678         } catch (Exception e) {
679           logger.info(
680               "invalid tiflash table replica sync progress key. key = " + kv.getKey().toString());
681           continue;
682         }
683         double progress;
684         try {
685           progress = Double.parseDouble(kv.getValue().toString());
686         } catch (Exception e) {
687           logger.info(
688               "invalid tiflash table replica sync progress value. value = "
689                   + kv.getValue().toString());
690           continue;
691         }
692         progressMap.put(tableId, progress);
693       }
694       tiflashReplicaMap = progressMap;
695       break;
696     }
697   }

699   public double getTiFlashReplicaProgress(long tableId) {
700     return tiflashReplicaMap.getOrDefault(tableId, 0.0);
701   }

719   private void initCluster() {
720     logger.info("init cluster: start");
721     GetMembersResponse resp = null;
722     List<URI> pdAddrs = new ArrayList<>(getConf().getPdAddrs());
723     // shuffle PD addresses so that clients call getMembers from different PD
724     Collections.shuffle(pdAddrs);
725     this.pdAddrs = pdAddrs;
726     this.etcdClient =
727         Client.builder()
728             .endpoints(pdAddrs)
729             .executorService(
730                 Executors.newCachedThreadPool(
731                     new ThreadFactoryBuilder()
732                         .setNameFormat("etcd-conn-manager-pool-%d")
733                         .setDaemon(true)
734                         .build()))
735             .build();
736     logger.info("init host mapping: start");
737     this.hostMapping =
738         Optional.ofNullable(getConf().getHostMapping())
739             .orElseGet(() -> new DefaultHostMapping(this.etcdClient, conf.getNetworkMappingName()));
740     logger.info("init host mapping: end");
741     // The first request may cost too much latency
742     long originTimeout = this.timeout;
743     this.timeout = conf.getPdFirstGetMemberTimeout();
744     for (URI u : pdAddrs) {
745       logger.info("get members with pd " + u + ": start");
746       resp = getMembers(defaultBackOffer(), u);
747       logger.info("get members with pd " + u + ": end");
748       if (resp != null) {
749         break;
750       }
751     }
752     if (resp == null) {
753       logger.error("Could not get leader member with: " + pdAddrs);
754     }
755 
756     this.timeout = originTimeout;
757     checkNotNull(resp, "Failed to init client for PD cluster.");
758     long clusterId = resp.getHeader().getClusterId();
759     header = RequestHeader.newBuilder().setClusterId(clusterId).build();
760     tsoReq = TsoRequest.newBuilder().setHeader(header).setCount(1).build();
761     this.tiflashReplicaMap = new ConcurrentHashMap<>();
762     this.pdAddrs =
763         resp.getMembersList()
764             .stream()
765             .map(mem -> addrToUri(mem.getClientUrls(0)))
766             .collect(Collectors.toList());
767     logger.info("init cluster with address: " + this.pdAddrs);
768 
769     String leaderUrlStr = resp.getLeader().getClientUrls(0);
770     leaderUrlStr = uriToAddr(addrToUri(leaderUrlStr));
771     logger.info("createLeaderClientWrapper with leader " + leaderUrlStr + ": start");
772     createLeaderClientWrapper(leaderUrlStr);
773     logger.info("createLeaderClientWrapper with leader " + leaderUrlStr + ": end");
774     service =
775         Executors.newSingleThreadScheduledExecutor(
776             new ThreadFactoryBuilder()
777                 .setNameFormat("PDClient-update-leader-pool-%d")
778                 .setDaemon(true)
779                 .build());
780     service.scheduleAtFixedRate(
781         () -> {
782           // Wrap this with a try catch block in case schedule update fails
783           try {
784             tryUpdateLeader();
785           } catch (Exception e) {
786             logger.warn("Update leader failed", e);
787           }
788         },
789         10,
790         10,
791         TimeUnit.SECONDS);
792     if (conf.isTiFlashEnabled()) {
793       tiflashReplicaService =
794           Executors.newSingleThreadScheduledExecutor(
795               new ThreadFactoryBuilder()
796                   .setNameFormat("PDClient-tiflash-replica-pool-%d")
797                   .setDaemon(true)
798                   .build());
799       tiflashReplicaService.scheduleAtFixedRate(
800           this::updateTiFlashReplicaStatus, 10, 10, TimeUnit.SECONDS);
801     }
802     logger.info("init cluster: finish");
803   }
