44  public abstract class AbstractGRPCClient<
45          BlockingStubT extends AbstractStub<BlockingStubT>,
46          FutureStubT extends AbstractFutureStub<FutureStubT>>
47      implements AutoCloseable {
48    protected final Logger logger = LoggerFactory.getLogger(this.getClass());
49    protected final ChannelFactory channelFactory;
50    protected TiConfiguration conf;
51    protected long timeout;
52    protected BlockingStubT blockingStub;
53    protected FutureStubT asyncStub;
54  
55    protected AbstractGRPCClient(TiConfiguration conf, ChannelFactory channelFactory) {
56      this.conf = conf;
57      this.timeout = conf.getTimeout();
58      this.channelFactory = channelFactory;
59    }
60  
61    protected AbstractGRPCClient(
62        TiConfiguration conf,
63        ChannelFactory channelFactory,
64        BlockingStubT blockingStub,
65        FutureStubT asyncStub) {
66      this.conf = conf;
67      this.timeout = conf.getTimeout();
68      this.channelFactory = channelFactory;
69      this.blockingStub = blockingStub;
70      this.asyncStub = asyncStub;
71    }

73    public TiConfiguration getConf() {
74      return conf;
75    }

77    // TODO: Seems a little bit messy for lambda part
78    public <ReqT, RespT> RespT callWithRetry(
79        BackOffer backOffer,
80        MethodDescriptor<ReqT, RespT> method,
81        Supplier<ReqT> requestFactory,
82        ErrorHandler<RespT> handler) {
83      if (logger.isTraceEnabled()) {
84        logger.trace(String.format("Calling %s...", method.getFullMethodName()));
85      }
86      RetryPolicy<RespT> policy = new Builder<RespT>(backOffer).create(handler);
87      RespT resp =
88          policy.callWithRetry(
89              () -> {
90                BlockingStubT stub = getBlockingStub();
91                return ClientCalls.blockingUnaryCall(
92                    stub.getChannel(), method, stub.getCallOptions(), requestFactory.get());
93              },
94              method.getFullMethodName(),
95              backOffer);
96  
97      if (logger.isTraceEnabled()) {
98        logger.trace(String.format("leaving %s...", method.getFullMethodName()));
99      }
100     return resp;
101   }
102 
103   protected <ReqT, RespT> void callAsyncWithRetry(
104       BackOffer backOffer,
105       MethodDescriptor<ReqT, RespT> method,
106       Supplier<ReqT> requestFactory,
107       StreamObserver<RespT> responseObserver,
108       ErrorHandler<RespT> handler) {
109     logger.debug(String.format("Calling %s...", method.getFullMethodName()));
110 
111     RetryPolicy<RespT> policy = new Builder<RespT>(backOffer).create(handler);
112     policy.callWithRetry(
113         () -> {
114           FutureStubT stub = getAsyncStub();
115           ClientCalls.asyncUnaryCall(
116               stub.getChannel().newCall(method, stub.getCallOptions()),
117               requestFactory.get(),
118               responseObserver);
119           return null;
120         },
121         method.getFullMethodName(),
122         backOffer);
123     logger.debug(String.format("leaving %s...", method.getFullMethodName()));
124   }

126   <ReqT, RespT> StreamObserver<ReqT> callBidiStreamingWithRetry(
127       BackOffer backOffer,
128       MethodDescriptor<ReqT, RespT> method,
129       StreamObserver<RespT> responseObserver,
130       ErrorHandler<StreamObserver<ReqT>> handler) {
131     logger.debug(String.format("Calling %s...", method.getFullMethodName()));
132 
133     RetryPolicy<StreamObserver<ReqT>> policy =
134         new Builder<StreamObserver<ReqT>>(backOffer).create(handler);
135     StreamObserver<ReqT> observer =
136         policy.callWithRetry(
137             () -> {
138               FutureStubT stub = getAsyncStub();
139               return asyncBidiStreamingCall(
140                   stub.getChannel().newCall(method, stub.getCallOptions()), responseObserver);
141             },
142             method.getFullMethodName(),
143             backOffer);
144     logger.debug(String.format("leaving %s...", method.getFullMethodName()));
145     return observer;
146   }

148   public <ReqT, RespT> StreamingResponse callServerStreamingWithRetry(
149       BackOffer backOffer,
150       MethodDescriptor<ReqT, RespT> method,
151       Supplier<ReqT> requestFactory,
152       ErrorHandler<StreamingResponse> handler) {
153     logger.debug(String.format("Calling %s...", method.getFullMethodName()));
154 
155     RetryPolicy<StreamingResponse> policy =
156         new Builder<StreamingResponse>(backOffer).create(handler);
157     StreamingResponse response =
158         policy.callWithRetry(
159             () -> {
160               BlockingStubT stub = getBlockingStub();
161               return new StreamingResponse(
162                   blockingServerStreamingCall(
163                       stub.getChannel(), method, stub.getCallOptions(), requestFactory.get()));
164             },
165             method.getFullMethodName(),
166             backOffer);
167     logger.debug(String.format("leaving %s...", method.getFullMethodName()));
168     return response;
169   }

183   private boolean doCheckHealth(BackOffer backOffer, String addressStr, HostMapping hostMapping) {
184     while (true) {
185       backOffer.checkTimeout();
186 
187       try {
188         ManagedChannel channel = channelFactory.getChannel(addressStr, hostMapping);
189         HealthGrpc.HealthBlockingStub stub =
190             HealthGrpc.newBlockingStub(channel)
191                 .withDeadlineAfter(getTimeout(), TimeUnit.MILLISECONDS);
192         HealthCheckRequest req = HealthCheckRequest.newBuilder().build();
193         HealthCheckResponse resp = stub.check(req);
194         return resp.getStatus() == HealthCheckResponse.ServingStatus.SERVING;
195       } catch (Exception e) {
196         logger.warn("check health failed, addr: {}, caused by: {}", addressStr, e.getMessage());
197         backOffer.doBackOff(BackOffFuncType.BoCheckHealth, e);
198       }
199     }
200   }
201 
202   protected boolean checkHealth(BackOffer backOffer, String addressStr, HostMapping hostMapping) {
203     try {
204       return doCheckHealth(backOffer, addressStr, hostMapping);
205     } catch (Exception e) {
206       return false;
207     }
208   }
