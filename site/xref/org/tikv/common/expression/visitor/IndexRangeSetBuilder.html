18  package org.tikv.common.expression.visitor;
19  
20  import com.google.common.collect.Range;
21  import com.google.common.collect.RangeSet;
22  import com.google.common.collect.TreeRangeSet;
23  import java.util.HashMap;
24  import java.util.Map;
25  import org.tikv.common.expression.ColumnRef;
26  import org.tikv.common.expression.ComparisonBinaryExpression;
27  import org.tikv.common.expression.ComparisonBinaryExpression.NormalizedPredicate;
28  import org.tikv.common.expression.StringRegExpression;
29  import org.tikv.common.key.TypedKey;
30  import org.tikv.common.meta.TiIndexColumn;
31  import org.tikv.common.meta.TiIndexInfo;
32  import org.tikv.common.meta.TiTableInfo;
33  import org.tikv.common.types.DataType;
34  
35  public class IndexRangeSetBuilder extends RangeSetBuilder<TypedKey> {
36  
37    private final Map<ColumnRef, Integer> lengths; // length of corresponding ColumnRef
38  
39    public IndexRangeSetBuilder(TiTableInfo table, TiIndexInfo index) {
40      Map<ColumnRef, Integer> result = new HashMap<>();
41      if (table != null && index != null) {
42        for (TiIndexColumn indexColumn : index.getIndexColumns()) {
43          ColumnRef columnRef = ColumnRef.create(indexColumn.getName(), table);
44          result.put(columnRef, (int) indexColumn.getLength());
45        }
46      }
47      this.lengths = result;
48    }
49  
50    @Override
51    protected RangeSet<TypedKey> visit(ComparisonBinaryExpression node, Void context) {
52      NormalizedPredicate predicate = node.normalize();
53      if (predicate == null) {
54        throwOnError(node);
55      }
56      // In order to match a prefix index, we have to cut the literal by prefix length.
57      // e.g., for table t:
58      // CREATE TABLE `t` {
59      //     `b` VARCHAR(10) DEFAULT NULL,
60      //     KEY `prefix_index` (`b`(2))
61      // }
62      //
63      // b(2) > "bbc" -> ["bb", +∞)
64      // b(2) >= "bbc" -> ["bb", +∞)
65      // b(2) < "bbc" -> (-∞, "bb"]
66      // b(2) <= "bbc" -> (-∞, "bb"]
67      // b(2) = "bbc" -> ["bb", "bb"]
68      // b(2) > "b" -> ["b", +∞)
69      // b(2) >= "b" -> ["b", +∞)
70      // b(2) < "b" -> (-∞, "b"]
71      // b(2) <= "b" -> (-∞, "b"]
72      //
73      // For varchar, `b`(2) will take first two characters(bytes) as prefix index.
74      // TODO: Note that TiDB only supports UTF-8, we need to check if prefix index behave differently
75      // under other encoding methods
76      int prefixLen = lengths.getOrDefault(predicate.getColumnRef(), DataType.UNSPECIFIED_LEN);
77      TypedKey literal = predicate.getTypedLiteral(prefixLen);
78      boolean loose = !DataType.isLengthUnSpecified(prefixLen);
79      // With prefix length specified, the filter is loosen and so should the ranges
80      return visitComparisonBinaryExpr(node, context, literal, loose);
81    }
82  
83    @Override
84    protected RangeSet<TypedKey> visit(StringRegExpression node, Void context) {
85      ColumnRef columnRef = node.getColumnRef();
86      // In order to match a prefix index, we have to cut the literal by prefix length.
87      // e.g., for table t:
88      // CREATE TABLE `t` {
89      //     `c1` VARCHAR(10) DEFAULT NULL,
90      //     KEY `prefix_index` (`c`(2))
91      // }
92      // when the predicate is `c1` LIKE 'abc%', the index range should be ['ab', 'ab'].
93      // when the predicate is `c1` LIKE 'a%', the index range should be ['a', 'b').
94      // for varchar, `c1`(2) will take first two characters(bytes) as prefix index.
95      // TODO: Note that TiDB only supports UTF-8, we need to check if prefix index behave differently
96      // under other encoding methods
97      int prefixLen = lengths.getOrDefault(columnRef, DataType.UNSPECIFIED_LEN);
98      TypedKey literal = node.getTypedLiteral(prefixLen);
99      RangeSet<TypedKey> ranges = TreeRangeSet.create();
100 
101     switch (node.getRegType()) {
102       case STARTS_WITH:
103         ranges.add(Range.atLeast(literal).intersection(Range.lessThan(literal.next())));
104         break;
105       default:
106         throwOnError(node);
107     }
108     return ranges;
109   }
110 }
