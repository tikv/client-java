18  package org.tikv.common.expression.visitor;
19  
20  import java.util.Objects;
21  import java.util.Set;
22  import org.tikv.common.expression.*;
23  import org.tikv.common.expression.ComparisonBinaryExpression.NormalizedPredicate;
24  import org.tikv.common.expression.FuncCallExpr.Type;
25  import org.tikv.common.predicates.PredicateUtils;
26  
27  /**
28   * PartAndFilterExprRewriter takes partition expression as an input. Rewriting rule is based on the
29   * type of partition expression. 1. If partition expression is a columnRef, no rewriting will be
30   * performed. 2. If partition expression is year and the expression to be rewritten in the form of y
31   * &lt; '1995-10-10' then its right hand child will be replaced with "1995". 3. If partition
32   * expression is year and the expression to be rewritten in the form of year(y) &lt; '1995' then its
33   * left hand child will be replaced with y.
34   */
35  public class PartAndFilterExprRewriter extends DefaultVisitor<Expression, Void> {
36    private final Expression partExpr;
37    private final Set<ColumnRef> columnRefs;
38  
39    private boolean unsupportedPartFnFound;
40  
41    public PartAndFilterExprRewriter(Expression partExpr) {
42      Objects.requireNonNull(partExpr, "partition expression cannot be null");
43      this.partExpr = partExpr;
44      this.columnRefs = PredicateUtils.extractColumnRefFromExpression(partExpr);
45    }
46  
47    private boolean isYear() {
48      return partExpr instanceof FuncCallExpr && ((FuncCallExpr) partExpr).getFuncTp() == Type.YEAR;
49    }
50  
51    private boolean isColumnRef() {
52      return partExpr instanceof ColumnRef;
53    }
54  
55    @Override
56    protected Expression process(Expression node, Void context) {
57      for (Expression expr : node.getChildren()) {
58        expr.accept(this, context);
59      }
60      return node;
61    }
62  
63    public Expression visit(LogicalBinaryExpression node, Void context) {
64      Expression left = node.getLeft().accept(this, null);
65      Expression right = node.getRight().accept(this, null);
66      return new LogicalBinaryExpression(node.getCompType(), left, right);
67    }
68  
69    @Override
70    public Expression visit(FuncCallExpr node, Void context) {
71      if (node.getFuncTp() == Type.YEAR) {
72        return node.getExpression();
73      }
74      // other's is not supported right now.
75      // TODO: when adding new type in FuncCallExpr, please also modify here
76      // accordingly.
77      return node;
78    }
79  
80    @Override
81    public Expression visit(Constant node, Void context) {
82      return node;
83    }
84  
85    @Override
86    public Expression visit(ComparisonBinaryExpression node, Void context) {
87      NormalizedPredicate predicate = node.normalize();
88      // predicate maybe null if node's left or right does not have a column ref or a constant.
89      if (predicate != null) {
90        if (!columnRefs.contains(predicate.getColumnRef())) {
91          return node;
92        }
93        // we only support year for now.
94        if (isYear()) {
95          FuncCallExpr year = new FuncCallExpr(predicate.getValue(), Type.YEAR);
96          Constant newLiteral = year.eval(predicate.getValue());
97          return new ComparisonBinaryExpression(node.getComparisonType(), node.getLeft(), newLiteral);
98        } else if (isColumnRef()) {
99          return node;
100       }
101       unsupportedPartFnFound = true;
102       return null;
103     }
104 
105     // when we find a node in form like [year(y) < 1995], we need rewrite the left child.
106     Expression left = node.getLeft().accept(this, null);
107     Expression right = node.getRight().accept(this, null);
108     return new ComparisonBinaryExpression(node.getComparisonType(), left, right);
109   }
110 
111   public Expression rewrite(Expression target) {
112     return target.accept(this, null);
113   }
114 
115   public boolean isUnsupportedPartFnFound() {
116     return unsupportedPartFnFound;
117   }
118 }
