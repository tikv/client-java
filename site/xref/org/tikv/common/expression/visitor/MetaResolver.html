18  package org.tikv.common.expression.visitor;
19  
20  import java.util.List;
21  import java.util.Objects;
22  import org.tikv.common.expression.ColumnRef;
23  import org.tikv.common.expression.ComparisonBinaryExpression;
24  import org.tikv.common.expression.ComparisonBinaryExpression.NormalizedPredicate;
25  import org.tikv.common.expression.Expression;
26  import org.tikv.common.meta.TiTableInfo;
27  
28  public class MetaResolver extends DefaultVisitor<Void, Expression> {
29    private final TiTableInfo table;
30  
31    public MetaResolver(TiTableInfo table) {
32      this.table = table;
33    }
34  
35    public static void resolve(Expression expression, TiTableInfo table) {
36      MetaResolver resolver = new MetaResolver(table);
37      resolver.resolve(expression);
38    }
39  
40    public static void resolve(List<? extends Expression> expressions, TiTableInfo table) {
41      MetaResolver resolver = new MetaResolver(table);
42      resolver.resolve(expressions);
43    }
44  
45    public void resolve(List<? extends Expression> expressions) {
46      expressions.forEach(expression -> expression.accept(this, null));
47    }
48  
49    public void resolve(Expression expression) {
50      Objects.requireNonNull(expression, "expression is null");
51      expression.accept(this, null);
52    }
53  
54    @Override
55    protected Void visit(ComparisonBinaryExpression node, Expression parent) {
56      NormalizedPredicate predicate = node.normalize();
57      // TODO(Zhexuan Yang): fix this if we have complex ComparisonBinaryExpression
58      // We may need add a expressionRewriter to address this.
59      if (predicate != null) {
60        visit(predicate.getColumnRef(), node);
61        // do not set the constant data type to the column ref data type if they are the
62        // same catalog because it may narrow the constant type, and cause wrong result.
63        // for example when the filter is `bit_col op long_constant`, set long_constant
64        // to bit type will truncated the long_constant, and may cause wrong result
65        if (predicate.getValue().getDataType() == null
66            || !predicate
67                .getValue()
68                .getDataType()
69                .isSameCatalog(predicate.getColumnRef().getDataType()))
70          predicate.getValue().setDataType(predicate.getColumnRef().getDataType());
71      }
72      return null;
73    }
74  
75    @Override
76    protected Void visit(ColumnRef node, Expression parent) {
77      node.resolve(table);
78      return null;
79    }
80  }
