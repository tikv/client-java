18  package org.tikv.common.expression;
19  
20  import static org.tikv.common.types.MySQLType.*;
21  
22  import java.util.HashMap;
23  import java.util.Map;
24  import org.tikv.common.types.MySQLType;
25  
26  public class TypeBlocklist extends Blocklist {
27    private static final Map<MySQLType, String> typeToMySQLMap = initialTypeMap();
28  
29    public TypeBlocklist(String typesString) {
30      super(typesString);
31    }
32  
33    private static HashMap<MySQLType, String> initialTypeMap() {
34      HashMap<MySQLType, String> map = new HashMap<>();
35      map.put(TypeDecimal, "decimal");
36      map.put(TypeTiny, "tinyint");
37      map.put(TypeShort, "smallint");
38      map.put(TypeLong, "int");
39      map.put(TypeFloat, "float");
40      map.put(TypeDouble, "double");
41      map.put(TypeNull, "null");
42      map.put(TypeTimestamp, "timestamp");
43      map.put(TypeLonglong, "bigint");
44      map.put(TypeInt24, "mediumint");
45      map.put(TypeDate, "date");
46      map.put(TypeDuration, "time");
47      map.put(TypeDatetime, "datetime");
48      map.put(TypeYear, "year");
49      map.put(TypeNewDate, "date");
50      map.put(TypeVarchar, "varchar");
51      map.put(TypeJSON, "json");
52      map.put(TypeNewDecimal, "decimal");
53      map.put(TypeEnum, "enum");
54      map.put(TypeSet, "set");
55      map.put(TypeTinyBlob, "tinytext");
56      map.put(TypeMediumBlob, "mediumtext");
57      map.put(TypeLongBlob, "longtext");
58      map.put(TypeBlob, "text");
59      map.put(TypeVarString, "varString");
60      map.put(TypeString, "string");
61      return map;
62    }
63  
64    public boolean isUnsupportedType(MySQLType sqlType) {
65      return isUnsupported(typeToMySQLMap.getOrDefault(sqlType, ""));
66    }
67  }
