18  package org.tikv.common.columnar;
19  
20  import java.math.BigDecimal;
21  import org.tikv.common.row.Row;
22  import org.tikv.common.types.DataType;
23  
24  /**
25   * An implementation of {@link TiColumnVector}. It is a faked column vector; the underlying data is
26   * in row format.
27   */
28  public class TiRowColumnVector extends TiColumnVector {
29    /** Represents the column index of original row */
30    private final int colIdx;
31    /** row-wise format data and data is already decoded */
32    private Row[] rows;
33    /** Sets up the data type of this column vector. */
34    public TiRowColumnVector(DataType type, int colIdx, Row[] rows, int numOfRows) {
35      super(type, numOfRows);
36      this.rows = rows;
37      this.colIdx = colIdx;
38    }
39  
40    /**
41     * Cleans up memory for this column vector. The column vector is not usable after this.
42     *
43     * <p>This overwrites `AutoCloseable.close` to remove the `throws` clause, as column vector is
44     * in-memory and we don't expect any exception to happen during closing.
45     */
46    @Override
47    public void close() {
48      this.rows = null;
49    }
50  
51    /** Returns true if this column vector contains any null values. */
52    @Override
53    public boolean hasNull() {
54      throw new UnsupportedOperationException(
55          "row-wise column vector does not support this operation");
56    }
57  
58    /** Returns the number of nulls in this column vector. */
59    @Override
60    public int numNulls() {
61      throw new UnsupportedOperationException(
62          "row-wise column vector does not support this operation");
63    }
64  
65    /** Returns whether the value at rowId is NULL. */
66    @Override
67    public boolean isNullAt(int rowId) {
68      return rows[rowId].get(colIdx, null) == null;
69    }
70  
71    /**
72     * Returns the boolean type value for rowId. The return value is undefined and can be anything, if
73     * the slot for rowId is null.
74     */
75    @Override
76    public boolean getBoolean(int rowId) {
77      return rows[rowId].getLong(colIdx) == 1;
78    }
79  
80    /**
81     * Returns the byte type value for rowId. The return value is undefined and can be anything, if
82     * the slot for rowId is null.
83     */
84    @Override
85    public byte getByte(int rowId) {
86      return (byte) rows[rowId].getLong(colIdx);
87    }
88  
89    /**
90     * Returns the short type value for rowId. The return value is undefined and can be anything, if
91     * the slot for rowId is null.
92     */
93    @Override
94    public short getShort(int rowId) {
95      return (short) rows[rowId].getLong(colIdx);
96    }
97  
98    /**
99     * Returns the int type value for rowId. The return value is undefined and can be anything, if the
100    * slot for rowId is null.
101    */
102   @Override
103   public int getInt(int rowId) {
104     return (int) rows[rowId].getLong(colIdx);
105   }
106 
107   /**
108    * Returns the long type value for rowId. The return value is undefined and can be anything, if
109    * the slot for rowId is null.
110    */
111   @Override
112   public long getLong(int rowId) {
113     return rows[rowId].getLong(colIdx);
114   }
115 
116   /**
117    * Returns the float type value for rowId. The return value is undefined and can be anything, if
118    * the slot for rowId is null.
119    */
120   @Override
121   public float getFloat(int rowId) {
122     return ((Number) rows[rowId].getDouble(colIdx)).floatValue();
123   }
124 
125   /**
126    * Returns the double type value for rowId. The return value is undefined and can be anything, if
127    * the slot for rowId is null.
128    */
129   @Override
130   public double getDouble(int rowId) {
131     return rows[rowId].getDouble(colIdx);
132   }
133 
134   /**
135    * Returns the decimal type value for rowId. If the slot for rowId is null, it should return null.
136    */
137   @Override
138   public BigDecimal getDecimal(int rowId, int precision, int scale) {
139     Object val = rows[rowId].get(colIdx, null);
140     if (val instanceof BigDecimal) {
141       return (BigDecimal) val;
142     }
143 
144     if (val instanceof Long) {
145       return BigDecimal.valueOf((long) val);
146     }
147 
148     throw new UnsupportedOperationException(
149         String.format(
150             "failed to getDecimal and the value is %s:%s", val.getClass().getCanonicalName(), val));
151   }
152 
153   /**
154    * Returns the string type value for rowId. If the slot for rowId is null, it should return null.
155    * Note that the returned UTF8String may point to the data of this column vector, please copy it
156    * if you want to keep it after this column vector is freed.
157    */
158   @Override
159   public String getUTF8String(int rowId) {
160     return rows[rowId].getString(colIdx);
161   }
162 
163   /**
164    * Returns the binary type value for rowId. If the slot for rowId is null, it should return null.
165    */
166   @Override
167   public byte[] getBinary(int rowId) {
168     return rows[rowId].getBytes(colIdx);
169   }
170 
171   /** @return child [[TiColumnVector]] at the given ordinal. */
172   @Override
173   protected TiColumnVector getChild(int ordinal) {
174     throw new UnsupportedOperationException(
175         "row-wise column vector does not support this operation");
176   }
177 }
