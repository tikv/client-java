91      this(
92          session,
93          startTime,
94          lockTTL,
95          TXN_COMMIT_BATCH_SIZE,
96          TXN_COMMIT_BATCH_SIZE,

100         3,
101         createExecutorService(WRITE_BUFFER_SIZE));

107       long lockTTL,
108       long txnPrewriteBatchSize,
109       long txnCommitBatchSize,
110       int writeBufferSize,
111       int writeThreadPerTask,
112       boolean retryCommitSecondaryKeys,

121     this.txnCommitBatchSize = txnCommitBatchSize;
122     this.writeBufferSize = writeBufferSize;
123     this.writeThreadPerTask = writeThreadPerTask;
124     this.prewriteMaxRetryTimes = prewriteMaxRetryTimes;
125     this.executorService = executorService;
126   }

128   private static ExecutorService createExecutorService(int size) {
129     return Executors.newFixedThreadPool(
130         size, new ThreadFactoryBuilder().setNameFormat("2pc-pool-%d").setDaemon(true).build());
131   }

140    * 2pc - prewrite primary key
141    *
142    * @param backOffer
143    * @param primaryKey
144    * @param value
145    */
146   public void prewritePrimaryKey(BackOffer backOffer, byte[] primaryKey, byte[] value)
147       throws TiBatchWriteException {
148     this.doPrewritePrimaryKeyWithRetry(
149         backOffer, ByteString.copyFrom(primaryKey), ByteString.copyFrom(value));
150   }
151 
152   private void doPrewritePrimaryKeyWithRetry(BackOffer backOffer, ByteString key, ByteString value)
153       throws TiBatchWriteException {
154     Pair<TiRegion, TiStore> pair = this.regionManager.getRegionStorePairByKey(key, backOffer);
155     TiRegion tiRegion = pair.first;
156     TiStore store = pair.second;
157 
158     Kvrpcpb.Mutation mutation;
159     if (!value.isEmpty()) {
160       mutation = Kvrpcpb.Mutation.newBuilder().setKey(key).setValue(value).setOp(Op.Put).build();
161     } else {
162       mutation = Kvrpcpb.Mutation.newBuilder().setKey(key).setOp(Op.Del).build();
163     }
164     List<Kvrpcpb.Mutation> mutationList = Collections.singletonList(mutation);
165 
166     // send rpc request to tikv server
167     long lockTTL = getTxnLockTTL(this.startTs);
168     ClientRPCResult prewriteResult =
169         this.kvClient.prewrite(
170             backOffer, mutationList, key, lockTTL, this.startTs, tiRegion, store);
171     if (!prewriteResult.isSuccess() && !prewriteResult.isRetry()) {
172       throw new TiBatchWriteException("prewrite primary key error", prewriteResult.getException());
173     }
174     if (prewriteResult.isRetry()) {
175       try {
176         backOffer.doBackOff(
177             BackOffFunction.BackOffFuncType.BoRegionMiss,
178             new GrpcException(
179                 String.format("Txn prewrite primary key failed, regionId=%s", tiRegion.getId()),
180                 prewriteResult.getException()));
181         // re-split keys and commit again.
182         this.doPrewritePrimaryKeyWithRetry(backOffer, key, value);
183       } catch (GrpcException e) {
184         String errorMsg =
185             String.format(
186                 "Txn prewrite primary key error, re-split commit failed, regionId=%s, detail=%s",
187                 tiRegion.getId(), e.getMessage());
188         throw new TiBatchWriteException(errorMsg, e);
189       }
190     }
191 
192     LOG.info("prewrite primary key {} successfully", KeyUtils.formatBytes(key));
193   }

195   /**
196    * 2pc - commit primary key
197    *
198    * @param backOffer
199    * @param key
200    */
201   public void commitPrimaryKey(BackOffer backOffer, byte[] key, long commitTs)
202       throws TiBatchWriteException {
203     doCommitPrimaryKeyWithRetry(backOffer, ByteString.copyFrom(key), commitTs);
204   }
205 
206   private void doCommitPrimaryKeyWithRetry(BackOffer backOffer, ByteString key, long commitTs)
207       throws TiBatchWriteException {
208     Pair<TiRegion, TiStore> pair = this.regionManager.getRegionStorePairByKey(key, backOffer);
209     TiRegion tiRegion = pair.first;
210     TiStore store = pair.second;
211     ByteString[] keys = new ByteString[] {key};
212 
213     // send rpc request to tikv server
214     ClientRPCResult commitResult =
215         this.kvClient.commit(backOffer, keys, this.startTs, commitTs, tiRegion, store);
216 
217     if (!commitResult.isSuccess()) {
218       if (!commitResult.isRetry()) {
219         throw new TiBatchWriteException("commit primary key error", commitResult.getException());
220       } else {
221         backOffer.doBackOff(
222             BackOffFunction.BackOffFuncType.BoRegionMiss,
223             new GrpcException(
224                 String.format("Txn commit primary key failed, regionId=%s", tiRegion.getId()),
225                 commitResult.getException()));
226         // re-split keys and commit again.
227         this.doCommitPrimaryKeyWithRetry(backOffer, key, commitTs);
228       }
229     }
230 
231     LOG.info("commit primary key {} successfully", KeyUtils.formatBytes(key));
232   }

234   /**
235    * 2pc - prewrite secondary keys
236    *
237    * @param primaryKey
238    * @param pairs
239    */
240   public void prewriteSecondaryKeys(
241       byte[] primaryKey, Iterator<BytePairWrapper> pairs, int maxBackOfferMS)
242       throws TiBatchWriteException {
243     Iterator<Pair<ByteString, ByteString>> byteStringKeys =
244         new Iterator<Pair<ByteString, ByteString>>() {
245 
246           @Override
247           public boolean hasNext() {
248             return pairs.hasNext();
249           }
250 
251           @Override
252           public Pair<ByteString, ByteString> next() {
253             BytePairWrapper pair = pairs.next();
254             return new Pair<>(
255                 ByteString.copyFrom(pair.getKey()), ByteString.copyFrom(pair.getValue()));
256           }
257         };
258 
259     doPrewriteSecondaryKeys(ByteString.copyFrom(primaryKey), byteStringKeys, maxBackOfferMS);
260   }
261 
262   private void doPrewriteSecondaryKeys(
263       ByteString primaryKey, Iterator<Pair<ByteString, ByteString>> pairs, int maxBackOfferMS)
264       throws TiBatchWriteException {
265     try {
266       int taskBufferSize = writeThreadPerTask * 2;
267       int totalSize = 0, cnt = 0;
268       Pair<ByteString, ByteString> pair;
269       ExecutorCompletionService<Void> completionService =
270           new ExecutorCompletionService<>(executorService);
271       while (pairs.hasNext()) {
272         int size = 0;
273         ByteString[] keyBytes = new ByteString[writeBufferSize];
274         ByteString[] valueBytes = new ByteString[writeBufferSize];
275         while (size < writeBufferSize && pairs.hasNext()) {
276           pair = pairs.next();
277           keyBytes[size] = pair.first;
278           valueBytes[size] = pair.second;
279           size++;
280         }
281         int curSize = size;
282         cnt++;
283         if (cnt > taskBufferSize) {
284           // consume one task if reaches task limit
285           completionService.take().get();
286         }
287         BackOffer backOffer =
288             ConcreteBackOffer.newCustomBackOff(
289                 maxBackOfferMS, regionManager.getPDClient().getClusterId());
290         completionService.submit(
291             () -> {
292               doPrewriteSecondaryKeysInBatchesWithRetry(
293                   backOffer, primaryKey, keyBytes, valueBytes, curSize, 0);
294               return null;
295             });
296 
297         totalSize = totalSize + size;
298       }
299 
300       for (int i = 0; i < Math.min(taskBufferSize, cnt); i++) {
301         completionService.take().get();
302       }
303 
304     } catch (InterruptedException e) {
305       Thread.currentThread().interrupt();
306       throw new TiBatchWriteException("Current thread interrupted.", e);
307     } catch (ExecutionException e) {
308       throw new TiBatchWriteException("Execution exception met.", e);
309     }
310   }

312   private void doPrewriteSecondaryKeysInBatchesWithRetry(
313       BackOffer backOffer,
314       ByteString primaryKey,
315       ByteString[] keys,
316       ByteString[] values,
317       int size,
318       int level)
319       throws TiBatchWriteException {
320     if (keys == null || keys.length == 0 || values == null || values.length == 0 || size <= 0) {
321       // return success
322       return;
323     }
324 
325     Map<ByteString, Kvrpcpb.Mutation> mutations = new LinkedHashMap<>();
326     for (int i = 0; i < size; i++) {
327       ByteString key = keys[i];
328       ByteString value = values[i];
329 
330       Kvrpcpb.Mutation mutation;
331       if (!value.isEmpty()) {
332         mutation = Kvrpcpb.Mutation.newBuilder().setKey(key).setValue(value).setOp(Op.Put).build();
333       } else {
334         // value can be null (table with one primary key integer column, data is encoded in key)
335         mutation = Kvrpcpb.Mutation.newBuilder().setKey(key).setOp(Op.Del).build();
336       }
337       mutations.put(key, mutation);
338     }
339 
340     // groups keys by region
341     GroupKeyResult groupResult = this.groupKeysByRegion(keys, size, backOffer);
342     List<BatchKeys> batchKeyList = new LinkedList<>();
343     Map<Pair<TiRegion, TiStore>, List<ByteString>> groupKeyMap = groupResult.getGroupsResult();
344 
345     for (Map.Entry<Pair<TiRegion, TiStore>, List<ByteString>> entry : groupKeyMap.entrySet()) {
346       TiRegion tiRegion = entry.getKey().first;
347       TiStore store = entry.getKey().second;
348       this.appendBatchBySize(batchKeyList, tiRegion, store, entry.getValue(), true, mutations);
349     }
350 
351     // For prewrite, stop sending other requests after receiving first error.
352     for (BatchKeys batchKeys : batchKeyList) {
353       TiRegion oldRegion = batchKeys.getRegion();
354       TiRegion currentRegion =
355           this.regionManager.getRegionByKey(oldRegion.getStartKey(), backOffer);
356       if (oldRegion.equals(currentRegion)) {
357         doPrewriteSecondaryKeySingleBatchWithRetry(backOffer, primaryKey, batchKeys, mutations);
358       } else {
359         if (level > prewriteMaxRetryTimes) {
360           throw new TiBatchWriteException(
361               String.format(
362                   "> max retry number %s, oldRegion=%s, currentRegion=%s",
363                   prewriteMaxRetryTimes, oldRegion, currentRegion));
364         }
365         LOG.info(
366             String.format(
367                 "oldRegion=%s != currentRegion=%s, will re-fetch region info and retry",
368                 oldRegion, currentRegion));
369         retryPrewriteBatch(backOffer, primaryKey, batchKeys, mutations, level <= 0 ? 1 : level + 1);
370       }
371     }
372   }
373 
374   private void retryPrewriteBatch(
375       BackOffer backOffer,
376       ByteString primaryKey,
377       BatchKeys batchKeys,
378       Map<ByteString, Kvrpcpb.Mutation> mutations,
379       int level) {
380 
381     int size = batchKeys.getKeys().size();
382     ByteString[] keyBytes = new ByteString[size];
383     ByteString[] valueBytes = new ByteString[size];
384     int i = 0;
385     for (ByteString k : batchKeys.getKeys()) {
386       keyBytes[i] = k;
387       valueBytes[i] = mutations.get(k).getValue();
388       i++;
389     }
390     doPrewriteSecondaryKeysInBatchesWithRetry(
391         backOffer, primaryKey, keyBytes, valueBytes, size, level);
392   }

394   private void doPrewriteSecondaryKeySingleBatchWithRetry(
395       BackOffer backOffer,
396       ByteString primaryKey,
397       BatchKeys batchKeys,
398       Map<ByteString, Kvrpcpb.Mutation> mutations)
399       throws TiBatchWriteException {
400     LOG.debug(
401         "start prewrite secondary key, row={}, size={}KB, regionId={}",
402         batchKeys.getKeys().size(),
403         batchKeys.getSizeInKB(),
404         batchKeys.getRegion().getId());
405 
406     List<ByteString> keyList = batchKeys.getKeys();
407     int batchSize = keyList.size();
408     List<Kvrpcpb.Mutation> mutationList = new ArrayList<>(batchSize);
409     for (ByteString key : keyList) {
410       mutationList.add(mutations.get(key));
411     }
412     // send rpc request to tikv server
413     int txnSize = batchKeys.getKeys().size();
414     long lockTTL = getTxnLockTTL(this.startTs, txnSize);
415     ClientRPCResult prewriteResult =
416         this.kvClient.prewrite(
417             backOffer,
418             mutationList,
419             primaryKey,
420             lockTTL,
421             this.startTs,
422             batchKeys.getRegion(),
423             batchKeys.getStore());
424     if (!prewriteResult.isSuccess() && !prewriteResult.isRetry()) {
425       throw new TiBatchWriteException(
426           "prewrite secondary key error", prewriteResult.getException());
427     }
428     if (prewriteResult.isRetry()) {
429       LOG.info("prewrite secondary key fail, will backoff and retry");
430       try {
431         backOffer.doBackOff(
432             BackOffFunction.BackOffFuncType.BoRegionMiss,
433             new GrpcException(
434                 String.format(
435                     "Txn prewrite secondary key SingleBatch failed, regionId=%s",
436                     batchKeys.getRegion().getId()),
437                 prewriteResult.getException()));
438         // re-split keys and commit again.
439         retryPrewriteBatch(backOffer, primaryKey, batchKeys, mutations, 0);
440       } catch (GrpcException e) {
441         String errorMsg =
442             String.format(
443                 "Txn prewrite secondary key SingleBatch error, re-split commit failed, regionId=%s, detail=%s",
444                 batchKeys.getRegion().getId(), e.getMessage());
445         throw new TiBatchWriteException(errorMsg, e);
446       }
447     }
448     LOG.debug(
449         "prewrite secondary key successfully, row={}, size={}KB, regionId={}",
450         batchKeys.getKeys().size(),
451         batchKeys.getSizeInKB(),
452         batchKeys.getRegion().getId());
453   }

455   private void appendBatchBySize(
456       List<BatchKeys> batchKeyList,
457       TiRegion tiRegion,
458       TiStore store,
459       List<ByteString> keys,
460       boolean sizeIncludeValue,
461       Map<ByteString, Kvrpcpb.Mutation> mutations) {
462     long commitBatchSize = sizeIncludeValue ? txnPrewriteBatchSize : txnCommitBatchSize;
463 
464     int start;
465     int end;
466     if (keys == null) {
467       return;
468     }
469     int len = keys.size();
470     for (start = 0; start < len; start = end) {
471       int sizeInBytes = 0;
472       for (end = start; end < len && sizeInBytes < commitBatchSize; end++) {
473         if (sizeIncludeValue) {
474           sizeInBytes += this.keyValueSize(keys.get(end), mutations);
475         } else {
476           sizeInBytes += this.keySize(keys.get(end));
477         }
478       }
479       BatchKeys batchKeys = new BatchKeys(tiRegion, store, keys.subList(start, end), sizeInBytes);
480       batchKeyList.add(batchKeys);
481     }
482   }
483 
484   private long keyValueSize(ByteString key, Map<ByteString, Kvrpcpb.Mutation> mutations) {
485     long size = key.size();
486     Kvrpcpb.Mutation mutation = mutations.get(key);
487     if (mutation != null) {
488       size += mutation.getValue().toByteArray().length;
489     }
490 
491     return size;
492   }

504   public void commitSecondaryKeys(

514           }
515

525   private void doCommitSecondaryKeys(
526       Iterator<ByteString> keys, long commitTs, int commitBackOfferMS)
527       throws TiBatchWriteException {
528     try {
529       int taskBufferSize = writeThreadPerTask * 2;
530       int totalSize = 0, cnt = 0;
531       ExecutorCompletionService<Void> completionService =
532           new ExecutorCompletionService<>(executorService);
533       while (keys.hasNext()) {
534         int size = 0;
535         ByteString[] keyBytes = new ByteString[writeBufferSize];
536         while (size < writeBufferSize && keys.hasNext()) {
537           keyBytes[size] = keys.next();
538           size++;
539         }
540         int curSize = size;
541         cnt++;
542         if (cnt > taskBufferSize) {
543           // consume one task if reaches task limit
544           completionService.take().get();
545         }
546         BackOffer backOffer =
547             ConcreteBackOffer.newCustomBackOff(
548                 commitBackOfferMS, regionManager.getPDClient().getClusterId());
549         completionService.submit(
550             () -> {
551               doCommitSecondaryKeysWithRetry(backOffer, keyBytes, curSize, commitTs);
552               return null;
553             });
554 
555         totalSize = totalSize + size;
556       }
557 
558       for (int i = 0; i < Math.min(taskBufferSize, cnt); i++) {
559         completionService.take().get();
560       }
561 
562     } catch (InterruptedException e) {
563       Thread.currentThread().interrupt();
564       throw new TiBatchWriteException("Current thread interrupted.", e);
565     } catch (ExecutionException e) {
566       throw new TiBatchWriteException("Execution exception met.", e);
567     }
568   }

570   private void doCommitSecondaryKeysWithRetry(
571       BackOffer backOffer, ByteString[] keys, int size, long commitTs)
572       throws TiBatchWriteException {
573     if (keys == null || keys.length == 0 || size <= 0) {
574       return;
575     }
576 
577     // groups keys by region
578     GroupKeyResult groupResult = this.groupKeysByRegion(keys, size, backOffer);
579     List<BatchKeys> batchKeyList = new ArrayList<>();
580     Map<Pair<TiRegion, TiStore>, List<ByteString>> groupKeyMap = groupResult.getGroupsResult();
581 
582     for (Map.Entry<Pair<TiRegion, TiStore>, List<ByteString>> entry : groupKeyMap.entrySet()) {
583       TiRegion tiRegion = entry.getKey().first;
584       TiStore store = entry.getKey().second;
585       this.appendBatchBySize(batchKeyList, tiRegion, store, entry.getValue(), false, null);
586     }
587 
588     for (BatchKeys batchKeys : batchKeyList) {
589       doCommitSecondaryKeySingleBatchWithRetry(backOffer, batchKeys, commitTs);
590     }
591   }

593   private void doCommitSecondaryKeySingleBatchWithRetry(
594       BackOffer backOffer, BatchKeys batchKeys, long commitTs) throws TiBatchWriteException {
595     LOG.info(
596         "start commit secondary key, row={}, size={}KB, regionId={}",
597         batchKeys.getKeys().size(),
598         batchKeys.getSizeInKB(),
599         batchKeys.getRegion().getId());
600     List<ByteString> keysCommit = batchKeys.getKeys();
601     ByteString[] keys = new ByteString[keysCommit.size()];
602     keysCommit.toArray(keys);
603     // send rpc request to tikv server
604     ClientRPCResult commitResult =
605         this.kvClient.commit(
606             backOffer, keys, this.startTs, commitTs, batchKeys.getRegion(), batchKeys.getStore());
607     if (retryCommitSecondaryKeys && commitResult.isRetry()) {
608       doCommitSecondaryKeysWithRetry(backOffer, keys, keysCommit.size(), commitTs);
609     } else if (!commitResult.isSuccess()) {
610       String error =
611           String.format("Txn commit secondary key error, regionId=%s", batchKeys.getRegion());
612       LOG.warn(error);
613       throw new TiBatchWriteException("commit secondary key error", commitResult.getException());
614     }
615     LOG.info(
616         "commit {} rows successfully, size={}KB, regionId={}",
617         batchKeys.getKeys().size(),
618         batchKeys.getSizeInKB(),
619         batchKeys.getRegion().getId());
620   }

622   private GroupKeyResult groupKeysByRegion(ByteString[] keys, int size, BackOffer backOffer)
623       throws TiBatchWriteException {
624     Map<Pair<TiRegion, TiStore>, List<ByteString>> groups = new HashMap<>();
625     int index = 0;
626     try {
627       for (; index < size; index++) {
628         ByteString key = keys[index];
629         Pair<TiRegion, TiStore> pair = this.regionManager.getRegionStorePairByKey(key, backOffer);
630         if (pair != null) {
631           groups.computeIfAbsent(pair, e -> new ArrayList<>()).add(key);
632         }
633       }
634     } catch (Exception e) {
635       throw new TiBatchWriteException("Txn groupKeysByRegion error", e);
636     }
637     GroupKeyResult result = new GroupKeyResult();
638     result.setGroupsResult(groups);
639     return result;
640   }
