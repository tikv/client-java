34  import org.tikv.common.region.TiRegion;
35  import org.tikv.common.region.TiStore;

42  /**
43   * see funcionts `keepAlive` `close` `sendTxnHeartBeat` in
44   * https://github.com/pingcap/tidb/blob/master/store/tikv/2pc.go
45   */
46  public class TTLManager {
47    /** 20 seconds */
48    public static final int MANAGED_LOCK_TTL = 20000;
49  
50    private static final Logger LOG = LoggerFactory.getLogger(TTLManager.class);
51    /** status */
52    private static final int STATE_UNINITIALIZED = 0;
53  
54    private static final int STATE_RUNNING = 1;
55    private static final int STATE_CLOSED = 2;
56    /** 10 seconds */
57    private static final int SCHEDULER_PERIOD = MANAGED_LOCK_TTL / 2;
58    /** 5 seconds */
59    private static final int SCHEDULER_INITIAL_DELAY = MANAGED_LOCK_TTL / 4;
60  
61    private final long startTS;
62    private final ByteString primaryLock;
63    private final TxnKVClient kvClient;
64    private final RegionManager regionManager;
65    private final ScheduledExecutorService scheduler;
66    private final AtomicInteger state;
67  
68    public TTLManager(TiSession session, long startTS, byte[] primaryKey) {
69      this.startTS = startTS;
70      this.primaryLock = ByteString.copyFrom(primaryKey);
71      this.state = new AtomicInteger(STATE_UNINITIALIZED);
72  
73      this.kvClient = session.createTxnClient();
74      this.regionManager = kvClient.getRegionManager();
75  
76      scheduler =
77          new ScheduledThreadPoolExecutor(
78              1,
79              new BasicThreadFactory.Builder()
80                  .namingPattern("ttl-manager-pool-%d")
81                  .daemon(false)
82                  .build());
83    }
84  
85    public void keepAlive() {
86      if (state.compareAndSet(STATE_UNINITIALIZED, STATE_RUNNING)) {
87        scheduler.scheduleAtFixedRate(
88            this::doKeepAlive, SCHEDULER_INITIAL_DELAY, SCHEDULER_PERIOD, TimeUnit.MILLISECONDS);
89      } else {
90        LOG.warn("keepAlive failed state={} key={}", state.get(), KeyUtils.formatBytes(primaryLock));
91      }
92    }
93  
94    private void doKeepAlive() {
95      BackOffer bo =
96          ConcreteBackOffer.newCustomBackOff(
97              MANAGED_LOCK_TTL, regionManager.getPDClient().getClusterId());
98      long uptime = kvClient.getTimestamp().getPhysical() - TiTimestamp.extractPhysical(startTS);
99      long ttl = uptime + MANAGED_LOCK_TTL;
100 
101     LOG.info("doKeepAlive key={} uptime={} ttl={}", KeyUtils.formatBytes(primaryLock), uptime, ttl);
102 
103     try {
104       sendTxnHeartBeat(bo, ttl);
105       LOG.info("doKeepAlive success");
106     } catch (Exception e) {
107       LOG.warn("doKeepAlive error", e);
108     }
109   }
110 
111   private void sendTxnHeartBeat(BackOffer bo, long ttl) {
112     Pair<TiRegion, TiStore> pair = regionManager.getRegionStorePairByKey(primaryLock);
113     TiRegion tiRegion = pair.first;
114     TiStore store = pair.second;
115 
116     ClientRPCResult result = kvClient.txnHeartBeat(bo, primaryLock, startTS, ttl, tiRegion, store);
117 
118     if (!result.isSuccess() && !result.isRetry()) {
119       throw new TiBatchWriteException("sendTxnHeartBeat error", result.getException());
120     }
121     if (result.isRetry()) {
122       try {
123         bo.doBackOff(
124             BackOffFunction.BackOffFuncType.BoRegionMiss,
125             new GrpcException(
126                 String.format("sendTxnHeartBeat failed, regionId=%s", tiRegion.getId()),
127                 result.getException()));
128         this.regionManager.invalidateStore(store.getStore().getId());
129         this.regionManager.invalidateRegion(tiRegion);
130         // re-split keys and commit again.
131         sendTxnHeartBeat(bo, ttl);
132       } catch (GrpcException e) {
133         String errorMsg =
134             String.format(
135                 "sendTxnHeartBeat error, regionId=%s, detail=%s", tiRegion.getId(), e.getMessage());
136         throw new TiBatchWriteException(errorMsg, e);
137       }
138     }
139 
140     LOG.debug(
141         "sendTxnHeartBeat success key={} ttl={} success", KeyUtils.formatBytes(primaryLock), ttl);
142   }
143 
144   public void close() throws InterruptedException {
145     if (state.compareAndSet(STATE_RUNNING, STATE_CLOSED)) {
146       scheduler.shutdown();
147     }
148   }
149 }
