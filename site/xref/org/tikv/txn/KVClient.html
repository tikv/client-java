55    public KVClient(TiConfiguration conf, RegionStoreClientBuilder clientBuilder, TiSession session) {
56      Objects.requireNonNull(conf, "conf is null");
57      Objects.requireNonNull(clientBuilder, "clientBuilder is null");
58      this.tiSession = session;
59      this.conf = conf;
60      this.clientBuilder = clientBuilder;
61      executorService =
62          Executors.newFixedThreadPool(
63              conf.getKvClientConcurrency(),
64              new ThreadFactoryBuilder().setNameFormat("kvclient-pool-%d").setDaemon(true).build());
65    }
66  
67    @Override
68    public void close() {
69      if (executorService != null) {
70        executorService.shutdownNow();
71      }
72    }

74    /**
75     * Get a key-value pair from TiKV if key exists
76     *
77     * @param key key
78     * @return a ByteString value if key exists, ByteString.EMPTY if key does not exist
79     */
80    public ByteString get(ByteString key, long version) throws GrpcException {
81      BackOffer backOffer =
82          ConcreteBackOffer.newGetBackOff(
83              clientBuilder.getRegionManager().getPDClient().getClusterId());
84      while (true) {
85        RegionStoreClient client = clientBuilder.build(key);
86        try {
87          return client.get(backOffer, key, version);
88        } catch (final TiKVException e) {
89          backOffer.doBackOff(BackOffFunction.BackOffFuncType.BoRegionMiss, e);
90        }
91      }
92    }

108   /**
109    * Scan key-value pairs from TiKV in range [startKey, endKey)
110    *
111    * @param startKey start key, inclusive
112    * @param endKey end key, exclusive
113    * @return list of key-value pairs in range
114    */
115   public List<Kvrpcpb.KvPair> scan(ByteString startKey, ByteString endKey, long version)
116       throws GrpcException {
117     Iterator<Kvrpcpb.KvPair> iterator =
118         scanIterator(conf, clientBuilder, startKey, endKey, version);
119     List<Kvrpcpb.KvPair> result = new ArrayList<>();
120     iterator.forEachRemaining(result::add);
121     return result;
122   }

143   public synchronized void ingest(List<Pair<ByteString, ByteString>> list) throws GrpcException {
144     if (list.isEmpty()) {
145       return;
146     }
147 
148     Key min = Key.MAX;
149     Key max = Key.MIN;
150     Map<ByteString, ByteString> map = new HashMap<>(list.size());
151 
152     for (Pair<ByteString, ByteString> pair : list) {
153       map.put(pair.first, pair.second);
154       Key key = Key.toRawKey(pair.first.toByteArray());
155       if (key.compareTo(min) < 0) {
156         min = key;
157       }
158       if (key.compareTo(max) > 0) {
159         max = key;
160       }
161     }
162 
163     SwitchTiKVModeClient switchTiKVModeClient = tiSession.getSwitchTiKVModeClient();
164 
165     try {
166       // switch to normal mode
167       switchTiKVModeClient.switchTiKVToNormalMode();
168 
169       // region split
170       List<byte[]> splitKeys = new ArrayList<>(2);
171       splitKeys.add(min.getBytes());
172       splitKeys.add(max.next().getBytes());
173 
174       tiSession.splitRegionAndScatter(splitKeys);
175       tiSession.getRegionManager().invalidateAll();
176 
177       // switch to import mode
178       switchTiKVModeClient.keepTiKVToImportMode();
179 
180       // group keys by region
181       List<ByteString> keyList = list.stream().map(pair -> pair.first).collect(Collectors.toList());
182       Map<TiRegion, List<ByteString>> groupKeys =
183           groupKeysByRegion(
184               clientBuilder.getRegionManager(),
185               keyList,
186               ConcreteBackOffer.newRawKVBackOff(tiSession.getPDClient().getClusterId()));
187 
188       // ingest for each region
189       for (Map.Entry<TiRegion, List<ByteString>> entry : groupKeys.entrySet()) {
190         TiRegion region = entry.getKey();
191         List<ByteString> keys = entry.getValue();
192         List<Pair<ByteString, ByteString>> kvs =
193             keys.stream().map(k -> Pair.create(k, map.get(k))).collect(Collectors.toList());
194         doIngest(region, kvs);
195       }
196     } finally {
197       // swith tikv to normal mode
198       switchTiKVModeClient.stopKeepTiKVToImportMode();
199       switchTiKVModeClient.switchTiKVToNormalMode();
200     }
201   }

203   private List<Kvrpcpb.KvPair> doSendBatchGet(
204       BackOffer backOffer, List<ByteString> keys, long version) {
205     ExecutorCompletionService<List<Kvrpcpb.KvPair>> completionService =
206         new ExecutorCompletionService<>(executorService);
207 
208     List<Batch> batches =
209         getBatches(backOffer, keys, BATCH_GET_SIZE, MAX_BATCH_LIMIT, this.clientBuilder);
210 
211     for (Batch batch : batches) {
212       completionService.submit(
213           () -> doSendBatchGetInBatchesWithRetry(batch.getBackOffer(), batch, version));
214     }
215 
216     return getKvPairs(completionService, batches, BackOffer.BATCH_GET_MAX_BACKOFF);
217   }
218 
219   private List<Kvrpcpb.KvPair> doSendBatchGetInBatchesWithRetry(
220       BackOffer backOffer, Batch batch, long version) {
221     TiRegion oldRegion = batch.getRegion();
222     TiRegion currentRegion =
223         clientBuilder.getRegionManager().getRegionByKey(oldRegion.getStartKey());
224 
225     if (oldRegion.equals(currentRegion)) {
226       RegionStoreClient client = clientBuilder.build(batch.getRegion());
227       // set resolvedLocks for the new client
228       if (!resolvedLocks.isEmpty()) {
229         client.addResolvedLocks(version, resolvedLocks);
230       }
231       try {
232         return client.batchGet(backOffer, batch.getKeys(), version);
233       } catch (final TiKVException e) {
234         backOffer.doBackOff(BackOffFunction.BackOffFuncType.BoRegionMiss, e);
235         clientBuilder.getRegionManager().invalidateRegion(batch.getRegion());
236         logger.warn("ReSplitting ranges for BatchGetRequest", e);
237 
238         // get resolved locks and retry
239         resolvedLocks = client.getResolvedLocks(version);
240         return doSendBatchGetWithRefetchRegion(backOffer, batch, version);
241       }
242     } else {
243       return doSendBatchGetWithRefetchRegion(backOffer, batch, version);
244     }
245   }
246 
247   private List<Kvrpcpb.KvPair> doSendBatchGetWithRefetchRegion(
248       BackOffer backOffer, Batch batch, long version) {
249     List<Batch> retryBatches =
250         getBatches(backOffer, batch.getKeys(), BATCH_GET_SIZE, MAX_BATCH_LIMIT, this.clientBuilder);
251 
252     ArrayList<Kvrpcpb.KvPair> results = new ArrayList<>();
253     for (Batch retryBatch : retryBatches) {
254       // recursive calls
255       List<Kvrpcpb.KvPair> batchResult =
256           doSendBatchGetInBatchesWithRetry(backOffer, retryBatch, version);
257       results.addAll(batchResult);
258     }
259     return results;
260   }

280   private void doIngest(TiRegion region, List<Pair<ByteString, ByteString>> sortedList)
281       throws GrpcException {
282     if (sortedList.isEmpty()) {
283       return;
284     }
285 
286     ByteString uuid = ByteString.copyFrom(genUUID());
287     Key minKey = Key.toRawKey(sortedList.get(0).first);
288     Key maxKey = Key.toRawKey(sortedList.get(sortedList.size() - 1).first);
289     ImporterClient importerClient = new ImporterClient(tiSession, uuid, minKey, maxKey, region, 0L);
290     importerClient.write(sortedList.iterator());
291   }
